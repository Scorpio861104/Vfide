/* ─────────── Minimal VFIDE interfaces ─────────── */
 
interface ISeerView {
    function score(address user) external view returns (uint16); // 0..1000
    function isBlocked(address user) external view returns (bool);
}
 
interface IProofLedger {
    function logEvent(address user, uint8 etype, bytes32 ctx) external;
}
 
interface IEmergencyBreaker {
    function isSystemPaused() external view returns (bool);
}
 
/**
 * Vault factory view: resolve a user's vault address.
 * User funds live in vaults (not wallets), so eligibility checks use the vault balance.
 */
interface IVaultFactoryView {
    function vaultOf(address user) external view returns (address);
}
 
/** Minimal VFIDE token balance interface (for vault balances). */
interface IVFIDE {
    function balanceOf(address a) external view returns (uint256);
}
 
/* ───────────────────────── CouncilElection (ProofScore-Weighted, Seer-Verified) ─────────────────────────
   Principles:
   - Wealth ≠ power. Votes are weighted ONLY by ProofScore (0..1000).
   - Voter eligibility requires: active vault + small minimum balance (eligibility only).
   - Candidate eligibility requires: active vault + Seer score threshold + not blocked + no consecutive terms.
   - Single-choice ballot per voter.
   - 1-year terms; no back-to-back service.
 
   Lifecycle:
     1) Governance starts an election window: set times, council size (bounded), and clears previous state.
     2) Candidates self-nominate (if eligible).
     3) Eligible voters cast exactly one vote for one candidate (weight = ProofScore).
     4) Finalize: top N candidates become the new Council; results stored & logged.
 
   Notes:
   - EmergencyBreaker guard blocks mutations when system is paused.
   - ProofLedger logs key actions (best effort).
   - Sorting winners is O(n^2) on-chain; capped by maxCandidates to stay safe on gas.
---------------------------------------------------------------------------------------------------------- */
contract CouncilElection is Ownable2Step, ReentrancyGuard {
    /* ───────── Wiring ───────── */
    ISeerView public seer;
    IVaultFactoryView public factory;
    IVFIDE public vfide;
    IEmergencyBreaker public breaker;   // optional
    IProofLedger public proof;          // optional
 
    /* ───────── Config ───────── */
    uint64  public termLength = 365 days;
    uint8   public councilSize = 9;           // seats
    uint8   public maxCandidates = 200;       // safety bound for gas
 
    // Voter eligibility (vault min), not used as weight
    uint256 public minVoterBalance = 1 ether; // default 1 VFIDE (adjust to actual decimals)
 
    // Candidate gates
    uint16  public minCandidateScore = 600;   // ProofScore threshold
    bool    public requireActiveVaultForCandidate = true;
 
    // Election window
    uint64  public nominationsOpenAt;
    uint64  public nominationsCloseAt;
    uint64  public votingOpenAt;
    uint64  public votingCloseAt;
 
    /* ───────── State ───────── */
 
    // Current council roster
    address[] public council;
    mapping(address => uint64) public ineligibleUntilTermEnd; // disallow consecutive terms
 
    // Candidates & votes
    address[] public candidates;
    mapping(address => bool) public isCandidate;
    mapping(address => uint256) public votesFor;  // ProofScore-weighted tallies
    mapping(address => bool) public hasVoted;     // one ballot per voter
 
    bool public finalized; // election finalized flag
    uint64 public councilTermStart; // when the current term started
 
    /* ───────── Events ───────── */
    event WiringSet(address seer, address factory, address vfide, address breaker, address proof);
    event ConfigUpdated(uint64 termLength, uint8 councilSize, uint8 maxCandidates, uint256 minVoterBalance, uint16 minCandidateScore);
    event WindowSet(uint64 nomOpen, uint64 nomClose, uint64 voteOpen, uint64 voteClose);
 
    event Nominated(address indexed candidate);
    event VoteCast(address indexed voter, address indexed candidate, uint256 weight);
    event Finalized(address[] winners, uint64 termStart, uint64 termEnd);
 
    /* ───────── Errors ───────── */
    error Paused();
    error Window();
    error Gate();
    error TooManyCandidates();
    error AlreadyCandidate();
    error NotCandidate();
    error AlreadyVoted();
    error FinalizedError();
    error ZeroAddress();
 
    /* ───────── Modifiers ───────── */
    modifier breakerGuard() {
        if (address(breaker) != address(0) && breaker.isSystemPaused()) revert Paused();
        _;
    }

    function setWiring(address seer_, address factory_, address vfide_, address breaker_, address proof_) external onlyOwner {
        if (seer_ == address(0) || factory_ == address(0) || vfide_ == address(0)) revert ZeroAddress();
        seer    = ISeerView(seer_);
        factory = IVaultFactoryView(factory_);
        vfide   = IVFIDE(vfide_);
        breaker = IEmergencyBreaker(breaker_);
        proof   = IProofLedger(proof_);
        emit WiringSet(seer_, factory_, vfide_, breaker_, proof_);
    }
 
    function setConfig(
        uint64 termLength_,
        uint8 councilSize_,
        uint8 maxCandidates_,
        uint256 minVoterBalance_,
        uint16 minCandidateScore_,
        bool requireActiveVaultForCandidate_
    ) external onlyOwner {
        require(councilSize_ >= 3 && councilSize_ <= 25, "bad size");
        require(maxCandidates_ >= councilSize_ && maxCandidates_ <= 500, "bad max");
        require(minCandidateScore_ <= 1000, "score>1000");
        termLength = termLength_;
        councilSize = councilSize_;
        maxCandidates = maxCandidates_;
        minVoterBalance = minVoterBalance_;
        minCandidateScore = minCandidateScore_;
        requireActiveVaultForCandidate = requireActiveVaultForCandidate_;
        emit ConfigUpdated(termLength, councilSize, maxCandidates, minVoterBalance, minCandidateScore);
    }
 
    function setWindow(
        uint64 nomOpen,
        uint64 nomClose,
        uint64 voteOpen,
        uint64 voteClose
    ) external onlyOwner {
        require(nomOpen < nomClose && voteOpen < voteClose && nomClose <= voteOpen, "bad window");
        nominationsOpenAt = nomOpen;
        nominationsCloseAt = nomClose;
        votingOpenAt = voteOpen;
        votingCloseAt = voteClose;
        finalized = false;
        // clear previous election tallies/candidates is caller's responsibility via new instance or off-chain conventions
        emit WindowSet(nomOpen, nomClose, voteOpen, voteClose);
    }
 
    /* ───────── Helpers ───────── */
 
    function _now() internal view returns (uint64) {
        return uint64(block.timestamp);
    }
 
    function _hasActiveVault(address user) internal view returns (bool ok, address vault) {
        vault = factory.vaultOf(user);
        ok = (vault != address(0));
    }
 
    function _vaultBalance(address user) internal view returns (uint256) {
        (, address vault) = _hasActiveVault(user);
        if (vault == address(0)) return 0;
        return vfide.balanceOf(vault);
    }
 
    /* ───────── Candidate Nomination ───────── */
 
    function nominate() external breakerGuard nonReentrant {
        if (_now() < nominationsOpenAt || _now() > nominationsCloseAt) revert Window();
        if (isCandidate[msg.sender]) revert AlreadyCandidate();
 
        // Seer gates for candidates
        if (seer.isBlocked(msg.sender)) revert Gate();
        uint16 s = seer.score(msg.sender);
        if (s < minCandidateScore) revert Gate();
 
        // Active vault requirement (optional)
        if (requireActiveVaultForCandidate) {
            (bool ok,) = _hasActiveVault(msg.sender);
            if (!ok) revert Gate();
        }
 
        // No consecutive terms
        if (ineligibleUntilTermEnd[msg.sender] > _now()) revert Gate();
 
        // Register candidate
        if (candidates.length >= maxCandidates) revert TooManyCandidates();
        isCandidate[msg.sender] = true;
        candidates.push(msg.sender);
 
        emit Nominated(msg.sender);
        _log(240 /*Nominated*/, keccak256(abi.encode(msg.sender)));
    }
 
    /* ───────── Voting (single-choice; weight = ProofScore) ───────── */
 
    function vote(address candidate) external breakerGuard nonReentrant {
        if (_now() < votingOpenAt || _now() > votingCloseAt) revert Window();
        if (hasVoted[msg.sender]) revert AlreadyVoted();
        if (!isCandidate[candidate]) revert NotCandidate();
 
        // Voter eligibility: active vault + minimum balance (eligibility only)
        (bool ok,) = _hasActiveVault(msg.sender);
        if (!ok) revert Gate();
        if (_vaultBalance(msg.sender) < minVoterBalance) revert Gate();
 
        // Blocked voters cannot vote
        if (seer.isBlocked(msg.sender)) revert Gate();
 
        // Weight purely by ProofScore
        uint256 weight = uint256(seer.score(msg.sender)); // 0..1000
        require(weight > 0, "zero weight");
 
        hasVoted[msg.sender] = true;
        votesFor[candidate] += weight;
 
        emit VoteCast(msg.sender, candidate, weight);
        _log(241 /*VoteCast*/, keccak256(abi.encode(msg.sender, candidate, weight)));
    }
 
    /* ───────── Finalization ───────── */
 
    function finalize() external breakerGuard nonReentrant {
        if (_now() <= votingCloseAt) revert Window();
    if (finalized) revert FinalizedError();
 
        // Select top N
        uint256 n = candidates.length;
        require(n > 0, "no candidates");
 
        // Simple O(n^2) selection; bounded by maxCandidates
        address[] memory pool = new address[](n);
        for (uint256 i=0; i<n; i++) pool[i] = candidates[i];
 
        // sort by votes desc (naive)
        for (uint256 i=0; i<n; i++) {
            for (uint256 j=i+1; j<n; j++) {
                if (votesFor[pool[j]] > votesFor[pool[i]]) {
                    address tmp = pool[i]; pool[i] = pool[j]; pool[j] = tmp;
                }
            }
        }
 
        // build winners (skip ineligible that sneaked between nomination and finalization)
        address[] memory winners = new address[](councilSize);
        uint256 filled = 0;
        for (uint256 k=0; k<n && filled < councilSize; k++) {
            address cand = pool[k];
            // Re-check Seer block at finalize; replace if blocked now
            if (seer.isBlocked(cand)) continue;
            winners[filled++] = cand;
        }
        require(filled == councilSize, "insufficient winners");
 
        // Set council and mark them ineligible until term end (prevent consecutive terms)
        delete council;
        for (uint256 m=0; m<filled; m++) {
            council.push(winners[m]);
            ineligibleUntilTermEnd[winners[m]] = _now() + termLength;
        }
 
        councilTermStart = _now();
        finalized = true;
 
        emit Finalized(council, councilTermStart, councilTermStart + termLength);
        _log(242 /*ElectionFinalized*/, keccak256(abi.encode(council, councilTermStart)));
    }
 
    /* ───────── Views ───────── */
 
    function councilMembers() external view returns (address[] memory) { return council; }
    function candidateCount() external view returns (uint256) { return candidates.length; }
    function candidateAt(uint256 idx) external view returns (address) { return candidates[idx]; }
 
    /* ───────── Internal logging ───────── */
 
    function _log(uint8 etype, bytes32 ctx) internal {
        if (address(proof) != address(0)) {
            try proof.logEvent(address(this), etype, ctx) {} catch {}
        }
    }
