/* -------------------------------------------------------------------------
 * DevReserveVestingVault (Linear vesting: start + cliff + duration)
 *
 * - Irrevocable linear vesting after schedule set (owner/DAO can optionally lock it).
 * - No pulling from users; it only disperses this contract's VFIDE to the beneficiary.
 * - Optional EmergencyBreaker: claim() reverts when system paused.
 * - Optional ProofLedger: logs schedule/beneficiary/claim events (best effort).
 *
 * Typical flow:
 *   1) Deploy with initialOwner and token address.
 *   2) Transfer VFIDE into this vault externally.
 *   3) setTotalAllocation(...) once (must be <= current balance).
 *   4) setBeneficiary(...).
 *   5) setSchedule(start, cliffSecs, durationSecs)  OR  setScheduleToPresale(presaleStartTs).
 *   6) (optional) lockSchedule(), lockBeneficiary().
 *   7) Beneficiary calls claim() as time passes.
 *
 * Defaults (per VFIDE plan):
 *   - startTs = presale launch timestamp (T0)
 *   - cliff   = 90 days (3 months) AFTER T0 — nothing can be claimed before this
 *   - duration= (3 months cliff) + 36 months linear = ~39 months total from T0
 *     i.e., linear vesting runs AFTER the cliff, for 36 months.
 * ------------------------------------------------------------------------- */
contract DevReserveVestingVault is Ownable2Step, ReentrancyGuard {
    /* ---------- Default schedule constants (can be changed via setters before locking) ---------- */
    uint64 public constant DEFAULT_CLIFF_SECS = 90 days;         // 3 months after T0
    uint64 public constant DEFAULT_TOTAL_DURATION_SECS = 39 * 30 days; // ~39 months total (3 cliff + 36 linear)
 
    /* ---------- Wiring ---------- */
    IVFIDE public vfide;
    IProofLedger public proof;          // optional
    IEmergencyBreaker public breaker;   // optional
    address public dao;                 // post-handover governance (optional)
 
    /* ---------- Vesting state ---------- */
    address public beneficiary;         // recipient of vested tokens
 
    uint64  public startTs;             // vesting start (unix seconds) — set to presale T0 in defaults
    uint64  public cliffSecs;           // cliff length in seconds
    uint64  public durationSecs;        // total vesting duration from start (includes cliff)
 
    uint256 public totalAllocation;     // total tokens to vest linearly
    uint256 public released;            // amount already claimed
 
    bool public scheduleLocked;         // prevents changing schedule
    bool public beneficiaryLocked;      // prevents changing beneficiary
 
    /* ---------- Events ---------- */
    event DAOUpdated(address dao);
    event VFIDESet(address token);
    event ProofSinkSet(address proof);
    event BreakerSet(address breaker);
 
    event BeneficiarySet(address indexed beneficiary);
    event BeneficiaryLocked();
 
    event ScheduleSet(uint64 startTs, uint64 cliffSecs, uint64 durationSecs);
    event ScheduleLocked();
 
    event AllocationSet(uint256 totalAllocation);
    event AllocationIncreased(uint256 newAllocation);
 
    event Claimed(address indexed beneficiary, uint256 amount, uint256 cumulativeReleased);
    event SweptExcess(address indexed to, uint256 amount);
 
    /* ---------- Errors ---------- */
    error ZeroAddress();
    error NotGov();
    error Paused();
    error Locked();
    error BadParams();
    error NotEnoughBalance();
    error NothingToClaim();
 
    /* ---------- Modifiers ---------- */
    modifier onlyGov() {
        if (msg.sender != owner() && msg.sender != dao) revert NotGov();
        _;
    }

    /* ---------- Governance wiring ---------- */
 
    function setDAO(address dao_) external onlyGov {
        if (dao_ == address(0)) revert ZeroAddress();
        dao = dao_;
        emit DAOUpdated(dao_);
    }
 
    function setProofSink(address p_) external onlyGov {
        proof = IProofLedger(p_);
        emit ProofSinkSet(p_);
    }
 
    function setBreaker(address b_) external onlyGov {
        breaker = IEmergencyBreaker(b_);
        emit BreakerSet(b_);
    }
 
    function setVFIDE(address token) external onlyGov {
        if (token == address(0)) revert ZeroAddress();
        vfide = IVFIDE(token);
        emit VFIDESet(token);
    }
 
    /* ---------- One-time configuration ---------- */
 
    function setBeneficiary(address b) external onlyGov {
        if (beneficiaryLocked) revert Locked();
        if (b == address(0)) revert ZeroAddress();
        beneficiary = b;
        emit BeneficiarySet(b);
        _log(201 /*BeneficiarySet*/, keccak256(abi.encode(b)));
    }
 
    function lockBeneficiary() external onlyGov {
        beneficiaryLocked = true;
        emit BeneficiaryLocked();
    }
 
    /// @notice Convenience: set schedule to the VFIDE defaults with T0 = presaleStartTs.
    /// @dev     cliff = 90 days; total duration = ~39 months (3 cliff + 36 linear).
    function setScheduleToPresale(uint64 presaleStartTs) external onlyGov {
        setSchedule(presaleStartTs, DEFAULT_CLIFF_SECS, DEFAULT_TOTAL_DURATION_SECS);
    }
 
    /// @notice Set the vesting schedule (start + cliff + duration). Can be done once; then lock.
    /// @dev    durationSecs is TOTAL duration from start (i.e., includes the cliff interval).
    function setSchedule(uint64 start_, uint64 cliff_, uint64 duration_) public onlyGov {
        if (scheduleLocked) revert Locked();
        if (duration_ == 0 || start_ == 0) revert BadParams();
        // Require cliff <= duration
        require(cliff_ <= duration_, "cliff>duration");
 
        startTs      = start_;
        cliffSecs    = cliff_;
        durationSecs = duration_;
        emit ScheduleSet(start_, cliff_, duration_);
        _log(202 /*ScheduleSet*/, keccak256(abi.encode(start_, cliff_, duration_)));
    }
 
    function lockSchedule() external onlyGov {
        scheduleLocked = true;
        emit ScheduleLocked();
    }
 
    /// @notice Sets the total amount to vest (must be <= current token balance).
    function setTotalAllocation(uint256 amount) external onlyGov {
        require(released == 0, "already started");
        require(vfide.balanceOf(address(this)) >= amount, "balance < amount");
        totalAllocation = amount;
        emit AllocationSet(amount);
        _log(203 /*AllocationSet*/, keccak256(abi.encode(amount)));
    }
 
    /// @notice Increase allocation if more tokens are funded later (cannot decrease).
    function adjustTotalAllocation(uint256 newAmount) external onlyGov {
        require(newAmount >= totalAllocation, "cannot decrease");
        require(vfide.balanceOf(address(this)) >= newAmount, "balance < new");
        totalAllocation = newAmount;
        emit AllocationIncreased(newAmount);
        _log(204 /*AllocationIncreased*/, keccak256(abi.encode(newAmount)));
    }
 
    /* ---------- Claiming ---------- */
 
    function claim() external nonReentrant {
        _revertIfPaused();
        address to = beneficiary;
        if (to == address(0)) revert ZeroAddress();
 
        uint256 claimable = claimableNow();
        if (claimable == 0) revert NothingToClaim();
 
        released += claimable;
 
        // transfer
        bool ok = vfide.transfer(to, claimable);
        require(ok, "transfer failed");
 
        emit Claimed(to, claimable, released);
        _log(205 /*Claimed*/, keccak256(abi.encode(to, claimable, released)));
    }
 
    /* ---------- Views ---------- */
 
    /// @notice Returns amount vested at a given timestamp.
    /// @dev    0 before cliff; 100% after (start + duration). Linear between.
    function vestedAt(uint64 ts) public view returns (uint256) {
        if (totalAllocation == 0 || startTs == 0 || durationSecs == 0) return 0;
 
        // Before cliff: 0
        if (ts < startTs + cliffSecs) return 0;
 
        // After full duration: 100%
        if (ts >= startTs + durationSecs) return totalAllocation;
 
        // Linear vest from start over duration
        uint256 elapsed = ts - startTs;
        return (totalAllocation * elapsed) / durationSecs;
    }
 
    function vestedNow() public view returns (uint256) {
        return vestedAt(uint64(block.timestamp));
    }
 
    function claimableNow() public view returns (uint256) {
        uint256 v = vestedNow();
        if (v <= released) return 0;
        uint256 due = v - released;
 
        // Also cannot exceed current balance (handles edge cases)
        uint256 bal = vfide.balanceOf(address(this));
        if (due > bal) return bal;
        return due;
    }
 
    /* ---------- Post-vesting housekeeping ---------- */
 
    /// @notice After full vest and full release, DAO can sweep any leftover dust.
    function sweepExcess(address to) external onlyGov nonReentrant {
        if (to == address(0)) revert ZeroAddress();
        // Full vest & claimed
        require(vestedNow() == totalAllocation && released == totalAllocation, "not fully released");
        uint256 bal = vfide.balanceOf(address(this));
        if (bal == 0) return;
        bool ok = vfide.transfer(to, bal);
        require(ok, "transfer failed");
        emit SweptExcess(to, bal);
        _log(206 /*SweptExcess*/, keccak256(abi.encode(to, bal)));
    }
 
    /* ---------- Internals ---------- */
 
    function _revertIfPaused() internal view {
        if (address(breaker) != address(0)) {
            if (breaker.isSystemPaused()) revert Paused();
            // Optionally, check a dedicated vesting module id if you plan to use one:
            // require(!breaker.isModulePaused(6), "vesting paused");
        }
    }
 
    function _log(uint8 etype, bytes32 ctx) internal {
        if (address(proof) != address(0)) {
            try proof.logEvent(address(this), etype, ctx) {} catch {}
        }
    }
