contract Seer is Ownable2Step {

    // Allowlist for system actors (routers, AMMs, custodians) at the VAULT level

    function setThresholds(

        uint16 riskMedMin_,
        uint16 riskHighMin_,
        uint16 minVoteScore_,
        uint16 minDAOScore_,
        uint16 maintainScore_

        require(riskMedMin_ >= riskHighMin_, "SEER:med<high");
        require(riskHighMin_ >= 1, "SEER:high<1");
        require(

            riskMedMin_    <= MAX_SCORE &&
            riskHighMin_   <= MAX_SCORE &&
            minVoteScore_  <= MAX_SCORE &&
            minDAOScore_   <= MAX_SCORE &&
            maintainScore_ <= MAX_SCORE,
            "SEER:>max"

        require(minVoteScore_ <= minDAOScore_, "SEER:vote>dao");
        require(maintainScore_ <= minDAOScore_, "SEER:maint>dao");
 
        riskLowMin    = riskLowMin_;
        riskMedMin    = riskMedMin_;
        riskHighMin   = riskHighMin_;
        minVoteScore  = minVoteScore_;
        minDAOScore   = minDAOScore_;
        maintainScore = maintainScore_;
 
        emit ThresholdsUpdated(riskLowMin, riskMedMin, riskHighMin, minVoteScore, minDAOScore, maintainScore);
        _log(ProofEvents.EventType.SeerThresholdsUpdated, address(0), bytes32(0));
    }

    /* ───── Nominations & membership ───── */
 
    /// @notice Nominate a candidate for DAO membership (anyone can nominate).
    function nominate(address candidate) external {
        require(candidate != address(0), "SEER:cand=0");
        require(block.timestamp >= lastNominateAt[msg.sender] + nominationCooldown, "SEER:cooldown");
        require(_score(candidate) >= minDAOScore, "SEER:score<dao");
 
        lastNominateAt[msg.sender] = uint64(block.timestamp);
        if (_nominated.add(candidate)) {
            emit Nominated(msg.sender, candidate);
            _log(ProofEvents.EventType.SeerNominated, candidate, keccak256(abi.encode(msg.sender, candidate)));
        }
    }
 
    /// @notice Withdraw a nomination (governance only; prevents arbitrary users from clearing nominations).
    function withdrawNomination(address candidate) external onlyGov {
        require(candidate != address(0), "SEER:cand=0");
        if (_nominated.remove(candidate)) {
            emit NominationWithdrawn(msg.sender, candidate);
            _log(ProofEvents.EventType.SeerNominationWithdrawn, candidate, keccak256(abi.encode(msg.sender, candidate)));
        }
    }
 
    /// @notice Approve a nominated candidate as DAO member (governance).
    function approveMember(address account) external onlyGov {
        require(account != address(0), "SEER:acc=0");
        require(_nominated.contains(account), "SEER:not-nominated");
        require(_score(account) >= minDAOScore, "SEER:score<dao");
        _nominated.remove(account);
        if (_members.add(account)) {
            emit ApprovedMember(msg.sender, account);
            _log(ProofEvents.EventType.SeerApprovedMember, account, keccak256(abi.encode(account)));
        }
    }
 
    /// @notice Governance or anyone can checkpoint a member to auto-revoke if score fell below maintainScore.
    function checkpoint(address account) public {
        require(account != address(0), "SEER:acc=0");
        if (!_members.contains(account)) return;
 
        uint16 s = _score(account);
        if (s < maintainScore && block.timestamp >= lastRevokeAt[account] + revokeCooldown) {
            lastRevokeAt[account] = uint64(block.timestamp);
            _members.remove(account);
            emit RevokedMember(msg.sender, account);
            _log(ProofEvents.EventType.SeerRevokedMember, account, keccak256(abi.encode(s)));
        }
    }
 
    function checkpointMany(address[] calldata accounts) external {
        for (uint256 i = 0; i < accounts.length; i++) checkpoint(accounts[i]);
    }
 
    /* ───── RiskGate: transfer check (view) ─────
       VFIDEToken (or any integrator) can call this before allowing a transfer.
       Returns (ok, reason) so callers can choose revert messaging. */
 
    function isTransferAllowed(address vaultFrom, address vaultTo, uint256 /*amount*/)
        external
        view
        returns (bool ok, string memory reason)
    {
        if (!riskGateEnabled) return (true, "");
 
        // Factory must be set to resolve vault ownership
        if (address(factory) == address(0)) return (false, "risk:factory-unset");
 
        // basic vault checks (Token also does this)
        if (!factory.isVault(vaultFrom)) return (false, "risk:from-not-vault");
        if (!factory.isVault(vaultTo))   return (false, "risk:to-not-vault");
 
        if (riskExemptVault[vaultFrom] || riskExemptVault[vaultTo]) return (true, "");
 
        address fromOwner = factory.ownerOfVault(vaultFrom);
        address toOwner   = factory.ownerOfVault(vaultTo);
        if (fromOwner == address(0) || toOwner == address(0)) return (false, "risk:owner-missing");
 
        uint16 sFrom = _score(fromOwner);
        uint16 sTo   = _score(toOwner);
 
        // blocked rule
        if (blockIfBlocked) {
            if (sFrom < riskHighMin) return (false, "risk:sender-blocked");
            if (sTo   < riskHighMin) return (false, "risk:receiver-blocked");
        }
        // minimums
        if (enforceSenderMin && sFrom < minSenderScore)     return (false, "risk:sender-min");
        if (enforceReceiverMin && sTo < minReceiverScore)   return (false, "risk:receiver-min");
 
        return (true, "");
    }
 
    /* ───── GuardianLock signaling ───── */
 
    /// @notice Seer can proactively flag a user (reason is off-chain cataloged).
    function flag(address user, uint8 reason) external onlyGov {
        uint16 s = _score(user);
        emit RiskFlag(user, reason, s);
        if (address(guardianLock) != address(0)) {
            // best-effort; GuardianLock implementors should auth msg.sender as Seer if desired
            try guardianLock.flagFromSeer(user, reason) { } catch { }
        }
    }

    function score(address user) external view returns (uint16) { return _score(user); }
 
    function riskOf(address user) external view returns (Risk) {
        uint16 s = _score(user);
        if (s >= riskLowMin)  return Risk.Low;
        if (s >= riskMedMin)  return Risk.Medium;
        if (s >= riskHighMin) return Risk.High;
        return Risk.Blocked;
    }

    function isEligibleToVote(address user) external view returns (bool) { return _score(user) >= minVoteScore; }
    function isEligibleForDAO(address user) external view returns (bool) { return _score(user) >= minDAOScore; }
