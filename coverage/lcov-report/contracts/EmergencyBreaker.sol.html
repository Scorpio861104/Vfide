// SPDX-License-Identifier: MIT
pragma solidity 0.8.30;
 
import "@openzeppelin/contracts/access/Ownable2Step.sol";
import "@openzeppelin/contracts/security/ReentrancyGuard.sol";
 
/* ───────── Minimal VFIDE interfaces ───────── */
 
interface IProofLedger {
    function logEvent(address user, uint8 etype, bytes32 ctx) external;
}
 
/**
 * Optional Seer health interface.
 * If unset, Seer health checks are skipped.
 */
interface ISeerHealth {
    function systemHealthy() external view returns (bool);
    function isBlocked(address user) external view returns (bool);
    function score(address user) external view returns (uint16); // 0..1000
}
 
/* ────────────────────────────── EmergencyBreaker (Multi-Layer) ──────────────────────────────
   Features:
   - System-wide breaker (global pause) + module flags.
   - Trusted auto-triggers (PanicGuard, GuardianLock, Seer, etc.) can pause globally or modules.
   - Cooldown before clearing a global pause; Seer health gate on resume (optional).
   - DAO/Owner control; ProofLedger logging (best effort).
   - Lightweight view helpers for all modules to query pause state.
 
   Usage in other modules:
     require(!breaker.isSystemPaused(), "sys paused");
     require(!breaker.isModulePaused(EmergencyBreaker.Module.Token), "token paused");
------------------------------------------------------------------------------------------------ */
contract EmergencyBreaker is Ownable2Step, ReentrancyGuard {
    /* ───────── Governance / wiring ───────── */
 
    address public dao;                // DAO executor after handover
    IProofLedger public proof;         // optional audit sink
    ISeerHealth public seer;           // optional health source
 
    event DAOUpdated(address indexed dao);
    event ProofSinkSet(address indexed proof);
    event SeerSet(address indexed seer);
 
    error NotGov();
    error ZeroAddress();
 
    modifier onlyGov() {
        if (msg.sender != owner() && msg.sender != dao) revert NotGov();
        _;
    }
 
    constructor(address initialOwner) {
        if (initialOwner == address(0)) revert ZeroAddress();
        _transferOwnership(initialOwner);
    }
 
    function setDAO(address dao_) external onlyGov {
        if (dao_ == address(0)) revert ZeroAddress();
        dao = dao_;
        emit DAOUpdated(dao_);
    }
 
    function setProofSink(address p_) external onlyGov {
        proof = IProofLedger(p_); // 0 allowed (disable)
        emit ProofSinkSet(p_);
    }
 
    function setSeer(address s_) external onlyGov {
        seer = ISeerHealth(s_); // 0 allowed (disable checks)
        emit SeerSet(s_);
    }
 
    /* ───────── Modules & flags ───────── */
 
    enum Module {
        Core,        // 0 – general catch-all
        Token,       // 1
        Vaults,      // 2 (VaultFactory/UserVault)
        Seer,        // 3
        Merchants,   // 4 (Registry/Escrow/Portal/Gateway/Reviews)
        Presale,     // 5
        Staking,     // 6
        Router,      // 7 (BurnRouter / funding)
        Handover,    // 8 (SystemHandover/HandoverGate)
        Guardians,   // 9 (GuardianRegistry/Lock/PanicGuard)
        Misc         // 10
    }
 
    bool public globalPaused;
    mapping(uint8 => bool) public modulePaused;     // Module => paused?
 
    function isSystemPaused() external view returns (bool) { return globalPaused; }
    function isModulePaused(Module m) external view returns (bool) { return modulePaused[uint8(m)]; }
 
    /* ───────── Trusted triggers (auto-trip sources) ───────── */
 
    mapping(address => bool) public isTrustedTrigger; // PanicGuard, GuardianLock, Seer, etc.
 
    event TrustedTriggerSet(address indexed trigger, bool trusted);
 
    function setTrustedTrigger(address trigger, bool trusted) external onlyGov {
        if (trigger == address(0)) revert ZeroAddress();
        isTrustedTrigger[trigger] = trusted;
        emit TrustedTriggerSet(trigger, trusted);
    }
 
    /* ───────── Trip metadata & policy ───────── */
 
    uint64 public lastTripAt;               // last time global was tripped
    uint8  public lastTripReason;           // opaque code set by trigger
    address public lastTripBy;              // msg.sender who tripped
    bytes32 public lastTripCtx;             // compact context
 
    uint64 public cooldownSecs = 6 hours;   // minimum cooldown before resume
    uint16 public minResumeScore = 600;     // optional Seer score gate on resume (if seer set)
 
    event CooldownSet(uint64 seconds_);
    event MinResumeScoreSet(uint16 score);
 
    function setCooldown(uint64 seconds_) external onlyGov {
        require(seconds_ >= 5 minutes, "cooldown too small");
        cooldownSecs = seconds_;
        emit CooldownSet(seconds_);
    }
 
    function setMinResumeScore(uint16 score) external onlyGov {
        require(score <= 1000, "score>max");
        minResumeScore = score;
        emit MinResumeScoreSet(score);
    }
 
    /* ───────── Manual controls (DAO/Owner) ───────── */
 
    event GlobalPaused(uint8 reason, address indexed by, bytes32 ctx);
    event GlobalResumed(address indexed by);
    event ModulePaused(uint8 indexed module, uint8 reason, address indexed by, bytes32 ctx);
    event ModuleResumed(uint8 indexed module, address indexed by);
 
    function pauseGlobal(uint8 reason, bytes32 ctx) external onlyGov nonReentrant {
        _pauseGlobal(reason, ctx);
    }
 
    function resumeGlobal() external onlyGov nonReentrant {
        _resumeGlobal();
    }
 
    function pauseModule(Module m, uint8 reason, bytes32 ctx) external onlyGov nonReentrant {
        modulePaused[uint8(m)] = true;
        emit ModulePaused(uint8(m), reason, msg.sender, ctx);
        _log(msg.sender, 101 /*ModulePaused*/, keccak256(abi.encode(m, reason, ctx)));
    }
 
    function resumeModule(Module m) external onlyGov nonReentrant {
        modulePaused[uint8(m)] = false;
        emit ModuleResumed(uint8(m), msg.sender);
        _log(msg.sender, 102 /*ModuleResumed*/, keccak256(abi.encode(m)));
    }
 
    /* ───────── Auto-triggers (trusted modules) ─────────
       Severity: 0 = info (no pause), 1 = module pause, 2+ = GLOBAL pause.
       If severity==1, 'mod' must be provided.
    ------------------------------------------------------------------ */
 
    event AutoTrigger(address indexed trigger, uint8 severity, uint8 reason, uint8 module, bytes32 ctx);
 
    error NotTrusted();
 
    function autoTrip(uint8 severity, uint8 reason, Module mod, bytes32 ctx) external nonReentrant {
        if (!isTrustedTrigger[msg.sender]) revert NotTrusted();
 
        emit AutoTrigger(msg.sender, severity, reason, uint8(mod), ctx);
        _log(tx.origin, 103 /*AutoTrigger*/, keccak256(abi.encode(msg.sender, severity, reason, mod, ctx)));
 
        if (severity == 0) {
            return; // telemetry only
        } else if (severity == 1) {
            modulePaused[uint8(mod)] = true;
            emit ModulePaused(uint8(mod), reason, msg.sender, ctx);
            _log(msg.sender, 101 /*ModulePaused*/, keccak256(abi.encode(mod, reason, ctx)));
        } else {
            _pauseGlobal(reason, ctx);
        }
    }
 
    /* ───────── Resume gating ───────── */
 
    function canResumeNow() public view returns (bool ok, string memory why) {
        if (!globalPaused) return (true, "not paused");
        if (block.timestamp < lastTripAt + cooldownSecs) {
            return (false, "cooldown");
        }
        if (address(seer) != address(0)) {
            // Optional seer health: require systemHealthy() OR (dao not blocked & dao score >= min)
            bool healthy = false;
            try seer.systemHealthy() returns (bool h) { healthy = h; } catch { healthy = false; }
            if (!healthy) {
                if (dao != address(0)) {
                    try seer.isBlocked(dao) returns (bool b) { if (b) return (false, "dao blocked"); } catch {}
                    try seer.score(dao) returns (uint16 s) { if (s < minResumeScore) return (false, "dao score"); } catch {}
                }
            }
        }
        return (true, "");
    }
 
    /* ───────── Internal pause/resume helpers ───────── */
 
    function _pauseGlobal(uint8 reason, bytes32 ctx) internal {
        globalPaused = true;
        lastTripAt = uint64(block.timestamp);
        lastTripReason = reason;
        lastTripBy = msg.sender;
        lastTripCtx = ctx;
        emit GlobalPaused(reason, msg.sender, ctx);
        _log(msg.sender, 100 /*GlobalPaused*/, keccak256(abi.encode(reason, msg.sender, ctx)));
    }
 
    function _resumeGlobal() internal {
        (bool ok, string memory why) = canResumeNow();
        require(ok, why);
        globalPaused = false;
        emit GlobalResumed(msg.sender);
        _log(msg.sender, 104 /*GlobalResumed*/, keccak256(abi.encode(lastTripAt, lastTripReason)));
    }
 
    /* ───────── Logging ───────── */
 
    function _log(address user, uint8 et, bytes32 ctx) internal {
        if (address(proof) != address(0)) {
            try proof.logEvent(user, et, ctx) {} catch {}
        }
    }
}
