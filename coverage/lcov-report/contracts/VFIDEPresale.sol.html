// SPDX-License-Identifier: MIT
pragma solidity 0.8.30;
 
import "@openzeppelin/contracts/access/Ownable2Step.sol";
import "@openzeppelin/contracts/security/ReentrancyGuard.sol";
import "@openzeppelin/contracts/token/ERC20/IERC20.sol";
import "@openzeppelin/contracts/utils/cryptography/ECDSA.sol";
 
/**
 * @title VFIDEPresale (vault-routed + anti-whale + referrals + vouchers)
 * @notice Shared 75M pool (incl. referral bonuses). Three lock tiers (180d/90d/0d).
 *         All allocations & claims deliver to the buyer’s vault (auto-created).
 */
interface IProofLedgerPS {
    enum EventType { Unknown, PresaleBuy, ReferralReferrer, ReferralReferee, PresaleClaim }
    function logEvent(address user, EventType etype, bytes32 ctx) external;
}
 
interface IVaultFactoryPS {
    function ensureVault(address user) external returns (address);
    function isVault(address v) external view returns (bool);
}
 
contract VFIDEPresale is Ownable2Step, ReentrancyGuard {
    using ECDSA for bytes32;
 
    /* -------------------- Constants -------------------- */
    uint256 public constant CAP = 75_000_000 * 1e18; // total presale allocation (incl. referral bonuses)
    uint256 public constant REFERRER_BPS = 300;      // 3%
    uint256 public constant REFEREE_BPS  = 200;      // 2%
    uint256 public constant MAX_BPS      = 10_000;
 
    // Lock tiers
    uint8 public constant TIER_180D = 0;
    uint8 public constant TIER_90D  = 1;
    uint8 public constant TIER_0D   = 2;
 
    /* -------------------- Storage ---------------------- */
    IERC20 public immutable token;     // VFIDE token (pre-seeded to this contract)
    address public signer;             // trusted voucher signer
    IProofLedgerPS public proof;       // optional ProofLedger
    IVaultFactoryPS public factory;    // vault registry/creator
 
    // Global sold counter (includes referral bonuses)
    uint256 public sold;
 
    // Sum of all outstanding locked amounts (O(1) reserve guard)
    uint256 public totalLocked;
 
    // Voucher replay protection: digest => used
    mapping(bytes32 => bool) public usedVoucher;
 
    // Referral one-time guard (per referee) & mapping
    mapping(address => bool) public referralClaimed; // referee => claimed once?
    mapping(address => address) public referrerOf;   // referee => referrer
 
    // Locked allocations per user (escrow base purchase and unlock time)
    struct LockInfo {
        uint256 amount;
        uint64  unlockTime;
    }
    mapping(address => LockInfo) public locks; // track by USER; claim sends to their vault
 
    // Anti-whale caps (incl. referral bonuses)
    mapping(address => uint256) public totalReceived; // purchases + referee bonus + referrer bonuses
    uint256 public maxPerAddress = 500_000 * 1e18;    // 0 disables
    uint256 public maxPerTx      = 250_000 * 1e18;    // 0 disables
 
    /* -------------------- Events ----------------------- */
    event SignerSet(address signer);
    event ProofLedgerSet(address proof);
    event FactorySet(address factory);
    event Bought(address indexed buyer, uint256 vfideAmount, address indexed referrer);
    event Locked(address indexed user, uint256 amount, uint64 unlockTime, uint8 tier);
    event Claimed(address indexed user, uint256 amount);
    event Withdrawn(address to, uint256 amount);
    event MaxPerAddressSet(uint256 maxPerAddress);
    event MaxPerTxSet(uint256 maxPerTx);
 
    /* -------------------- Errors ----------------------- */
    error InvalidSigner();
    error VoucherExpired();
    error VoucherUsed();
    error CapExceeded();
    error ZeroAmount();
    error InvalidTier();
    error ReferralInvalid();
    error NothingToClaim();
    error NotUnlocked();
    error WhaleLimitExceeded();
 
    /* -------------------- Vouchers --------------------- */
    struct Voucher {
        address buyer;     // must match msg.sender
        uint256 amount;    // base allocation (excl. referral bonuses)
        uint8   tier;      // 0=180d, 1=90d, 2=0d
        uint64  expiry;    // unix time after which voucher invalid
        bytes32 nonce;     // unique per voucher
    }
 
    /* -------------------- Constructor ------------------ */
    constructor(address token_, address signer_, address factory_) {
        Erequire(token_ != address(0) && signer_ != address(0) && factory_ != address(0), "zero");
        token = IERC20(token_);
        signer = signer_;
        factory = IVaultFactoryPS(factory_);
        emit SignerSet(signer_);
        emit FactorySet(factory_);
    }
 
    /* -------------------- Admin ------------------------ */
    function setSigner(address signer_) external onlyOwner {
        require(signer_ != address(0), "signer=0");
        signer = signer_;
        emit SignerSet(signer_);
    }
 
    function setProofLedger(address proof_) external onlyOwner {
        proof = IProofLedgerPS(proof_);
        emit ProofLedgerSet(proof_);
    }
 
    function setFactory(address factory_) external onlyOwner {
        require(factory_ != address(0), "factory=0");
        factory = IVaultFactoryPS(factory_);
        emit FactorySet(factory_);
    }
 
    function setMaxPerAddress(uint256 max_) external onlyOwner {
        maxPerAddress = max_;
        emit MaxPerAddressSet(max_);
    }
 
    function setMaxPerTx(uint256 max_) external onlyOwner {
        maxPerTx = max_;
        emit MaxPerTxSet(max_);
    }
 
    /**
     * @notice Owner can withdraw unsold tokens (e.g., to treasury) after presale is done.
     * @dev Guard: must leave enough tokens to cover outstanding locks (totalLocked).
     */
    function withdrawUnsold(address to, uint256 amount) external onlyOwner nonReentrant {
        require(to != address(0), "to=0");
        uint256 bal = token.balanceOf(address(this));
        require(bal >= amount, "insufficient");
        require(bal - amount >= totalLocked, "insufficient reserves");
        require(token.transfer(to, amount), "transfer");
        emit Withdrawn(to, amount);
    }
 
    /* -------------------- Buy -------------------------- */
    function buy(Voucher calldata v, bytes calldata signature, address ref) external nonReentrant {
        // Basic checks
        if (v.buyer != msg.sender) revert InvalidSigner();
        if (v.amount == 0) revert ZeroAmount();
        if (v.expiry < block.timestamp) revert VoucherExpired();
        if (!_validTier(v.tier)) revert InvalidTier();
 
        // Signature & replay
        bytes32 digest = _voucherDigest(v);
        if (usedVoucher[digest]) revert VoucherUsed();
        address recovered = ECDSA.recover(digest, signature);
        if (recovered == address(0) || recovered != signer) revert InvalidSigner();
        usedVoucher[digest] = true;
 
        // Referrals
        uint256 referrerBonus = 0;
        uint256 refereeBonus  = 0;
        if (ref != address(0) && ref != msg.sender && !referralClaimed[msg.sender]) {
            if (referrerOf[ref] == msg.sender) revert ReferralInvalid(); // prevent simple loops
            referrerOf[msg.sender] = ref;
            referralClaimed[msg.sender] = true;
 
            referrerBonus = (v.amount * REFERRER_BPS) / MAX_BPS;
            refereeBonus  = (v.amount * REFEREE_BPS)  / MAX_BPS;
        } else if (ref != address(0)) {
            revert ReferralInvalid();
        }
 
        uint256 totalOut = v.amount + referrerBonus + refereeBonus;
 
        // Anti-whale caps
        if (maxPerTx != 0 && totalOut > maxPerTx) revert WhaleLimitExceeded();
        uint256 buyerNewTotal = totalReceived[msg.sender] + v.amount + refereeBonus;
        if (maxPerAddress != 0 && buyerNewTotal > maxPerAddress) revert WhaleLimitExceeded();
        if (referrerBonus > 0) {
            uint256 refNewTotal = totalReceived[ref] + referrerBonus;
            if (maxPerAddress != 0 && refNewTotal > maxPerAddress) revert WhaleLimitExceeded();
        }
 
        // Supply cap
        if (sold + totalOut > CAP) revert CapExceeded();
        sold += totalOut;
 
        // Effects: record totals
        totalReceived[msg.sender] = buyerNewTotal;
        if (referrerBonus > 0) totalReceived[ref] += referrerBonus;
 
        // Ensure/resolve vaults
        address buyerVault = factory.ensureVault(msg.sender);
        address refVault   = referrerBonus > 0 ? factory.ensureVault(ref) : address(0);
 
        // Delivery
        if (v.tier == TIER_0D) {
            _deliver(buyerVault, v.amount);
            if (referrerBonus > 0) _deliver(refVault, referrerBonus);
            if (refereeBonus  > 0) _deliver(buyerVault, refereeBonus);
        } else {
            uint64 unlockTime = uint64(block.timestamp + (v.tier == TIER_90D ? 90 days : 180 days));
            LockInfo storage L = locks[msg.sender];
            if (unlockTime > L.unlockTime) L.unlockTime = unlockTime;
            L.amount += v.amount; // only the base purchase is locked
            totalLocked += v.amount;
 
            // bonuses are immediate
            if (referrerBonus > 0) _deliver(refVault, referrerBonus);
            if (refereeBonus  > 0) _deliver(buyerVault, refereeBonus);
 
            emit Locked(msg.sender, v.amount, unlockTime, v.tier);
        }
 
        // Proof logging (best-effort)
        if (address(proof) != address(0)) {
            bytes32 ctx = keccak256(abi.encode(msg.sender, v.amount, v.tier, ref));
            _tryLog(msg.sender, IProofLedgerPS.EventType.PresaleBuy, ctx);
            if (ref != address(0)) {
                _tryLog(ref,        IProofLedgerPS.EventType.ReferralReferrer, ctx);
                _tryLog(msg.sender, IProofLedgerPS.EventType.ReferralReferee,  ctx);
            }
        }
 
        emit Bought(msg.sender, v.amount, ref);
    }
 
    /* -------------------- Claim ------------------------ */
    function claim() external nonReentrant {
        LockInfo memory L = locks[msg.sender];
        if (L.amount == 0) revert NothingToClaim();
        if (block.timestamp < L.unlockTime) revert NotUnlocked();
 
        // effects: update global locked reserve BEFORE transfer
        totalLocked -= L.amount;
        delete locks[msg.sender];
 
        address vault = factory.ensureVault(msg.sender);
        _deliver(vault, L.amount);
 
        emit Claimed(msg.sender, L.amount);
 
        if (address(proof) != address(0)) {
            bytes32 ctx = keccak256(abi.encode(msg.sender, L.amount));
            _tryLog(msg.sender, IProofLedgerPS.EventType.PresaleClaim, ctx);
        }
    }
 
    /* -------------------- Views ------------------------ */
    function remaining() external view returns (uint256) {
        return CAP - sold;
    }
 
    function poolBalance() external view returns (uint256) {
        return token.balanceOf(address(this));
    }
 
    /* -------------------- Internals -------------------- */
    function _deliver(address to, uint256 amount) internal {
        require(factory.isVault(to), "dest not vault");
        require(token.transfer(to, amount), "transfer");
    }
 
    function _voucherDigest(Voucher calldata v) internal view returns (bytes32) {
        // EIP-191 style: prefix + domain separation
        return keccak256(
            abi.encodePacked(
                "\x19Ethereum Signed Message:\n32",
                keccak256(
                    abi.encode(
                        address(this),
                        block.chainid,
                        v.buyer,
                        v.amount,
                        v.tier,
                        v.expiry,
                        v.nonce
                    )
                )
            )
        );
    }
 
    function _validTier(uint8 t) internal pure returns (bool) {
        return (t == TIER_180D || t == TIER_90D || t == TIER_0D);
    }
 
    function _tryLog(address user, IProofLedgerPS.EventType et, bytes32 ctx) internal {
        // best-effort logging, swallow errors
        try proof.logEvent(user, et, ctx) {} catch {}
    }
}
