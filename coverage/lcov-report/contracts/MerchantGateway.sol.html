/* ───────────────────────────────── MerchantGateway (Smart Settlement) ─────────────────────────────────
   Orchestrates score-aware fees + safety checks and routes purchases to CommerceEscrow.
 
   Flow:
   1) Preview (optional): returns fee bps/amount & net amount.
   2) buy(): 
      - Ask MerchantRegistry.canAcceptPayment(merchant, buyer) for safety (Seer/allow/deny/status).
      - Pull fee from buyer's vault -> sanctumSink (Gateway must be systemExempt or later be a vault).
      - Call escrow.initiate(merchant, netAmount, meta). Escrow then pulls netAmount from the buyer's vault.
   3) ProofLedger logging for auditability (best-effort).
 
   Governance:
   - Owner (pre-handover) or DAO (post) sets wiring and fee mapping params.
-------------------------------------------------------------------------------------------------------*/
contract MerchantGateway is Ownable2Step, ReentrancyGuard {
    /* ───────── Wiring ───────── */
    IVFIDELike        public vfide;
    IVaultFactoryView public factory;
    IMerchantRegistry public registry;
    ICommerceEscrow   public escrow;
    ISeerView         public seer;        // for score-aware fees and extra guardrails
    IProofLedger      public proof;       // optional audit sink
    address           public dao;         // post-handover governance
    address           public sanctumSink; // fee recipient (e.g., SanctumFund)
 
    /* ───────── Score-aware fee params ─────────
       Linear map: score ∈ [scoreLow, scoreHigh] → feeBps ∈ [feeBpsHigh, feeBpsLow]
       - Lower scores pay higher fees (discourages risky behavior, funds insurance/charity/burns)
       - Caps enforced by minFeeBps / maxFeeBps
    ---------------------------------------------------------------- */
    uint16 public constant MAX_SCORE = 1000;
    uint16 public scoreLow   = 300;
    uint16 public scoreHigh  = 900;
    uint16 public feeBpsLow  = 100;  // 1.00% fee for strong buyers
    uint16 public feeBpsHigh = 500;  // 5.00% fee for weak buyers
    uint16 public minFeeBps  = 0;    // global floor
    uint16 public maxFeeBps  = 1000; // global cap (10%)
 
    // Safety knobs
    bool   public requireSeerHealthy = true;  // if true, block if seer.isBlocked(buyer/merchant)
    uint16 public minBuyerScore      = 300;   // additional min for buyer
    uint16 public minMerchantScore   = 650;   // higher bar for merchants (aligned with MerchantRegistry defaults)
 
    /* ───────── Events ───────── */

    /// @notice Preview fee & net for a given merchant/amount with current params and buyer’s score.
    function preview(address buyer, address merchant, uint256 grossAmount)
        external
        view
        returns (
            uint16 buyerScore,
            uint16 feeBps,
            uint256 feeAmount,
            uint256 netToEscrow
        )
    {
        require(grossAmount > 0, "amount=0");
        uint16 s = address(seer) == address(0) ? 0 : seer.score(buyer);
        uint16 bps = _feeBpsFor(s);
        uint256 fee = (grossAmount * bps) / 10000;
        uint256 net = grossAmount - fee;
        return (s, bps, fee, net);
    }
 
    /* ───────── Core purchase (score-aware fee + escrow) ───────── */
    /**
     * @dev Steps:
     *  1) Safety check via MerchantRegistry.
     *  2) Compute fee from buyer’s current Seer score.
     *  3) Pull fee from buyer’s vault → sanctumSink.
     *  4) Call escrow.initiate(merchant, net, meta). Escrow pulls `net` from buyer’s vault.
     *
     * Requirements:
     *  - VFIDE, Registry, Escrow, Factory, Sanctum, Seer must be set.
     *  - Buyer’s vault must have **allowance for this Gateway** (for fee) and **Escrow** (for net).
     *  - Gateway should be `systemExempt` on VFIDEToken until you vaultize it.
     */
    function buy(address merchant, uint256 grossAmount, string calldata meta)
        external
        nonReentrant
        returns (uint256 orderId)
    {
        if (address(vfide)   == address(0)) revert VFIDEUnset();
        if (address(registry)== address(0)) revert RegistryUnset();
        if (address(escrow)  == address(0)) revert EscrowUnset();
        if (address(factory) == address(0)) revert FactoryUnset();
        if (address(seer)    == address(0)) revert SeerUnset();
        if (sanctumSink      == address(0)) revert SanctumUnset();
 
        require(grossAmount > 0, "amount=0");
 
        // Registry safety (status + allow/deny + min scores)
        (bool ok, string memory reason) = registry.canAcceptPayment(merchant, msg.sender);
        if (!ok) revert RegistryRejects(reason);
 
        // Optional extra guardrails (defense in depth)
        if (requireSeerHealthy) {
            if (seer.isBlocked(msg.sender) || seer.isBlocked(merchant)) {
                revert RegistryRejects("seer:blocked");
            }
            require(seer.score(msg.sender) >= minBuyerScore, "buyer score");
            require(seer.score(merchant)  >= minMerchantScore, "merchant score");
        }
 
        // Compute fee + net (score-aware)
        uint16 sBuyer = seer.score(msg.sender);
        uint16 bps = _feeBpsFor(sBuyer);
        uint256 fee = (grossAmount * bps) / 10000;
        uint256 net = grossAmount - fee;
        if (net == 0) revert AmountTooSmall();
 
        // Resolve buyer's vault
        address vBuyer = factory.vaultOf(msg.sender);
        require(vBuyer != address(0) && factory.isVault(vBuyer), "no buyer vault");
 
        // Pull fee → Sanctum (Gateway must be approved for at least `fee`)
        if (fee > 0) {
            require(vfide.transferFrom(vBuyer, sanctumSink, fee), "fee xfer fail");
        }
 
        // Route to escrow (escrow will pull `net` from buyer's vault)
        orderId = escrow.initiate(merchant, net, meta);
 
        emit Purchase(orderId, msg.sender, merchant, sBuyer, bps, grossAmount, fee, net);
 
        // Best-effort ledger log (attribute to buyer; merchants can also index by orderId in off-chain)
        _log(msg.sender, 60 /*GatewayPurchase*/, keccak256(abi.encode(orderId, merchant, grossAmount, fee, net)));
    }
 
    /* ───────── Internals ───────── */
 
    function _feeBpsFor(uint16 score) internal view returns (uint16) {
        // Clamp score to [0..MAX_SCORE]
        if (score > MAX_SCORE) score = MAX_SCORE;
 
        // Piecewise linear mapping across [scoreLow..scoreHigh]
        if (score <= scoreLow) {
            return _clampBps(feeBpsHigh);
        } else if (score >= scoreHigh) {
            return _clampBps(feeBpsLow);
        } else {
            // interpolate between high→low as score rises
            uint256 spanScore = uint256(scoreHigh) - uint256(scoreLow);
            uint256 spanBps   = uint256(feeBpsHigh) - uint256(feeBpsLow); // >= 0
            uint256 offset    = uint256(score) - uint256(scoreLow);
            uint256 bps = uint256(feeBpsHigh) - ((spanBps * offset) / spanScore);
            return _clampBps(uint16(bps));
        }
    }
 
    function _clampBps(uint16 bps) internal view returns (uint16) {
        if (bps < minFeeBps) return minFeeBps;
        if (bps > maxFeeBps) return maxFeeBps;
        return bps;
        }
 
    function _log(address user, uint8 etype, bytes32 ctx) internal {
        if (address(proof) != address(0)) {
            try proof.logEvent(user, etype, ctx) {} catch {}
        }
    }
