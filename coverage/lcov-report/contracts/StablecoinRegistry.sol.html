// SPDX-License-Identifier: MIT
pragma solidity 0.8.30;
 
import "@openzeppelin/contracts/utils/structs/EnumerableSet.sol";
 
/**
 * @title StablecoinRegistry
 * @notice Governance-managed allowlist of stablecoins.
 * - Dual governance: either `controller` or `dao` may operate (onlyGov).
 * - Canonical set via EnumerableSet for on-chain enumeration.
 * - Mirror mapping `allowed` retained for legacy integrations.
 */
contract StablecoinRegistry {
    using EnumerableSet for EnumerableSet.AddressSet;
 
    /* -------------------------------------------------------- *
     *                           STATE                          *
     * -------------------------------------------------------- */
    address public controller;
    address public dao;
 
    EnumerableSet.AddressSet private _set;   // canonical list
    mapping(address => bool) public allowed; // mirror for quick lookups / legacy
 
    /* -------------------------------------------------------- *
     *                           EVENTS                         *
     * -------------------------------------------------------- */
    event ControllerUpdated(address indexed controller);
    event DaoUpdated(address indexed dao);
    event StablecoinSet(address indexed token, bool allowed);
 
    /* -------------------------------------------------------- *
     *                         MODIFIERS                        *
     * -------------------------------------------------------- */
    modifier onlyGov() {
        require(msg.sender == controller || msg.sender == dao, "not gov");
        _;
    }
 
    // Kept for future controller-only actions (currently unused but harmless).
    modifier onlyController() {
        require(msg.sender == controller, "not controller");
        _;
    }
 
    /* -------------------------------------------------------- *
     *                        CONSTRUCTOR                       *
     * -------------------------------------------------------- */
    constructor(address controller_, address dao_) {
        Erequire(controller_ != address(0) && dao_ != address(0), "zero");
        controller = controller_;
        dao = dao_;
        emit ControllerUpdated(controller_);
        emit DaoUpdated(dao_);
    }
 
    /* -------------------------------------------------------- *
     *                       GOVERNANCE                         *
     * -------------------------------------------------------- */
    function setController(address controller_) external onlyGov {
        require(controller_ != address(0), "zero");
        controller = controller_;
        emit ControllerUpdated(controller_);
    }
 
    function setDAO(address dao_) external onlyGov {
        require(dao_ != address(0), "zero");
        dao = dao_;
        emit DaoUpdated(dao_);
    }
 
    /**
     * @notice Set allow status for a single token (idempotent).
     */
    function setStablecoin(address token, bool allowed_) external onlyGov {
        require(token != address(0), "zero");
 
        bool currently = allowed[token];
        if (currently == allowed_) {
            return; // no-op
        }
 
        if (allowed_) {
            // add
            if (_set.add(token)) {
                allowed[token] = true;
                emit StablecoinSet(token, true);
            } else if (!currently) {
                // heal mapping if ever out of sync
                allowed[token] = true;
                emit StablecoinSet(token, true);
            }
        } else {
            // remove
            if (_set.remove(token)) {
                allowed[token] = false;
                emit StablecoinSet(token, false);
            } else if (currently) {
                // heal mapping if ever out of sync
                allowed[token] = false;
                emit StablecoinSet(token, false);
            }
        }
    }
 
    /**
     * @notice Batch set allow status for multiple tokens (idempotent per token).
     * @dev Skips zero addresses and no-op entries; emits one event per effective change.
     */
    function setStablecoins(address[] calldata tokens, bool allowed_) external onlyGov {
        uint256 len = tokens.length;
        for (uint256 i = 0; i < len; i++) {
            address token = tokens[i];
            if (token == address(0)) continue;
 
            bool currently = allowed[token];
            if (currently == allowed_) continue;
 
            if (allowed_) {
                if (_set.add(token)) {
                    allowed[token] = true;
                    emit StablecoinSet(token, true);
                } else if (!currently) {
                    allowed[token] = true;
                    emit StablecoinSet(token, true);
                }
            } else {
                if (_set.remove(token)) {
                    allowed[token] = false;
                    emit StablecoinSet(token, false);
                } else if (currently) {
                    allowed[token] = false;
                    emit StablecoinSet(token, false);
                }
            }
        }
    }
 
    /* -------------------------------------------------------- *
     *                          VIEWS                           *
     * -------------------------------------------------------- */
    function stablecoinCount() external view returns (uint256) {
        return _set.length();
    }
 
    function stablecoinAt(uint256 index) external view returns (address) {
        return _set.at(index);
    }
 
    function getStablecoins() external view returns (address[] memory) {
        return _set.values();
    }
 
    /// @notice ABI sugar for integrations that expect a named getter.
    function isStablecoin(address token) external view returns (bool) {
        return allowed[token];
    }
}
