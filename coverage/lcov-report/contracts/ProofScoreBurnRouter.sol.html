    /* ================================ Immutable / Core ================================ */

    IVaultFactoryView public factory;   // resolves vaultOf(user)
    ISeerView        public seer;      // ProofScore + blocked

    // Score→BPS mapping (linear). Best score => burnMinBps; worst score => burnMaxBps.
    uint16 public constant MAX_SCORE = 1000;
    uint16 public burnMinBps     = 2000;   // 20% burn at top score (default)
    uint16 public burnMaxBps     = 9000;   // 90% burn at bottom score (default)
    uint16 public blockedBurnBps = 10000;  // 100% burn if blocked (default)
 
    // Optional hard minimums (sender/receiver score) – off by default (0 disables)
    uint16 public minSenderScoreTx   = 0;
    uint16 public minReceiverScoreTx = 0;
 
    // Allowlist for special **vault** addresses (routers, custodians) to bypass risk checks

    /* ================================ Public flow ==================================== */
    /**
     * @notice Route/burn tokens based on your current ProofScore.
     * @dev Caller must be the **controller EOA** that owns a vault, and that vault must approve this router.
     *      Pulls from the caller's vault; burns a score-weighted portion; sends the rest to Sanctum.
     *
     *      Token-level note: mark this router **systemExempt** on VFIDEToken unless you make it a vault.
     */
    function route(uint256 amount) external nonReentrant {
        if (address(factory) == address(0)) revert FactoryUnset();
        if (address(seer) == address(0)) revert SeerUnset();
        if (sanctum == address(0)) revert SanctumUnset();
        require(amount > 0, "amount=0");
 
        // Resolve caller's vault
        address user = msg.sender;
        address v = factory.vaultOf(user);
        if (v == address(0) || !factory.isVault(v)) revert NoVault();
 
        // Risk gate (sender focus)
        if (riskGateEnabled && !riskExemptVault[v]) {
            if (blockIfBlocked && seer.isBlocked(user)) revert RiskRejected("sender blocked");
            if (minSenderScoreTx > 0) {
                uint16 s0 = seer.score(user);
                if (s0 < minSenderScoreTx) revert RiskRejected("sender score");
            }
        }
 
        // Pull from vault (requires prior vault -> router allowance)
        require(vfide.transferFrom(v, address(this), amount), "pull fail");
 
        // Compute split based on current score
        (uint16 s, uint16 burnBps) = _burnBpsFor(user);
        uint256 toSanctum = (amount * (10000 - burnBps)) / 10000;
        uint256 toBurn    = amount - toSanctum; // exact remainder
 
        // Interactions
        if (toSanctum > 0) {
            require(vfide.transfer(sanctum, toSanctum), "xfer sanctum");
        }
        if (toBurn > 0) {
            vfide.burn(toBurn);
        }
 
        emit Routed(user, v, s, burnBps, amount, toBurn, toSanctum);
 
        // Best-effort ledger log
        if (address(proof) != address(0)) {
            bytes32 ctx = keccak256(abi.encode(user, s, burnBps, amount, toBurn, toSanctum));
            try proof.logEvent(user, IProofLedger.EventType.BurnRouted, ctx) {} catch {}
        }
    }

    function preview(address user, uint256 amount)
        external
        view
        returns (uint16 score, uint16 burnBps, uint256 burnAmt, uint256 sanctumAmt)
    {
        if (user == address(0) || amount == 0) return (0, 0, 0, 0);
        (uint16 s, uint16 bps) = _burnBpsFor(user);
        uint256 sanctumOut = (amount * (10000 - bps)) / 10000;
        uint256 burnOut    = amount - sanctumOut;
        return (s, bps, burnOut, sanctumOut);
    }
 
    /* ================================ Internals ====================================== */
    function _burnBpsFor(address user) internal view returns (uint16 score, uint16 burnBps) {
        score = seer.score(user); // 0..1000
        if (riskGateEnabled && blockIfBlocked && seer.isBlocked(user)) {
            return (score, blockedBurnBps); // e.g., 100% burn by default
        }
        // Linear interpolation from [0..1000] → [burnMaxBps..burnMinBps]
        // s=0   => burnMaxBps
        // s=1000=> burnMinBps
        uint256 span = uint256(burnMaxBps) - uint256(burnMinBps); // >=0
        uint256 reduce = (span * score) / MAX_SCORE;               // 0..span
        uint256 bps = uint256(burnMaxBps) - reduce;
        return (score, uint16(bps));
    }
