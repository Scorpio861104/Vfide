pragma solidity 0.8.30;
 
import "@openzeppelin/contracts/access/Ownable2Step.sol";
import "@openzeppelin/contracts/security/ReentrancyGuard.sol";
 
/**
 * @title SystemHandover
 * @notice Coordinates non-revocable transfer of system ownership to a DAO
 *         after a fixed delay AND a ProofScore threshold check.
 *
 * Flow:
 *  - Deploy: owner = foundation (current admin). startTs = block.timestamp.
 *  - Owner registers system contracts (targets) to hand over.
 *  - Owner finalizes the DAO election ONCE with:
 *        (daoExecutor, daoOperator, electionHash)
 *    This seals targets and parameters (no edits afterward).
 *  - After 6 months AND if ProofScore(daoOperator) >= minDAOScore,
 *    anyone can call executeHandover() which calls transferOwnership(daoExecutor)
 *    on each target. DAO then calls acceptOwnership() on each target.
 *
 * Guarantees:
 *  - No revocation once election is finalized.
 *  - No parameter drift after finalize.
 *  - ProofScore and time-lock must both pass.
 */
 
interface IProofLedger {
    // ProofScore scale assumed 0..1000 (as used elsewhere in the system).
    function scoreOf(address user) external view returns (uint16);
}
 
interface IOwnable2StepLike {
    function transferOwnership(address newOwner) external;
}
 
contract SystemHandover is Ownable2Step, ReentrancyGuard {

                               CONFIG
    //////////////////////////////////////////////////////////////*/

    IProofLedger public immutable proof;     // ProofLedger reference (immutable)

    bytes32 public electionHash;             // off-chain record (e.g., IPFS hash) of the election result

    /// @notice Add a system contract to be handed over (before election is finalized).
    function addTarget(address target) external onlyOwner notSealed {
        if (target == address(0)) revert ZeroAddress();
        if (isTarget[target]) revert AlreadyTarget();
        isTarget[target] = true;
        _targets.push(target);
        emit TargetAdded(target);
    }
 
    /// @notice Remove a system contract from the list (before election is finalized).
    function removeTarget(address target) external onlyOwner notSealed {
        if (!isTarget[target]) revert NotTarget();
        isTarget[target] = false;
 
        // compact array (small list; O(n) is fine)
        uint256 len = _targets.length;
        for (uint256 i = 0; i < len; i++) {
            if (_targets[i] == target) {
                _targets[i] = _targets[len - 1];
                _targets.pop();
                break;
            }
        }
        emit TargetRemoved(target);
    }

     * @notice Finalize the DAO election (ONE-TIME). Seals parameters and target list.
     * @param daoExecutor_  The DAO contract that will receive ownership.
     * @param daoOperator_  An EOA representing the DAO (used for ProofScore check).
     * @param electionHash_ Off-chain record (e.g., IPFS CID hash) of election results.
     */
    function finalizeElection(address daoExecutor_, address daoOperator_, bytes32 electionHash_) external onlyOwner notSealed {
        if (daoExecutor_ == address(0) || daoOperator_ == address(0)) revert ZeroAddress();
 
        daoExecutor = daoExecutor_;
        daoOperator = daoOperator_;
        electionHash = electionHash_;
        electionFinalized = true;
 
        emit TargetsSealed(_targets.length);
        emit ElectionFinalized(daoExecutor_, daoOperator_, electionHash_);
    }

    function handoverReadyAt() public view returns (uint64) {
        return uint64(startTs + HANDOVER_DELAY);
    }
 
    /// @notice Returns whether all conditions for handover are met at the current time.
    function isHandoverReady() public view returns (bool) {
        if (!electionFinalized) return false;
        if (block.timestamp < handoverReadyAt()) return false;
        uint16 have = proof.scoreOf(daoOperator);
        if (have < minDAOScore) return false;
        return true;
    }
 
    /**
     * @notice Execute the handover transfer of ownership for all sealed targets.
     *         Callable by anyone once conditions are met. Non-revocable and one-shot.
     *         Each target must implement Ownable2Step.transferOwnership(daoExecutor).
     *         DAO will need to acceptOwnership() on each target afterward.
     */
    function executeHandover() external nonReentrant {
        if (handoverExecuted) revert AlreadyExecuted();
        if (!electionFinalized) revert NotFinalized();
        if (block.timestamp < handoverReadyAt()) revert TooSoon();
 
        uint16 have = proof.scoreOf(daoOperator);
        if (have < minDAOScore) revert LowProofScore(have, minDAOScore);
 
        // Transfer ownership of each target to the DAO executor.
        // Any failure in the list will revert the whole tx (intended: atomicity).
        for (uint256 i = 0; i < _targets.length; i++) {
            IOwnable2StepLike(_targets[i]).transferOwnership(daoExecutor);
        }
 
        handoverExecuted = true;
        emit HandoverExecuted(daoExecutor, _targets.length);
    }
