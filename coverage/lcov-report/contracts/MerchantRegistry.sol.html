/**
 * @title MerchantRegistry
 * @notice Governance-managed registry + enforcement for VFIDE merchants.
 *         - Merchants apply; governance approves.
 *         - Enforcement ON by default: Seer risk gates + auto suspension/revocation.
 *         - Exposes `canAcceptPayment(merchant, buyer)` for portals/routers to check pre-payment.
 *         - Vault-first payout model: stores merchant payout vault (synced via Factory).

    /// @notice Merchant applies. Creates or updates a Pending record.
    function merchantApply(string calldata name, bytes32 metaHash) external nonReentrant {
        address controller = msg.sender;
        Merchant storage mm = _m[controller];
 
        if (mm.status == Status.None) {
            // New applicant
            mm.controller    = controller;
            mm.status        = Status.Pending;
            mm.appliedAt     = uint64(block.timestamp);
            mm.name          = name;
            mm.metaHash      = metaHash;
 
            _all.add(controller);
            emit MerchantApplied(controller, name, metaHash);
            _log(controller, 10 /*etype: MerchantApplied*/, keccak256(abi.encode(controller, name, metaHash)));
        } else {
            // Update application metadata if not yet Active/Revoked
            require(mm.status == Status.Pending || mm.status == Status.Suspended, "bad status");
            mm.name     = name;
            mm.metaHash = metaHash;
            emit MerchantUpdated(controller, name, metaHash);
        }
    }
 
    /// @notice Governance approves a merchant. Syncs payout vault.
    function approve(address controller) external onlyGov nonReentrant {
        if (address(factory) == address(0)) revert FactoryUnset();
        if (address(seer) == address(0)) revert SeerUnset();
 
        Merchant storage mm = _mustExist(controller);
        if (mm.status != Status.Pending && mm.status != Status.Suspended) revert BadStatus();
 
        // Enforce merchant score threshold and not blocked
        uint16 s = seer.score(controller);
        if (enforcementEnabled) {
            if (blockIfBlocked) require(!seer.isBlocked(controller), "blocked");
            if (enforceMerchantMin) require(s >= minMerchantScore, "score<merchantMin");
        }
 
        // Ensure/sync payout vault
        address pv = _syncVault(controller);
        mm.status       = Status.Active;
        mm.approvedAt   = uint64(block.timestamp);
        mm.lastCheckpoint = uint64(block.timestamp);
 
        _active.add(controller);
        emit MerchantApproved(controller, pv);
        _log(controller, 11 /*etype: MerchantApproved*/, keccak256(abi.encode(controller, s, pv)));
    }
 
    /// @notice Governance can suspend an active merchant (reversible).
    function suspend(address controller, string calldata reason) external onlyGov nonReentrant {
        Merchant storage mm = _mustExist(controller);
        require(mm.status == Status.Active, "not active");
        mm.status = Status.Suspended;
        _active.remove(controller);
        emit MerchantSuspended(controller, _safeScore(controller));
        _log(controller, 12 /*etype: MerchantSuspended*/, keccak256(abi.encode(controller, reason)));
    }
 
    /// @notice Governance can revoke (final) – merchant must re-apply to return.
    function revoke(address controller, string calldata reason) external onlyGov nonReentrant {
        Merchant storage mm = _mustExist(controller);
        require(mm.status == Status.Active || mm.status == Status.Suspended || mm.status == Status.Pending, "bad status");
        mm.status = Status.Revoked;
        _active.remove(controller);
        emit MerchantRevoked(controller, _safeScore(controller));
        _log(controller, 13 /*etype: MerchantRevoked*/, keccak256(abi.encode(controller, reason)));
    }

    /// @notice Check and update a single merchant’s state based on Seer score.
    function checkpoint(address controller) public {
        if (address(seer) == address(0)) revert SeerUnset();
        Merchant storage mm = _mustExist(controller);
        uint16 s = seer.score(controller);
        mm.lastCheckpoint = uint64(block.timestamp);
 
        if (!enforcementEnabled) return;
 
        if (mm.status == Status.Active) {
            if (blockIfBlocked && seer.isBlocked(controller)) {
                mm.status = Status.Suspended;
                _active.remove(controller);
                emit MerchantSuspended(controller, s);
                _log(controller, 14 /*etype: MerchantAutoSuspended*/, keccak256(abi.encode(controller, s, "blocked")));
                return;
            }
            if (s < maintainMerchantScore) {
                mm.status = Status.Suspended;
                _active.remove(controller);
                emit MerchantSuspended(controller, s);
                _log(controller, 14 /*etype: MerchantAutoSuspended*/, keccak256(abi.encode(controller, s, "maint")));
            }
        } else if (mm.status == Status.Suspended) {
            // If merchant recovers above minMerchantScore, governance may reactivate via approve()
            if (blockIfBlocked && seer.isBlocked(controller)) {
                // optional: escalate to revoke if desired (commented):
                // mm.status = Status.Revoked;
                // _log(controller, 15 /*etype: MerchantAutoRevoked*/, keccak256(abi.encode(controller, s, "blocked-escalate")));
            }
        }
    }

    /* ───────────────────── Payment-time enforcement ───────────────── */
 
    /**
     * @notice View guard for portals/routers to check whether a payment is allowed.
     * @param merchantController merchant controller EOA
     * @param buyerController    buyer controller EOA
     * @return ok true if allowed, reason a short string reason if not
     */
    function canAcceptPayment(address merchantController, address buyerController)
        external
        view
        returns (bool ok, string memory reason)
    {
        Merchant storage mm = _m[merchantController];
        if (mm.status == Status.None)         return (false, "merchant:none");
        if (mm.status == Status.Pending)      return (false, "merchant:pending");
        if (mm.status == Status.Revoked)      return (false, "merchant:revoked");
        if (denyMerchant[merchantController]) return (false, "merchant:denied");
        if (denyBuyer[buyerController])       return (false, "buyer:denied");
 
        // Suspended merchants are blocked unless explicitly allow-listed
        if (mm.status == Status.Suspended && !allowMerchant[merchantController]) {
            return (false, "merchant:suspended");
        }
 
        // Exemptions skip score checks
        if (systemExemptUser[merchantController] || systemExemptUser[buyerController]) {
            return (true, "");
        }
 
        if (!enforcementEnabled) return (true, "");
 
        if (address(seer) == address(0)) return (false, "seer:unset");
 
        // Block "blocked" parties if enabled
        if (blockIfBlocked) {
            if (seer.isBlocked(merchantController)) return (false, "merchant:blocked");
            if (seer.isBlocked(buyerController))    return (false, "buyer:blocked");
        }
 
        // Score minimums
        if (enforceMerchantMin) {
            uint16 sm = seer.score(merchantController);
            if (sm < minMerchantScore) return (false, "merchant:score");
        }
        if (enforceBuyerMin) {
            uint16 sb = seer.score(buyerController);
            if (sb < minBuyerScore) return (false, "buyer:score");
        }
 
        return (true, "");
    }
