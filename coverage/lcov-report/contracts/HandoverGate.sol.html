/** Optional: our developer multisig that can switch to DAO mode */
interface IMultiSigDeveloperControl {

/* ───────────────────────── HandoverGate ─────────────────────────
   Final-stage controller for timed transition from developer control
   to DAO governance, with readiness + safety checks.
 
   Flow (typical):
   1) Deploy with initialOwner (dev governance).
   2) setDAOExecutor(dao), setSeer(), setProofSink(), setGovernables([...]),
      setMultisig(multisig) (optional), setHandoverStart(startTs) or use now.
   3) Optionally register a small set of DAO readiness signers (e.g., Gnosis owners)
      and setReadySignerThreshold(3). Each signer calls signalDAOReady().
   4) After `start + minDelay` (default 6 months) and readiness checks pass,
      anyone may call finalizeHandover().
   5) Gate pushes DAO to all governed modules; flips multisig to DAO mode; emits Finalized.
------------------------------------------------------------------- */
contract HandoverGate is Ownable2Step, ReentrancyGuard {
    /* ───────── Wiring ───────── */
 
    address public daoExecutor;        // the DAO executor to hand to (Gnosis Safe / Governor)
    ISeerView public seer;             // ProofScore / blocklist source
    IProofLedger public proof;         // optional audit sink
 
    // Optional developer multisig; we flip it into DAO mode at finalize()
    IMultiSigDeveloperControl public multisig;
 
    // Modules that expose setDAO(address)
    address[] public governables;
 
    /* ───────── Timing ───────── */
 
    uint64  public handoverStart;      // when the countdown begins
    uint64  public minDelay = 180 days; // default 6 months
    bool    public postponedOnce;      // single emergency extension used?
 
    /* ───────── Readiness ───────── */
 
    // DAO must have at least this ProofScore
    uint16  public minDAOScore = 600;  // sane default; adjustable
 
    // Optional DAO “ready” signal system
    mapping(address => bool) public isReadySigner;
    mapping(address => bool) public signerHasSignaled;
    uint16  public readySignerThreshold;  // e.g., 3 of N
    uint16  public readySignals;          // current count
 
    /* ───────── State ───────── */
 
    bool public finalized;              // once set, irreversible
    address public dao;                  // mirrors daoExecutor after finalize (for onlyGov)
 
    /* ───────── Events ───────── */
 
    event DAOExecutorSet(address indexed dao);
    event SeerSet(address indexed seer);
    event ProofSinkSet(address indexed proof);
    event MultisigSet(address indexed multisig);
    event GovernablesSet(address[] modules);
 
    event HandoverStartSet(uint64 startTs);
    event MinDelaySet(uint64 seconds_);
    event EmergencyPostpone(uint64 addedSeconds);
 
    event MinDAOScoreSet(uint16 score);
    event ReadySignerAdded(address indexed signer);
    event ReadySignerRemoved(address indexed signer);
    event ReadySignaled(address indexed signer);
    event ReadySignerThresholdSet(uint16 threshold);
 
    event Finalized(address indexed dao, uint64 timestamp);
 
    /* ───────── Errors ───────── */
 
    error ZeroAddress();
    error NotGov();
    error AlreadyFinalized();
    error NotEligible();
    error AlreadySignaled();
    error ThresholdTooHigh();
    error PostponeOnlyOnce();

    modifier onlyGov() {
        // Pre-handover: owner; Post-handover: dao
        if (msg.sender != owner() && msg.sender != dao) revert NotGov();
        _;
    }

    function setDAOExecutor(address dao_) external onlyGov {
        if (dao_ == address(0)) revert ZeroAddress();
        daoExecutor = dao_;
        emit DAOExecutorSet(dao_);
    }
 
    function setSeer(address seer_) external onlyGov {
        if (seer_ == address(0)) revert ZeroAddress();
        seer = ISeerView(seer_);
        emit SeerSet(seer_);
    }
 
    function setProofSink(address proof_) external onlyGov {
        proof = IProofLedger(proof_); // 0 allowed (disable)
        emit ProofSinkSet(proof_);

        emit MultisigSet(ms_);

        governables = modules;

        minDelay = seconds_;
        emit MinDelaySet(seconds_);
    }
 
    /// @notice Single-use emergency postpone; must be invoked BEFORE eligibility.
    function emergencyPostpone(uint64 addSeconds) external onlyGov {
        if (postponedOnce) revert PostponeOnlyOnce();
        if (_now() < eligibleAt()) {
            postponedOnce = true;
            minDelay += addSeconds;
            emit EmergencyPostpone(addSeconds);
            _log(msg.sender, 70 /*EmergencyPostpone*/, keccak256(abi.encode(addSeconds)));
        } else {
            // If already eligible, do nothing (no silent extensions)
            revert NotEligible();

    }

        minDAOScore = score;
        emit MinDAOScoreSet(score);
    }
 
    function addReadySigner(address a) external onlyGov {
        if (a == address(0)) revert ZeroAddress();
        isReadySigner[a] = true;
        emit ReadySignerAdded(a);
    }
 
    function removeReadySigner(address a) external onlyGov {
        isReadySigner[a] = false;
        // keep any prior signal; it no longer counts toward future thresholds
        if (signerHasSignaled[a]) {
            signerHasSignaled[a] = false;
            if (readySignals > 0) readySignals -= 1;
        }
        emit ReadySignerRemoved(a);
    }
 
    function setReadySignerThreshold(uint16 t) external onlyGov {
        // Can’t require <= number of signers on-chain cheaply; just ensure sensible range
        if (t > 1000) revert ThresholdTooHigh();
        readySignerThreshold = t;
        emit ReadySignerThresholdSet(t);
    }
 
    /// @notice Called by whitelisted readiness signers (e.g., DAO Safe owners) to signal DAO is ready.
    function signalDAOReady() external {
        address a = msg.sender;
        require(isReadySigner[a], "not signer");
        if (signerHasSignaled[a]) revert AlreadySignaled();
        signerHasSignaled[a] = true;
        readySignals += 1;
        emit ReadySignaled(a);
    }
 
    /* ───────── Eligibility + finalize ───────── */
 
    function eligibleAt() public view returns (uint64) {
        return handoverStart + minDelay;
    }
 
    function isTimeReady() public view returns (bool) {
        return _now() >= eligibleAt();
    }
 
    function isDAOScoreReady() public view returns (bool) {
        if (address(seer) == address(0) || daoExecutor == address(0)) return false;
        if (seer.isBlocked(daoExecutor)) return false;
        return seer.score(daoExecutor) >= minDAOScore;
    }
 
    function isQuorumReady() public view returns (bool) {
        // If threshold is zero, treat as disabled
        if (readySignerThreshold == 0) return true;
        return readySignals >= readySignerThreshold;
    }

        external
        view
        returns (bool timeReady, bool scoreReady, bool quorumReady, uint64 eligibleTimestamp)
    {
        return (isTimeReady(), isDAOScoreReady(), isQuorumReady(), eligibleAt());
    }
 
    /// @notice Final, one-way handover to DAO. Anyone can call once conditions are met.
    function finalizeHandover() external nonReentrant {
        if (finalized) revert AlreadyFinalized();
        if (!isTimeReady() || !isDAOScoreReady() || !isQuorumReady()) revert NotEligible();
 
        // Push DAO into all governed modules
        address dao_ = daoExecutor;
        for (uint256 i = 0; i < governables.length; i++) {
            address mod = governables[i];
            if (mod != address(0)) {
                // best-effort; if a single module reverts, we revert (safer than partial)
                IHasDAO(mod).setDAO(dao_);
            }
        }
 
        // Flip developer multisig into DAO mode (optional)
        if (address(multisig) != address(0)) {
            multisig.activateDAOMode(dao_);
        }
 
        finalized = true;
        dao = dao_; // from now on, onlyGov recognizes DAO
        emit Finalized(dao_, _now());
        _log(dao_, 71 /*HandoverFinalized*/, keccak256(abi.encode(dao_, eligibleAt(), readySignals)));
    }
 
    /* ───────── Views: list governables ───────── */
 
    function governablesCount() external view returns (uint256) {
        return governables.length;
    }
 
    /* ───────── Internals ───────── */
 
    function _now() internal view returns (uint64) {
        return uint64(block.timestamp);
    }
 
    function _log(address user, uint8 etype, bytes32 ctx) internal {
        if (address(proof) != address(0)) {
            try proof.logEvent(user, etype, ctx) {} catch {}
        }
    }
