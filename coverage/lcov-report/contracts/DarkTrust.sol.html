pragma solidity 0.8.30;
 
import "@openzeppelin/contracts/access/Ownable2Step.sol";
import "@openzeppelin/contracts/security/ReentrancyGuard.sol";
 
/* ─────────────────────────── Interfaces (loose-coupled, try/catch safe) ─────────────────────────── */
 
interface IProofLedger {
    function logEvent(address user, uint8 etype, bytes32 ctx) external;
}
 
interface ISeerView {
    function score(address user) external view returns (uint16); // 0..1000
    function isBlocked(address user) external view returns (bool);
}
 
/** Optional Seer tuning surface; if not present, calls are skipped. */
interface ISeerTuning {
    function setFlag(address user, uint8 code, bool on) external;
    function adjustScore(address user, int16 delta) external; // +/- points
    function setWatch(address user, uint64 untilTs) external; // place under watch
}
 
/** PanicGuard (hybrid) write we use to quarantine on high severity */
interface IPanicGuard {
    function setQuarantine(address user, uint64 untilTs) external;
}
 
/** EmergencyBreaker autoTrip interface (enum Module is uint8 at ABI level) */
interface IEmergencyBreaker {
    function autoTrip(uint8 severity, uint8 reason, uint8 module, bytes32 ctx) external;
}
 
/* ───────────────────────────────────────── DarkTrust ─────────────────────────────────────────
 
Role:
- Receives signals (transfers, trades, merchant events), maintains decayed meters, and classifies risk.
- On severity:
  * Low: log + (optional) light Seer score nudge.
  * Medium: Seer flag + watch window.
  * High: PanicGuard quarantine + EmergencyBreaker module pause (optional).
- Every action is ProofLedger-logged (best-effort).
 
Security:
- Owner → pre-handover, DAO → post-handover via setDAO().
- Trusted reporters can submit signals (vaults, escrow, merchant gateway). Public reporting is off by default.
 
--------------------------------------------------------------------------------------------------*/
contract DarkTrust is Ownable2Step, ReentrancyGuard {
    /* ───────── Wiring ───────── */
    address public dao;                  // post-handover governance
    IProofLedger public proof;           // optional audit sink
    ISeerView public seerView;           // read surface (scores/blocked)
    ISeerTuning public seerTune;         // optional tuning surface
    IPanicGuard public guard;            // optional quarantine
    IEmergencyBreaker public breaker;    // optional autoTrip
 
    /* ───────── Governance / Access ───────── */
    mapping(address => bool) public isTrustedReporter; // Vaults, Escrow, MerchantGateway, Portal
    bool public allowPublicReports;                    // default false
 
    modifier onlyGov() {
        if (msg.sender != owner() && msg.sender != dao) revert NotGov();
        _;
    }
    modifier onlyReporter() {
        if (!allowPublicReports && !isTrustedReporter[msg.sender]) revert NotReporter();
        _;
    }
 
    /* ───────── Risk meters (decayed counters) ─────────
       Exponential decay: value *= exp(-dt / halfLife)
       We approximate with exponent = 2^(-dt / halfLife) using fixed-point rational.
       For gas & simplicity, use a linearized clamp for big dt intervals: if dt > 10*half → zero.
    ------------------------------------------------------------------ */
    struct Meter {
        uint64 lastTs;   // last update timestamp
        uint256 value;   // decayed “units” (amount or count)
    }
 
    struct UserRisk {
        Meter amount;     // decayed total amount out
        Meter sends;      // decayed send count
        Meter roundTrips; // decayed ping-pong score
        uint64 lastToSame; // last peer address tag packed externally (simplified with salt)
        uint64 lastUpdate; // for housekeeping
        uint8 lastSeverity;
    }
    mapping(address => UserRisk) private _risk;
 
    /* ───────── Config knobs ───────── */
 
    // Half-lives (seconds) for decays
    uint64 public halfLifeAmount  = 60 minutes;
    uint64 public halfLifeCount   = 30 minutes;
    uint64 public halfLifeRound   = 90 minutes;
 
    // Thresholds for classification
    uint256 public burstAmountHi  = 1_000_000 ether; // adjust to token decimals
    uint256 public burstAmountMed =   250_000 ether;
    uint256 public burstCountHi   = 50;
    uint256 public burstCountMed  = 15;
 
    // Round-trip (wash/ping-pong) score thresholds
    uint256 public rtHi           = 40;
    uint256 public rtMed          = 15;
 
    // Seer penalties (bps of a “unit” → translated to points; or fixed deltas)
    int16   public penaltyLow     = -2;   // -2 points
    int16   public penaltyMed     = -6;
    int16   public penaltyHigh    = -15;
 
    // Quarantine & breaker behavior
    uint64  public quarantineSecsHigh = 24 hours;
    uint8   public breakerSeverityHigh = 2;   // 2+ → global, 1 → module
    uint8   public breakerReasonAbuse  = 7;   // opaque reason code for abuse
    uint8   public breakerModuleGuard  = 9;   // Guardians module id in breaker
 
    // Destination policies
    mapping(address => bool) public denyDestination;  // known scam/sink addresses
    mapping(address => bool) public allowDestination; // trusted sinks (exchanges, sanctum)
 
    // Global counters for mass-incident detection (lightweight)
    uint64 public globalHalfLife  = 30 minutes;
    Meter  public globalIncidents;
    uint256 public globalIncidentMed = 25; // if exceeded → module pause
    uint256 public globalIncidentHi  = 75; // if exceeded → global pause
 
    /* ───────── Events ───────── */
    event DAOUpdated(address dao);
    event ProofSinkSet(address proof);
    event SeerSet(address seerView, address seerTune);
    event GuardSet(address guard);
    event BreakerSet(address breaker);
 
    event ReporterSet(address indexed reporter, bool trusted);
    event PublicReportsToggled(bool allowed);
 
    event ConfigUpdated();
    event DestinationAllowSet(address indexed addr, bool allowed);
    event DestinationDenySet(address indexed addr, bool denied);
 
    event TransferSignal(address indexed owner, address indexed to, uint256 amount, uint8 severity);
    event RoundTripBump(address indexed a, address indexed b, uint256 newScoreA, uint256 newScoreB);
    event IncidentRaised(address indexed subject, uint8 severity, uint8 code);
 
    /* ───────── Errors ───────── */
    error NotGov();
    error NotReporter();
 
    /* ───────── Constructor ───────── */
    constructor(address initialOwner) {
        Erequire(initialOwner != address(0), "owner=0");
        _transferOwnership(initialOwner);
    }
 
    /* ───────── Wiring ───────── */
 
    function setDAO(address dao_) external onlyGov { dao = dao_; emit DAOUpdated(dao_); }
    function setProofSink(address p_) external onlyGov { proof = IProofLedger(p_); emit ProofSinkSet(p_); }
    function setSeer(address view_, address tune_) external onlyGov { seerView = ISeerView(view_); seerTune = ISeerTuning(tune_); emit SeerSet(view_, tune_); }
    function setGuard(address g_) external onlyGov { guard = IPanicGuard(g_); emit GuardSet(g_); }
    function setBreaker(address b_) external onlyGov { breaker = IEmergencyBreaker(b_); emit BreakerSet(b_); }
 
    function setReporter(address who, bool on) external onlyGov { isTrustedReporter[who]=on; emit ReporterSet(who,on); }
    function togglePublicReports(bool on) external onlyGov { allowPublicReports=on; emit PublicReportsToggled(on); }
 
    /* ───────── Config setters (batched for brevity) ───────── */
 
    function setDecayHalves(uint64 amount_, uint64 count_, uint64 round_, uint64 global_) external onlyGov {
        halfLifeAmount=amount_; halfLifeCount=count_; halfLifeRound=round_; globalHalfLife=global_;
        emit ConfigUpdated();
    }
    function setBurstThresholds(uint256 amtHi, uint256 amtMed, uint256 cntHi, uint256 cntMed) external onlyGov {
        burstAmountHi=amtHi; burstAmountMed=amtMed; burstCountHi=cntHi; burstCountMed=cntMed; emit ConfigUpdated();
    }
    function setRoundTripThresholds(uint256 hi, uint256 med) external onlyGov { rtHi=hi; rtMed=med; emit ConfigUpdated(); }
    function setPenalties(int16 low_, int16 med_, int16 high_) external onlyGov { penaltyLow=low_; penaltyMed=med_; penaltyHigh=high_; emit ConfigUpdated(); }
    function setQuarantineAndBreaker(uint64 qHigh, uint8 sevHigh, uint8 reasonCode, uint8 moduleId) external onlyGov {
        quarantineSecsHigh=qHigh; breakerSeverityHigh=sevHigh; breakerReasonAbuse=reasonCode; breakerModuleGuard=moduleId; emit ConfigUpdated();
    }
    function setGlobalIncidentThresholds(uint256 med_, uint256 hi_) external onlyGov { globalIncidentMed=med_; globalIncidentHi=hi_; emit ConfigUpdated(); }
    function setDestinationAllow(address a, bool on) external onlyGov { allowDestination[a]=on; emit DestinationAllowSet(a,on); }
    function setDestinationDeny(address a, bool on) external onlyGov { denyDestination[a]=on; emit DestinationDenySet(a,on); }
 
    /* ───────── Public/Trusted Signals ─────────
       Vaults/Escrow/Gateway should call this BEFORE or AFTER transfers (either works for modeling).
       You can also feed in off-chain detected anomalies (set amount=0).
    ------------------------------------------------------------------ */
    function reportTransfer(address owner, address to, uint256 amount) external onlyReporter nonReentrant {
        // Quick policies
        if (denyDestination[to]) {
            _raise(owner, 3 /*high*/, 2 /*denylist hit*/);
            _quarantine(owner);
            _tripBreakerHigh(owner, to, amount, 2);
            return;
        }
 
        // Update meters
        uint8 severity = _updateAndClassify(owner, to, amount);
 
        emit TransferSignal(owner, to, amount, severity);
        _log(owner, 150 /*DT_Transfer*/, keccak256(abi.encode(owner, to, amount, severity)));
 
        // Act on severity
        if (severity >= 3) {
            _quarantine(owner);
            _tripBreakerHigh(owner, to, amount, 1 /*burst*/);
            _penalize(owner, penaltyHigh, true);
            _raise(owner, severity, 1);
        } else if (severity == 2) {
            _flag(owner, 1 /*risk*/ , true);
            _watch(owner, 6 hours);
            _penalize(owner, penaltyMed, false);
            _raise(owner, severity, 1);
        } else if (severity == 1) {
            _penalize(owner, penaltyLow, false);
            _raise(owner, severity, 0);
        }
    }
 
    /// @notice Report a suspected wash-trade round trip between A and B.
    function reportRoundTrip(address a, address b, uint256 weight) external onlyReporter nonReentrant {
        // bump both users' roundTrip meters
        uint256 newA = _bumpRoundTrip(a, weight);
        uint256 newB = _bumpRoundTrip(b, weight);
        emit RoundTripBump(a, b, newA, newB);
        _log(a, 151 /*DT_RoundTrip*/, keccak256(abi.encode(a, b, weight, newA, newB)));
 
        uint8 sevA = (newA >= rtHi) ? 3 : (newA >= rtMed ? 2 : 1);
        uint8 sevB = (newB >= rtHi) ? 3 : (newB >= rtMed ? 2 : 1);
 
        if (sevA >= 3) { _quarantine(a); _tripBreakerHigh(a,b,weight,3); _penalize(a, penaltyHigh, true); _raise(a,3,3); }
        else if (sevA == 2) { _flag(a,2,true); _watch(a, 8 hours); _penalize(a, penaltyMed, false); _raise(a,2,3); }
        else { _penalize(a, penaltyLow, false); _raise(a,1,3); }
 
        if (sevB >= 3) { _quarantine(b); _tripBreakerHigh(b,a,weight,3); _penalize(b, penaltyHigh, true); _raise(b,3,3); }
        else if (sevB == 2) { _flag(b,2,true); _watch(b, 8 hours); _penalize(b, penaltyMed, false); _raise(b,2,3); }
        else { _penalize(b, penaltyLow, false); _raise(b,1,3); }
    }
 
    /* ───────── Admin Incident Hooks ───────── */
 
    /// @notice DAO or Gov can manually raise an incident on a subject (e.g., confirmed scam).
    function adminRaiseIncident(address subject, uint8 severity, uint8 code, bytes32 ctx) external onlyGov nonReentrant {
        _raise(subject, severity, code);
        if (severity >= 3) { _quarantine(subject); _tripBreakerHigh(subject,address(0),0,code); }
        else if (severity == 2) { _flag(subject, code, true); _watch(subject, 12 hours); }
        _log(subject, 159 /*DT_AdminIncident*/, keccak256(abi.encode(subject, severity, code, ctx)));
    }
 
    /* ───────── Views ───────── */
 
    function userMeters(address user) external view returns (
        uint64 amountLast, uint256 amountVal,
        uint64 sendsLast,  uint256 sendsVal,
        uint64 rtLast,     uint256 rtVal,
        uint8 lastSeverity
    ) {
        UserRisk storage r = _risk[user];
        return (r.amount.lastTs, r.amount.value, r.sends.lastTs, r.sends.value, r.roundTrips.lastTs, r.roundTrips.value, r.lastSeverity);
    }
 
    /* ───────── Internals: modeling & classification ───────── */
 
    function _updateAndClassify(address owner, address to, uint256 amount) internal returns (uint8) {
        // Decay/bump amount and sends
        uint256 aVal = _decayAndAdd(_risk[owner].amount, amount, halfLifeAmount);
        uint256 cVal = _decayAndAdd(_risk[owner].sends, 1,       halfLifeCount);
 
        // Round-trip heuristic: if sending to a recent peer, bump RT meter lightly
        if (!allowDestination[to]) {
            uint256 rt = _decayAndAdd(_risk[owner].roundTrips, (owner < to ? 2 : 1), halfLifeRound);
            if (rt >= rtHi) return 3;
            if (rt >= rtMed) return 2;
        }
 
        // Classify by bursts
        if (aVal >= burstAmountHi || cVal >= burstCountHi) return 3;
        if (aVal >= burstAmountMed || cVal >= burstCountMed) return 2;
        if (amount > 0) return 1;
        return 0;
    }
 
    function _bumpRoundTrip(address user, uint256 weight) internal returns (uint256 newVal) {
        newVal = _decayAndAdd(_risk[user].roundTrips, weight, halfLifeRound);
    }
 
    /// Exponential decay approximation with linear clamp for large gaps.
    function _decayAndAdd(Meter storage m, uint256 add, uint64 halfLife) internal returns (uint256 out) {
        uint64 nowTs = uint64(block.timestamp);
        if (m.lastTs == 0) {
            m.lastTs = nowTs;
            m.value = add;
            return m.value;
        }
        uint64 dt = nowTs - m.lastTs;
        if (dt >= 10 * halfLife) {
            m.value = add;
        } else if (halfLife == 0) {
            m.value = m.value + add;
        } else {
            // value *= 2^(-dt/halfLife)  ≈ value >> k when dt ~= k*halfLife (cheap & monotone)
            uint64 k = dt / halfLife;
            if (k > 0) m.value = m.value >> k;
            // small remainder ignored for gas; acceptable for coarse meters
            m.value = m.value + add;
        }
        m.lastTs = nowTs;
        return m.value;
    }
 
    /* ───────── Internals: actions ───────── */
 
    function _penalize(address user, int16 delta, bool strongWatch) internal {
        if (address(seerTune) != address(0) && delta != 0) {
            try seerTune.adjustScore(user, delta) {} catch {}
        }
        if (strongWatch) _watch(user, 24 hours);
    }
 
    function _flag(address user, uint8 code, bool on) internal {
        if (address(seerTune) != address(0)) {
            try seerTune.setFlag(user, code, on) {} catch {}
        }
    }
 
    function _watch(address user, uint64 secs) internal {
        if (address(seerTune) != address(0)) {
            try seerTune.setWatch(user, uint64(block.timestamp) + secs) {} catch {}
        }
    }
 
    function _quarantine(address user) internal {
        if (address(guard) != address(0)) {
            try guard.setQuarantine(user, uint64(block.timestamp) + quarantineSecsHigh) {} catch {}
        }
    }
 
    function _tripBreakerHigh(address a, address b, uint256 amount, uint8 subreason) internal {
        if (address(breaker) != address(0) && breakerSeverityHigh >= 1) {
            // severity 1 = module pause; 2+ = global per EmergencyBreaker design
            try breaker.autoTrip(breakerSeverityHigh, breakerReasonAbuse, breakerModuleGuard,
                keccak256(abi.encode(a, b, amount, subreason, block.number))) {} catch {}
        }
        // Raise into global incident meter for correlation
        uint256 gi = _decayAndAdd(globalIncidents, 1, globalHalfLife);
        if (address(breaker) != address(0)) {
            if (gi >= globalIncidentHi) {
                try breaker.autoTrip(3 /*global*/, breakerReasonAbuse, breakerModuleGuard,
                    keccak256(abi.encodePacked("mass-incident-hi", gi, block.number))) {} catch {}
            } else if (gi >= globalIncidentMed) {
                try breaker.autoTrip(1 /*module*/, breakerReasonAbuse, breakerModuleGuard,
                    keccak256(abi.encodePacked("mass-incident-med", gi, block.number))) {} catch {}
            }
        }
    }
 
    function _raise(address subject, uint8 severity, uint8 code) internal {
        // 0=info 1=low 2=med 3=high
        _risk[subject].lastSeverity = severity;
        emit IncidentRaised(subject, severity, code);
        _log(subject, 152 /*DT_Incident*/, keccak256(abi.encode(subject, severity, code, block.number)));
    }
 
    function _log(address user, uint8 etype, bytes32 ctx) internal {
        if (address(proof) != address(0)) {
            try proof.logEvent(user, etype, ctx) {} catch {}
        }
    }
