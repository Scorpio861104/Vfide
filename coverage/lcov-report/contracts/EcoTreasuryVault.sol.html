/* ----------------------------------------------------------------
 * EcoTreasuryVault (Smart Automated Ecosystem Treasury)
 * ----------------------------------------------------------------
 * - Holds VFIDE allocated to ecosystem purposes.
 * - Periodically “sweeps” balance and distributes per configured allocations.
 * - Applies Seer-driven trust throttle (reduces sweep size when system isn’t healthy).
 * - Obeys EmergencyBreaker global/module pauses.
 * - Emits detailed events and logs to ProofLedger (best effort).
 *
 * Notes:
 * - This contract never pulls from user vaults. It only transfers *out* its own VFIDE.
 * - Remainder (10_000 - sum(bps)) stays as on-vault reserve.
 */

    /// @notice Distribute current balance according to allocations (subject to throttle, cap, and breaker).
    function sweep() external nonReentrant {
        _revertIfPaused();
 
        // interval gate
        uint64 nowTs = uint64(block.timestamp);
        require(lastSweepAt + minSweepInterval <= nowTs, "interval");
        lastSweepAt = nowTs;
 
        uint256 gross = vfide.balanceOf(address(this));
        if (gross == 0) {
            emit Swept(0, 0, 0, 0, 0, 0, 0);
            _log(0, 0, 0, 0, 0, 0, 0, 0);
            return;
        }
 
        // apply per-cycle cap
        if (gross > perCycleCap) gross = perCycleCap;
 
        // compute throttle
        uint16 throttleBps = _throttleBps();
 
        // amount to distribute this cycle after throttle
        uint256 distributable = (gross * throttleBps) / 10_000;
 
        // split per allocations (may be < 10000, leaving remainder as reserve)
        uint256 toSanctum  = (distributable * alloc.sanctum)     / 10_000;
        uint256 toStaking  = (distributable * alloc.stakingPool) / 10_000;
        uint256 toDaoOps   = (distributable * alloc.daoOps)      / 10_000;
        uint256 toLiq      = (distributable * alloc.liquidityFund)/ 10_000;
 
        // perform transfers (best-effort per sink only if address set)
        uint256 sentSanctum = _sendIf(sinks.sanctum, toSanctum);
        uint256 sentStaking = _sendIf(sinks.stakingPool, toStaking);
        uint256 sentDaoOps  = _sendIf(sinks.daoOps, toDaoOps);
        uint256 sentLiq     = _sendIf(sinks.liquidityFund, toLiq);
 
        // anything not sent (unset sinks) remains as reserve; plus the unallocated portion
        uint256 totalSent = sentSanctum + sentStaking + sentDaoOps + sentLiq;
        uint256 retained  = vfide.balanceOf(address(this)); // actual remaining
 
        emit Swept(
            gross,
            throttleBps,
            sentSanctum,
            sentStaking,
            sentDaoOps,
            sentLiq,
            retained
        );
        _log(gross, throttleBps, sentSanctum, sentStaking, sentDaoOps, sentLiq, retained, block.number);
    }

    }

    function _revertIfPaused() internal view {
        if (address(breaker) != address(0)) {
            if (breaker.isSystemPaused()) revert Paused();
            // 7 == Router/Funding module ID per our shared enum map
            if (breaker.isModulePaused(7)) revert Paused();
        }
    }
    function _requireNotPausedView() internal view {
        if (address(breaker) != address(0)) {
            require(!breaker.isSystemPaused(), "sys paused");
            require(!breaker.isModulePaused(7), "router paused");
        }
    }
 
    function _throttleBps() internal view returns (uint16) {
        // If no Seer: full flow
        if (address(seer) == address(0)) return 10_000;
 
        // Try systemHealthy() first (may revert if not implemented)
        bool healthy = false;
        try seer.systemHealthy() returns (bool ok) { healthy = ok; } catch { healthy = false; }
        if (healthy) return 10_000;
 
        // If not healthy, scale by DAO score (if dao set), else by this contract’s score
        uint16 s = 0;
        if (dao != address(0)) {
            try seer.score(dao) returns (uint16 d) { s = d; } catch {}
        } else {
            try seer.score(address(this)) returns (uint16 d2) { s = d2; } catch {}
        }
        if (s >= minScoreForFullFlow) return 10_000;
 
        // linear map: [0..minScoreForFullFlow) → [minThrottleBps .. 10000)
        // bps = minThrottle + (s / minScoreForFullFlow) * (10000 - minThrottle)
        if (minScoreForFullFlow == 0) return minThrottleBps;
        uint256 span = 10_000 - uint256(minThrottleBps);
        uint256 scaled = uint256(minThrottleBps) + (uint256(s) * span) / uint256(minScoreForFullFlow);
        if (scaled > 10_000) scaled = 10_000;
        return uint16(scaled);
    }
 
    function _throttleBpsView() internal view returns (uint16) {
        return _throttleBps();
    }
 
    function _sendIf(address to, uint256 amt) internal returns (uint256 sent) {
        if (to == address(0) || amt == 0) return 0;
        bool ok = vfide.transfer(to, amt);
        return ok ? amt : 0;
    }
 
    function _log(
        uint256 gross,
        uint16 throttleBps,
        uint256 sSanctum,
        uint256 sStaking,
        uint256 sDao,
        uint256 sLiq,
        uint256 retained,
        uint256 blk
    ) internal {

            bytes32 ctx = keccak256(abi.encode(gross, throttleBps, sSanctum, sStaking, sDao, sLiq, retained, blk));
            // 120 = EcoTreasury sweep
            try proof.logEvent(address(this), 120, ctx) {} catch {}
        }
    }
