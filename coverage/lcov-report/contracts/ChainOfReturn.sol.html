/* ───────────────────────────── ChainOfReturn (Vault Recovery) ─────────────────────────────
   Adds **per-vault overrides** with **DAO bounds** while keeping the original flow:
   - Guardians (M-of-N), Seer gates, cooldown, approval expiry, breaker/guard/proof logging.
   - Recovery rotates the wallet that controls a vault (no fund clawback).
-------------------------------------------------------------------------------------------------------- */

    /// @notice Vault owner can set (or clear) custom recovery parameters within DAO bounds.
    function setVaultPolicy(
        address vault,
        bool    clear,
        uint16  minGuardianScore_,
        uint64  approvalExpiry_,
        uint64  cooldownSecs_
    ) external breakerGuard nonReentrant {
        _requireVault(vault);
        address owner_ = _vaultOwner(vault);
        if (msg.sender != owner_) revert NotOwner();
 
        if (clear) {
            delete vaultPolicy[vault];
            emit VaultPolicySet(vault, 0, 0, 0, true);
            _log(182 /*VaultPolicyCleared*/, keccak256(abi.encode(vault)));
            return;
        }
 
        // enforce bounds
        require(minGuardianScore_ >= bounds.minGuardianScoreMin && minGuardianScore_ <= bounds.minGuardianScoreMax, "g out of bounds");
        require(approvalExpiry_   >= bounds.approvalExpiryMin   && approvalExpiry_   <= bounds.approvalExpiryMax,   "exp out of bounds");
        require(cooldownSecs_     >= bounds.cooldownSecsMin     && cooldownSecs_     <= bounds.cooldownSecsMax,     "cd out of bounds");
 
        vaultPolicy[vault] = VaultPolicy({
            hasOverride: true,
            minGuardianScore: minGuardianScore_,
            approvalExpiry: approvalExpiry_,
            cooldownSecs: cooldownSecs_
        });
 
        emit VaultPolicySet(vault, minGuardianScore_, approvalExpiry_, cooldownSecs_, false);
        _log(183 /*VaultPolicySet*/, keccak256(abi.encode(vault, minGuardianScore_, approvalExpiry_, cooldownSecs_)));
    }
 
    /// Effective policy = vault override if set, else global defaults.
    function effectivePolicy(address vault) public view returns (uint16 gScore, uint64 expSecs, uint64 cdSecs) {
        VaultPolicy memory vp = vaultPolicy[vault];
        if (vp.hasOverride) return (vp.minGuardianScore, vp.approvalExpiry, vp.cooldownSecs);
        return (minGuardianScore, approvalExpiry, cooldownSecs);
    }

    function setGuardians(address vault, address[] calldata guardians, uint8 threshold) external breakerGuard nonReentrant {
        _requireVault(vault);
        address owner_ = _vaultOwner(vault);
        if (msg.sender != owner_) revert NotOwner();
        if (guardians.length == 0 || guardians.length > MAX_GUARDIANS) revert BadParams();
        if (threshold == 0 || threshold > guardians.length) revert BadParams();
 
        (uint16 gScoreReq,,) = effectivePolicy(vault);
 
        // uniqueness + Seer checks
        for (uint256 i=0; i<guardians.length; i++) {
            address gi = guardians[i];
            if (gi == address(0)) revert ZeroAddress();
            for (uint256 j=i+1; j<guardians.length; j++) require(gi != guardians[j], "dup guardian");
            require(!seer.isBlocked(gi), "g blocked");
            require(seer.score(gi) >= gScoreReq, "g score");
        }
 
        config[vault] = GuardianSet({ guardians: guardians, threshold: threshold, set: true });
        emit GuardiansConfigured(vault, guardians, threshold);
        _log(170 /*GuardiansSet*/, keccak256(abi.encode(vault, guardians, threshold)));
    }

        GuardianSet storage g = config[vault];
        return (g.guardians, g.threshold, g.set);

    function propose(address vault, address newWallet, bytes32 salt) external breakerGuard nonReentrant returns (bytes32 id) {
        if (newWallet == address(0)) revert ZeroAddress();
        _requireVault(vault);
        GuardianSet storage g = config[vault];
        if (!g.set) revert NotConfigured();
 
        bool proposerOk = (msg.sender == _vaultOwner(vault)) || _isGuardian(g, msg.sender);
        require(proposerOk, "not proposer");
 
        id = requestId(vault, newWallet, salt);
        if (_reqs[id].createdAt != 0) revert AlreadyExists();
 
        RecoveryRequest storage r = _reqs[id];
        r.vault      = vault;
        r.newWallet  = newWallet;
        r.proposer   = msg.sender;
        r.createdAt  = uint64(block.timestamp);
        r.state      = ReqState.Active;
        latestRequestOf[vault] = id;
 
        emit RecoveryProposed(id, vault, newWallet, msg.sender);
        _log(171 /*RecoveryProposed*/, keccak256(abi.encode(id, vault, newWallet, msg.sender)));
 
        if (address(seerTune) != address(0) && msg.sender != _vaultOwner(vault)) {
            try seerTune.setFlag(msg.sender, 5 /*recovery-propose*/, true) {} catch {}
        }
    }
 
    function approve(bytes32 id) external breakerGuard nonReentrant {
        RecoveryRequest storage r = _reqsMust(id);
        if (r.state != ReqState.Active) revert NotActive();
        GuardianSet storage g = config[r.vault];
        if (!_isGuardian(g, msg.sender)) revert NotGuardian();
        if (r.approved[msg.sender]) revert DuplicateApproval();
 
        (uint16 gScoreReq,,) = effectivePolicy(r.vault);
        require(!seer.isBlocked(msg.sender), "g blocked");
        require(seer.score(msg.sender) >= gScoreReq, "g score");
 
        r.approved[msg.sender] = true;
        r.approvals += 1;
 
        emit RecoveryApproved(id, msg.sender, r.approvals, g.threshold);
        _log(172 /*RecoveryApproved*/, keccak256(abi.encode(id, msg.sender, r.approvals, g.threshold)));
 
        if (r.approvals >= g.threshold && r.thresholdAt == 0) {
            (, , uint64 cd) = effectivePolicy(r.vault);
            r.thresholdAt = uint64(block.timestamp);
            emit ThresholdReached(id, r.thresholdAt, r.thresholdAt + cd);
            _log(173 /*ThresholdReached*/, keccak256(abi.encode(id, r.thresholdAt, cd)));
        }
    }
 
    function execute(bytes32 id) external breakerGuard nonReentrant {
        RecoveryRequest storage r = _reqsMust(id);
        if (r.state != ReqState.Active && r.state != ReqState.Executable) revert NotActive();
        GuardianSet storage g = config[r.vault];
        require(g.set, "no guardians");
 
        ( , uint64 exp, uint64 cd) = effectivePolicy(r.vault);
 
        require(r.approvals >= g.threshold, "need approvals");
        require(r.thresholdAt != 0, "no threshold");
        uint64 execAt = r.thresholdAt + cd;
        if (block.timestamp < execAt) revert NotExecutable();
        if (block.timestamp > r.createdAt + exp) revert Expired();
 
        _rotateVault(r.vault, r.newWallet);
 
        r.state = ReqState.Executed;
 
        emit RecoveryExecuted(id, r.vault, r.newWallet);
        _log(174 /*RecoveryExecuted*/, keccak256(abi.encode(id, r.vault, r.newWallet)));
    }
 
    function cancel(bytes32 id, string calldata reason) external breakerGuard nonReentrant {
        RecoveryRequest storage r = _reqsMust(id);
        if (r.state != ReqState.Active) revert NotActive();
        if (msg.sender != _vaultOwner(r.vault)) revert NotOwner();
 
        r.state = ReqState.Canceled;
        emit RecoveryCanceled(id, msg.sender, reason);
        _log(175 /*RecoveryCanceled*/, keccak256(abi.encode(id, msg.sender, keccak256(bytes(reason)))));
 
        if (enableQuarantine && address(guard) != address(0)) {
            try guard.setQuarantine(r.proposer, uint64(block.timestamp) + quarantineSecs) {} catch {}
        }
    }

    function _reqsMust(bytes32 id) internal view returns (RecoveryRequest storage) {
        RecoveryRequest storage r = _reqs[id];
        if (r.createdAt == 0) revert BadParams();
        return r;
    }
    function _isGuardian(GuardianSet storage g, address who) internal view returns (bool) {
        for (uint256 i=0; i<g.guardians.length; i++) if (g.guardians[i] == who) return true;
        return false;
    }
    function _requireVault(address vault) internal view {
        if (vault == address(0)) revert ZeroAddress();
        if (address(factory) != address(0)) require(factory.isVault(vault), "not vault");
    }
    function _vaultOwner(address vault) internal view returns (address owner_) {
        try IUserVaultOwnable(vault).owner() returns (address o) { owner_ = o; } catch { owner_ = address(0); }
    }
    function _rotateVault(address vault, address newWallet) internal {
        bool ok;
        // Prefer explicit recoverTo(newWallet)
        try IUserVaultRecoverTo(vault).recoverTo(newWallet) { ok = true; }
        catch {
            try IUserVaultOwnable(vault).transferOwnership(newWallet) { ok = true; }
            catch { ok = false; }
        }
        require(ok, "vault rotation failed");
    }

    function previewRequest(bytes32 id)
        external
        view
        returns (address vault, address newWallet, address proposer, uint64 createdAt, uint64 thresholdAt, uint64 executableAt, ReqState state, uint8 approvals, uint8 threshold)

        return (r.vault, r.newWallet, r.proposer, r.createdAt, r.thresholdAt, execAt, r.state, r.approvals, g.threshold);
    }
