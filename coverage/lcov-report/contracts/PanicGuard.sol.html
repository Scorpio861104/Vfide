/* ─────────────────────────── PanicGuard (Hybrid) ───────────────────────────
   - Active layer: canSend(owner, toUser, amount) -> bool (called by UserVault BEFORE transfer)
   - Reactive layer: recordSend(owner, toUser, amount) (called by UserVault AFTER successful transfer)
   - DAO/Owner governance (onlyGov)
   - Seer risk gates (blocked + min scores) ON by default
   - Dest allow/deny lists, system exemptions
   - Rate-limit & pattern heuristics (panic dump / burst / wash-like churn)
   - Auto-quarantine (temporary) on triggers; DAO can freeze/quarantine/unfreeze
   - Optional ProofLedger audit logs (best-effort)
-----------------------------------------------------------------------------*/

    /* ─────────────────────────── ACTIVE LAYER ───────────────────────────
       Called by UserVault BEFORE transfers: blocks unsafe flows in real time.
    ---------------------------------------------------------------------- */
    function canSend(address owner, address toUser, uint256 amount) external view returns (bool) {
        // System exemptions bypass everything
        if (systemExemptUser[owner] || systemExemptUser[toUser]) return true;
 
        // Hard stops
        if (globalPause) return false;
        if (frozen[owner]) return false;
        uint64 q = quarantineUntil[owner];
        if (q != 0 && block.timestamp < q) return false;
 
        // Destination policy
        if (denyDestUser[toUser]) return false;
 
        // If destination explicitly allowed, skip score checks (still respected hard stops above)
        bool destAllowed = allowDestUser[toUser];
 
        // Seer risk gates
        if (seerGatesEnabled && !destAllowed && address(seer) != address(0)) {
            if (blockIfBlocked) {
                if (seer.isBlocked(owner)) return false;
                if (seer.isBlocked(toUser)) return false;
            }
            if (enforceSenderMin && seer.score(owner) < minSenderScore) return false;
            if (enforceReceiverMin && seer.score(toUser) < minReceiverScore) return false;
        }
 
        // Lightweight pre-check using current decayed meters (no writes)
        if (monitoringEnabled) {
            Meter memory m = _peekDecay(owner);
            // projected values after this tx
            uint128 outAmtProj = m.outAmt + uint128(_safeAmt(amount));
            uint64  outCntProj = m.outCount + 1;
            uint32  rtProj     = m.roundTrips + ((toUser == m.lastCounterparty) ? 1 : 0);
 
            if (outAmtProj > _toU128(maxOutAmtPerWindow)) return false;
            if (outCntProj > maxOutCountPerWindow) return false;
            if (rtProj > maxRoundTripsPerWindow) return false;
        }
 
        return true;
    }
 
    /* ─────────────────────────── REACTIVE LAYER ───────────────────────────
       Called by UserVault AFTER a successful transfer to update meters and
       optionally auto-quarantine on pattern triggers.
       Auth: must be called by a valid vault whose owner == `owner`.
    ----------------------------------------------------------------------- */
    function recordSend(address owner, address toUser, uint256 amount) external nonReentrant {
        // authenticate caller: must be a vault that belongs to `owner`
        require(address(factory) != address(0), "factory unset");
        require(factory.isVault(msg.sender), "not vault");
        require(factory.ownerOfVault(msg.sender) == owner, "owner mismatch");
 
        if (!monitoringEnabled) return;
 
        // update decayed meters
        Meter storage m = meterOf[owner];
        _applyDecay(m);
 
        m.outAmt   = m.outAmt + uint128(_safeAmt(amount));
        m.outCount = m.outCount + 1;
 
        if (toUser == m.lastCounterparty) {
            // repeated back-and-forth with same counterparty (churn)
            unchecked { m.roundTrips += 1; }
        } else {
            m.lastCounterparty = toUser;
            // light decay of roundTrips when switching counterparties
            if (m.roundTrips > 0) m.roundTrips -= 1;
        }
 
        emit ReactiveObservation(owner, toUser, amount, m.outAmt, m.outCount, m.roundTrips);
 
        // triggers → auto quarantine
        bool triggered = false;
        if (m.outAmt > _toU128(maxOutAmtPerWindow)) {
            triggered = true;
            _reactiveQuarantine(owner, "amt");
        } else if (m.outCount > maxOutCountPerWindow) {
            triggered = true;
            _reactiveQuarantine(owner, "burst");
        } else if (m.roundTrips > maxRoundTripsPerWindow) {
            triggered = true;
            _reactiveQuarantine(owner, "churn");
        }
 
        if (triggered) {
            emit ReactiveTrigger(owner, "pattern");
            _log(owner, 4 /* ReactiveTrigger */, keccak256(abi.encode(owner, toUser, amount)));
        }
    }
 
    /* ─────────────────────────── Internal: decay & helpers ─────────────────────────── */
 
    function _applyDecay(Meter storage m) internal {
        uint64 ts = uint64(block.timestamp);
        uint64 last = m.lastTs;
        if (last == 0) { m.lastTs = ts; return; }
 
        uint64 dt = ts - last;
        if (dt == 0) return;
 
        // Exponential decay approximation using half-life:
        // remaining = value * 2^(-dt / halfLife)
        // Implement as repeated halving by k = dt / halfLife (coarse but gas-cheap)
        uint64 k = decayHalfLife == 0 ? 0 : (dt / decayHalfLife);
        if (k > 0) {
            // halve k times (cap to avoid excessive loops)
            uint64 steps = k > 6 ? 6 : k; // at most ~1/64; more than enough to flush bursts
            for (uint64 i = 0; i < steps; i++) {
                m.outAmt   = m.outAmt   >> 1;
                m.outCount = m.outCount >> 1;
                m.roundTrips = m.roundTrips >> 1;
            }
        }
        m.lastTs = ts;
    }
 
    function _peekDecay(address owner) internal view returns (Meter memory r) {
        r = meterOf[owner];
        if (r.lastTs == 0) return r;
        uint64 ts = uint64(block.timestamp);
        uint64 dt = ts - r.lastTs;
        uint64 k = decayHalfLife == 0 ? 0 : (dt / decayHalfLife);
        if (k > 0) {
            uint64 steps = k > 6 ? 6 : k;
            for (uint64 i = 0; i < steps; i++) {
                r.outAmt     = r.outAmt     >> 1;
                r.outCount   = r.outCount   >> 1;
                r.roundTrips = r.roundTrips >> 1;
            }
        }
    }
 
    function _reactiveQuarantine(address user, string memory reason) internal {
        if (!autoQuarantineEnabled) return;
        uint64 untilTs = uint64(block.timestamp) + autoQuarantinePeriod;
        quarantineUntil[user] = untilTs;
        emit Quarantined(user, untilTs);
        _log(user, 3 /* AutoQuarantine */, keccak256(abi.encode(user, reason, untilTs)));
    }
 
    function _safeAmt(uint256 a) internal pure returns (uint256) {
        // Clamp to 1e38 to safely cast to uint128 without overflow (covers typical ERC20 with 18 decimals).
        if (a > type(uint128).max) return type(uint128).max;
        return a;
    }
    function _toU128(uint256 a) internal pure returns (uint128) {
        return a > type(uint128).max ? type(uint128).max : uint128(a);
    }
 
    function _log(address user, uint8 etype, bytes32 ctx) internal {
        if (address(proof) != address(0)) {
            try proof.logEvent(user, etype, ctx) {} catch {}
        }
    }
