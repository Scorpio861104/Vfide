/* ───────────────────────────── GuardianRegistry ─────────────────────────────
   Central manager for guardians & recoveries in VFIDE.
 
   Key features:
   - Per-user guardian set with M-of-N threshold (user-managed; DAO override available).
   - Seer-gated guardian admission (min score, not blocked) with exemptions.
   - Recovery cases (guardian-initiated) with auto-freeze via PanicGuard and contest period.
   - Approvals tracked per-case; finalization callable by authorized finalizers (DAO / CoR).
   - ProofLedger logging (best effort), extensive events, safe modifiers.
 
   Notes:
   - Registry does NOT move funds. It coordinates approvals and signals to PanicGuard & CoR.
   - VaultFactory is used only for sanity (e.g., presence of user vaults for UX).

    /// @notice Guardian opens a recovery case for `user` proposing `newController`.
    function openRecovery(address user, address newController, bytes32 reason) external nonReentrant returns (uint256 caseId) {
        if (!_isGuardian(user, msg.sender)) revert NotGuardian();
        if (activeCaseOf[user] != 0) revert CaseActive();
        require(newController != address(0) && newController != user, "bad newController");
 
        caseId = nextCaseId++;
        RecoveryCase storage rc = _case[caseId];
        rc.user              = user;
        rc.newController     = newController;
        rc.openedBy          = msg.sender;
        rc.status            = CaseStatus.Open;
        rc.openedAt          = uint64(block.timestamp);
        rc.reason            = reason;
 
        // Auto-freeze via PanicGuard (optional)
        if (freezeOnOpen && address(guard) != address(0)) {
            uint64 untilTs = uint64(block.timestamp) + contestPeriodSecs;
            // Best effort; if it reverts, continue without freeze
            try guard.setQuarantine(user, untilTs) {
                rc.frozeOnOpen = true;
            } catch { rc.frozeOnOpen = false; }
        }
 
        // Auto-approve by opener (guardian)
        rc.approved[msg.sender] = true;
        rc.approvals = 1;
 
        activeCaseOf[user] = caseId;
        emit RecoveryOpened(caseId, user, msg.sender, newController, reason);
        _log(user, 80 /*RecoveryOpened*/, keccak256(abi.encode(caseId, newController, reason)));
    }
 
    /// @notice Guardian approves an open recovery case.
    function approve(uint256 caseId) external nonReentrant {
        RecoveryCase storage rc = _mustOpenCase(caseId);
        if (!_isGuardian(rc.user, msg.sender)) revert NotGuardian();
        if (rc.approved[msg.sender]) revert AlreadyApproved();
 
        rc.approved[msg.sender] = true;
        rc.approvals += 1;
 
        // Check threshold
        (uint8 thresh, uint8 count) = _thresholdAndCount(rc.user);
        bool met = rc.approvals >= thresh && thresh >= 1 && thresh <= count;
 
        if (met && rc.status == CaseStatus.Open) {
            rc.status = CaseStatus.ThresholdMet;
            rc.thresholdMetAt = uint64(block.timestamp);
            rc.finalizeAfter  = rc.thresholdMetAt + contestPeriodSecs;
            emit RecoveryThresholdMet(caseId, rc.finalizeAfter);
        }
 
        emit RecoveryApproved(caseId, msg.sender, rc.approvals, met);
        _log(rc.user, 81 /*RecoveryApproved*/, keccak256(abi.encode(caseId, msg.sender, rc.approvals)));
    }

    /// @notice Finalize after threshold met + contest period elapsed.
    ///         Callable by DAO or any address whitelisted as an authorized finalizer (e.g., ChainOfReturn).
    function finalize(uint256 caseId) external nonReentrant {
        RecoveryCase storage rc = _case[caseId];
        if (rc.status != CaseStatus.ThresholdMet) revert NotOpen();
        if (block.timestamp < rc.finalizeAfter) revert ContestNotElapsed();
        if (msg.sender != owner() && msg.sender != dao && !isAuthorizedFinalizer[msg.sender]) revert NotAuthorized();
 
        rc.status = CaseStatus.Finalized;
        activeCaseOf[rc.user] = 0;
 
        // Invoke Chain-of-Return executor if set and caller is authorized
        if (address(chain) != address(0)) {
            try chain.executeRecovery(rc.user, rc.newController) { } catch { /* ignore to keep on-chain record */ }
        }
 
        emit RecoveryFinalized(caseId, msg.sender);
        _log(rc.user, 83 /*RecoveryFinalized*/, keccak256(abi.encode(caseId, rc.newController)));
    }

    function _configureGuardiansFor(address user, address[] calldata guardians, uint8 threshold) internal {
        require(user != address(0), "user=0");
        uint256 n = guardians.length;
        require(n >= 1 && n <= maxGuardiansPerUser, "bad count");
 
        // Reset existing
        GuardianSet storage gs = _gset[user];
        // Remove all old guardians (EnumerableSet has no clear; rebuild by new set)
        // We overwrite by creating a new set via local, then swap by removing old items one-by-one
        // Simpler approach: remove while length > 0
        while (gs.set.length() > 0) {
            address g = gs.set.at(gs.set.length()-1);
            gs.set.remove(g);
        }
 
        // Add new guardians with Seer checks
        for (uint256 i = 0; i < n; i++) {
            address g = guardians[i];
            _validateGuardian(g);
            if (g == user) revert SeerGate(); // cannot add self
            if (!gs.set.add(g)) revert GuardianAlready();
        }
 
        // Threshold
        _requireValidThreshold(user, uint8(n), threshold);
        gs.threshold = threshold;
        gs.lastUpdated = uint64(block.timestamp);
 
        emit ThresholdUpdated(user, threshold);
        for (uint256 i = 0; i < n; i++) {
            emit GuardianAdded(user, guardians[i], uint8(i+1), threshold);
        }
 
        _log(user, 84 /*GuardiansConfigured*/, keccak256(abi.encode(user, n, threshold)));
    }
 
    function _addGuardianFor(address user, address guardian) internal {
        require(user != address(0), "user=0");
        _validateGuardian(guardian);
        GuardianSet storage gs = _gset[user];
        uint256 current = gs.set.length();
        if (current + 1 > maxGuardiansPerUser) revert TooManyGuardians();
        if (!gs.set.add(guardian)) revert GuardianAlready();
        gs.lastUpdated = uint64(block.timestamp);
        emit GuardianAdded(user, guardian, uint8(current+1), gs.threshold);
        _log(user, 85 /*GuardianAdded*/, keccak256(abi.encode(user, guardian)));
    }
 
    function _removeGuardianFor(address user, address guardian) internal {
        GuardianSet storage gs = _gset[user];
        if (!gs.set.remove(guardian)) revert GuardianMissing();
 
        uint8 newCount = uint8(gs.set.length());
        // If threshold now exceeds count, lower to count (but not below 1 if count>0)
        if (gs.threshold > newCount) {
            gs.threshold = newCount == 0 ? 0 : newCount;
            emit ThresholdUpdated(user, gs.threshold);
        }
        gs.lastUpdated = uint64(block.timestamp);
        emit GuardianRemoved(user, guardian, newCount, gs.threshold);
        _log(user, 86 /*GuardianRemoved*/, keccak256(abi.encode(user, guardian)));
    }

    function _validateGuardian(address g) internal view {
        if (g == address(0)) revert ZeroAddress();
        if (systemExemptUser[g]) return;
        if (address(seer) != address(0)) {
            if (blockIfBlocked && seer.isBlocked(g)) revert SeerGate();
            uint16 s = seer.score(g);
            if (s < minGuardianScore) revert SeerGate();
        }
