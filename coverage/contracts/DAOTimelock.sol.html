// SPDX-License-Identifier: MIT
pragma solidity 0.8.30;
 
import "@openzeppelin/contracts/access/Ownable2Step.sol";
import "@openzeppelin/contracts/security/ReentrancyGuard.sol";
 
/* ───────── Minimal VFIDE interfaces ───────── */
 
interface IProofLedger {
    function logEvent(address user, uint8 etype, bytes32 ctx) external;
}
 
interface ISeerView {
    function score(address user) external view returns (uint16); // 0..1000
    function isBlocked(address user) external view returns (bool);
}
 
interface IEmergencyBreaker {
    function isSystemPaused() external view returns (bool);
}
 
/* ─────────────────────────── DAOTimelock (VFIDE Integrated Trust Timelock) ───────────────────────────
   Features:
   - Timelock operations: queue, execute, cancel (OZ-style opId based on target/value/data/predecessor/salt).
   - Proposer/Executor allowlists + Seer/ProofScore gates for trust-aware governance.
   - EmergencyBreaker awareness (blocks queue/execute while system paused).
   - ProofLedger logging (best effort).
   - Dual governance: owner() pre-handover, DAO post-handover; scheduled auto-handover after 6 months.
 
   Notes:
   - Supports native ETH value in scheduled calls; the timelock must be pre-funded if value > 0.
   - Uses nonReentrant on state-changing entrypoints.
-------------------------------------------------------------------------------------------------------*/
contract DAOTimelock is Ownable2Step, ReentrancyGuard {
    /* ───────── Wiring ───────── */
    address       public dao;        // set after handover; becomes part of onlyGov
    ISeerView     public seer;       // optional; if unset, Seer gates are skipped
    IEmergencyBreaker public breaker;// optional; if unset, breaker checks are skipped
    IProofLedger  public proof;      // optional ProofLedger logging
 
    /* ───────── Delay config ───────── */
    uint64 public minDelay = 48 hours;  // base timelock delay
    uint64 public maxDelay = 30 days;   // safety upper bound for updates
 
    /* ───────── Seer trust gates ───────── */
    uint16 public minScoreProposer = 600; // proposers must have >= score (if seer set)
    uint16 public minDAOScore      = 600; // dao/executors score floor for execute (if seer set)
    bool   public enforceSeerForQueue   = true;
    bool   public enforceSeerForExecute = true;
 
    /* ───────── Roles ───────── */
    mapping(address => bool) public isProposer; // allowed to queue
    mapping(address => bool) public isExecutor; // allowed to execute
    // dao and owner are always implicit proposers/executors
 
    /* ───────── Handover (auto after 6 months by default) ───────── */
    uint64 public handoverStart;               // when countdown begins (deploy-time default)
    uint64 public handoverDelay = 180 days;    // 6 months
    bool   public handoverFinalized;
 
    /* ───────── Operations ───────── */
    struct Operation {
        uint64 readyAt;        // timestamp when executable
        bytes32 predecessor;   // dependency opId (0 if none)
        address proposer;      // who queued
        bool    done;          // executed
    }
    mapping(bytes32 => Operation) public ops;
 
    /* ───────── Events ───────── */
    event DAOUpdated(address indexed dao);
    event SeerSet(address indexed seer);
    event BreakerSet(address indexed breaker);
    event ProofSinkSet(address indexed proof);
 
    event MinDelayUpdated(uint64 newMinDelay);
    event SeerGatesUpdated(bool queueGate, bool executeGate, uint16 minProp, uint16 minDAO);
 
    event ProposerSet(address indexed who, bool on);
    event ExecutorSet(address indexed who, bool on);
 
    event OperationQueued(bytes32 indexed id, address indexed proposer, address target, uint256 value, bytes data, bytes32 predecessor, bytes32 salt, uint64 readyAt);
    event OperationExecuted(bytes32 indexed id, address indexed executor, address target, uint256 value, bytes data);
    event OperationCanceled(bytes32 indexed id, address indexed by);
 
    event HandoverFinalized(address indexed dao, uint64 at);
 
    /* ───────── Errors ───────── */
    error NotGov();
    error NotProposer();
    error NotExecutor();
    error Paused();
    error BadDelay();
    error SeerRejected();
    error OpNotReady();
    error OpDone();
    error OpMissing();
    error PredecessorIncomplete();
 
    /* ───────── Modifiers ───────── */
    modifier onlyGov() {
        if (msg.sender != owner() && msg.sender != dao) revert NotGov();
        _;
    }
 
    constructor(address initialOwner) {
        Erequire(initialOwner != address(0), "owner=0");
        _transferOwnership(initialOwner);
        handoverStart = uint64(block.timestamp);
    }
 
    /* ───────── Wiring setters ───────── */
 
    function setDAO(address dao_) external onlyGov {
        require(dao_ != address(0), "dao=0");
        dao = dao_;
        emit DAOUpdated(dao_);
    }
 
    function setSeer(address s_) external onlyGov {
        seer = ISeerView(s_);
        emit SeerSet(s_);
    }
 
    function setBreaker(address b_) external onlyGov {
        breaker = IEmergencyBreaker(b_);
        emit BreakerSet(b_);
    }
 
    function setProofSink(address p_) external onlyGov {
        proof = IProofLedger(p_);
        emit ProofSinkSet(p_);
    }
 
    /* ───────── Governance knobs ───────── */
 
    function setMinDelay(uint64 newDelay) external onlyGov {
        if (newDelay < 1 hours || newDelay > maxDelay) revert BadDelay();
        minDelay = newDelay;
        emit MinDelayUpdated(newDelay);
        _log(210 /*MinDelayUpdated*/, keccak256(abi.encode(newDelay)));
    }
 
    function setSeerGates(bool gateQueue, bool gateExecute, uint16 minProp, uint16 minDAO_) external onlyGov {
        require(minProp <= 1000 && minDAO_ <= 1000, "score>max");
        enforceSeerForQueue = gateQueue;
        enforceSeerForExecute = gateExecute;
        minScoreProposer = minProp;
        minDAOScore = minDAO_;
        emit SeerGatesUpdated(gateQueue, gateExecute, minProp, minDAO_);
        _log(211 /*SeerGatesUpdated*/, keccak256(abi.encode(gateQueue, gateExecute, minProp, minDAO_)));
    }
 
    function setProposer(address who, bool on) external onlyGov {
        require(who != address(0), "who=0");
        isProposer[who] = on;
        emit ProposerSet(who, on);
    }
 
    function setExecutor(address who, bool on) external onlyGov {
        require(who != address(0), "who=0");
        isExecutor[who] = on;
        emit ExecutorSet(who, on);
    }
 
    /* ───────── Handover controls ───────── */
 
    function setHandoverStart(uint64 startTs) external onlyGov {
        handoverStart = startTs;
    }
 
    function setHandoverDelay(uint64 delaySecs) external onlyGov {
        require(delaySecs >= 7 days, "short");
        handoverDelay = delaySecs;
    }
 
    function handoverEligibleAt() public view returns (uint64) {
        return handoverStart + handoverDelay;
    }
 
    function canFinalizeHandover() public view returns (bool) {
        if (handoverFinalized) return false;
        if (block.timestamp < handoverEligibleAt()) return false;
        if (address(seer) == address(0)) return true; // no extra gate
        // If seer is set, require DAO not blocked and above score floor
        if (dao == address(0)) return false;
        if (seer.isBlocked(dao)) return false;
        return seer.score(dao) >= minDAOScore;
    }
 
    /// @notice One-way: set DAO governance active after the window.
    function finalizeHandover() external nonReentrant {
        require(canFinalizeHandover(), "not eligible");
        handoverFinalized = true;
        emit HandoverFinalized(dao, uint64(block.timestamp));
        _log(212 /*HandoverFinalized*/, keccak256(abi.encode(dao, block.number)));
    }
 
    /* ───────── Breaker guard ───────── */
 
    function _revertIfPaused() internal view {
        if (address(breaker) != address(0) && breaker.isSystemPaused()) revert Paused();
    }
 
    /* ───────── Seer gates (helpers) ───────── */
 
    function _checkProposerTrust(address proposer) internal view {
        if (!enforceSeerForQueue || address(seer) == address(0)) return;
        if (seer.isBlocked(proposer)) revert SeerRejected();
        require(seer.score(proposer) >= minScoreProposer, "proposer score");
    }
 
    function _checkExecutorTrust(address executor) internal view {
        if (!enforceSeerForExecute || address(seer) == address(0)) return;
        if (seer.isBlocked(executor)) revert SeerRejected();
        // also apply DAO health floor if dao is set
        if (dao != address(0)) {
            require(!seer.isBlocked(dao), "dao blocked");
            require(seer.score(dao) >= minDAOScore, "dao score");
        }
    }
 
    /* ───────── OZ-style op id ───────── */
 
    function hashOperation(
        address target,
        uint256 value,
        bytes calldata data,
        bytes32 predecessor,
        bytes32 salt
    ) public pure returns (bytes32) {
        return keccak256(abi.encode(target, value, keccak256(data), predecessor, salt));
    }
 
    function isOperationPending(bytes32 id) public view returns (bool) {
        Operation memory op = ops[id];
        return op.readyAt != 0 && !op.done && block.timestamp < op.readyAt;
    }
 
    function isOperationReady(bytes32 id) public view returns (bool) {
        Operation memory op = ops[id];
        return op.readyAt != 0 && !op.done && block.timestamp >= op.readyAt;
    }
 
    function isOperationDone(bytes32 id) public view returns (bool) {
        return ops[id].done;
    }
 
    /* ───────── Queue / Execute / Cancel ───────── */
 
    /// @notice Queue a call; returns the operation id.
    function queue(
        address target,
        uint256 value,
        bytes calldata data,
        bytes32 predecessor,
        bytes32 salt,
        uint64 delay // must be >= minDelay
    ) external nonReentrant returns (bytes32 id) {
        _revertIfPaused();
 
        // Proposer role
        if (!(msg.sender == owner() || msg.sender == dao || isProposer[msg.sender])) revert NotProposer();
 
        _checkProposerTrust(msg.sender);
        require(delay >= minDelay, "delay<min");
        require(target != address(0), "target=0");
 
        id = hashOperation(target, value, data, predecessor, salt);
        Operation storage op = ops[id];
        require(op.readyAt == 0, "op exists");
 
        op.readyAt = uint64(block.timestamp) + delay;
        op.predecessor = predecessor;
        op.proposer = msg.sender;
        op.done = false;
 
        emit OperationQueued(id, msg.sender, target, value, data, predecessor, salt, op.readyAt);
        _log(213 /*OperationQueued*/, keccak256(abi.encode(id, target, value, keccak256(data), predecessor, salt, op.readyAt)));
    }
 
    /// @notice Execute a queued operation when ready.
    function execute(
        address target,
        uint256 value,
        bytes calldata data,
        bytes32 predecessor,
        bytes32 salt
    ) external payable nonReentrant returns (bytes memory returndata) {
        _revertIfPaused();
 
        // Executor role
        if (!(msg.sender == owner() || msg.sender == dao || isExecutor[msg.sender])) revert NotExecutor();
        _checkExecutorTrust(msg.sender);
 
        bytes32 id = hashOperation(target, value, data, predecessor, salt);
        Operation storage op = ops[id];
 
        if (op.readyAt == 0) revert OpMissing();
        if (op.done) revert OpDone();
        if (op.predecessor != 0x0) {
            if (!ops[op.predecessor].done) revert PredecessorIncomplete();
        }
        if (block.timestamp < op.readyAt) revert OpNotReady();
 
        // Perform call
        (bool ok, bytes memory ret) = target.call{value: value}(data);
        require(ok, _getRevertMsg(ret));
 
        op.done = true;
 
        emit OperationExecuted(id, msg.sender, target, value, data);
        _log(214 /*OperationExecuted*/, keccak256(abi.encode(id, target, value, keccak256(data), msg.sender)));
 
        return ret;
    }
 
    /// @notice Cancel a queued (not executed) operation.
    function cancel(
        address target,
        uint256 value,
        bytes calldata data,
        bytes32 predecessor,
        bytes32 salt
    ) external nonReentrant {
        bytes32 id = hashOperation(target, value, data, predecessor, salt);
        Operation storage op = ops[id];
        if (op.readyAt == 0) revert OpMissing();
        if (op.done) revert OpDone();
 
        // Only governance or original proposer may cancel
        if (msg.sender != owner() && msg.sender != dao && msg.sender != op.proposer) revert NotGov();
 
        delete ops[id];
 
        emit OperationCanceled(id, msg.sender);
        _log(215 /*OperationCanceled*/, keccak256(abi.encode(id, msg.sender)));
    }
 
    /* ───────── Utils ───────── */
 
    function _getRevertMsg(bytes memory ret) internal pure returns (string memory) {
        if (ret.length < 68) return "timelock: call failed";
        assembly {
            ret := add(ret, 0x04)
        }
        return abi.decode(ret, (string));
    }
 
    receive() external payable {}
    fallback() external payable {}
 
    /* ───────── Logging helper ───────── */
    function _log(uint8 etype, bytes32 ctx) internal {
        if (address(proof) != address(0)) {
            try proof.logEvent(address(this), etype, ctx) {} catch {}
        }
    }
}
