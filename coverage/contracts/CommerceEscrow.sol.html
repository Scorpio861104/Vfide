/* ─────────────────────── CommerceEscrow (VFIDE Trust Escrow) ───────────────────────
   - Vault-native: buyer funds flow from buyer's UserVault → this escrow; seller receives to seller's UserVault.
   - Seer trust gates: blocklisted actors forbidden; min ProofScore enforced for merchants via registry.
   - Risk channel: SeerTuning nudges score; PanicGuard can quarantine high-risk parties.
   - Breaker-aware: all mutations blocked while system/module paused.
   - ProofLedger logging: best-effort for every lifecycle change.
 
   Flow:
     1) open(...) by buyer (or front-end) defines buyer, seller, amount, timeouts.
     2) fund(escrowId) pulls VFIDE from buyerVault via transferFrom (buyer must approve this contract in their vault).
     3) release(...) sends funds to sellerVault (or refund(...) back to buyerVault).
     4) disputes: buyer/seller can openDispute before release deadline; resolveDispute by DAO/arbitrator splits funds.
     5) auto-release on timeout if enabled and no dispute.
 
   Notes:
     - We keep a simple state machine and cap risk on-chain. Deep fraud analytics should be handled by DarkTrust/Seer.
     - Module id assumption for Breaker: Merchants = 4 (per earlier enum), Router/Funding = 7.
------------------------------------------------------------------------------------- */

    function _checkParties(address buyer, address seller) internal view {
        if (seer.isBlocked(buyer) || seer.isBlocked(seller)) revert TrustGate();

        if (!merchants.isMerchant(seller)) revert TrustGate();
        uint16 need = _sellerMinScore();
        if (need > 0 && seer.score(seller) < need) revert TrustGate();

        address seller,
        uint256 amount,
        string calldata memo,

    /// @notice Create an escrow record; funds are not moved until `fund(...)`.
    /// @param autoReleaseAt  unix ts when auto-release is allowed (0 to disable)
    /// @param autoReleaseFlag    if true, anyone can call `autoRelease(...)` after autoReleaseAt + dispute buffer
    function open(
        address buyer,
        address seller,
        uint256 amount,
        uint64 autoReleaseAt,
    bool autoReleaseFlag,
        string calldata memo,
        bytes32 salt
    ) external breakerGuard nonReentrant returns (bytes32 id) {
        require(buyer != address(0) && seller != address(0), "addr=0");
        require(amount > 0, "amt=0");
        _checkParties(buyer, seller);
 
        // time bounds

            require(autoReleaseAt > block.timestamp, "auto ts");
            uint64 dur = autoReleaseAt - uint64(block.timestamp);
            if (dur < minEscrowDuration || dur > maxEscrowDuration) revert TimeRange();
        }

        if (escrows[id].status != Status.None) revert Exists();
 
        Escrow storage e = escrows[id];
        e.buyer       = buyer;
        e.seller      = seller;
        e.amount      = amount;
        e.createdAt   = uint64(block.timestamp);
    e.autoReleaseAt = autoReleaseFlag ? autoReleaseAt : 0;
    e.autoRelease = autoReleaseFlag;
        e.status      = Status.Open;
        e.memo        = memo;
 
    emit Opened(id, buyer, seller, amount, e.autoReleaseAt, autoReleaseFlag, memo);
    _log(60 /*Opened*/, keccak256(abi.encode(id, buyer, seller, amount, autoReleaseAt, autoReleaseFlag)));
    }
 
    /// @notice Pull funds from buyer's vault into escrow; buyer's vault must `approve` this contract for `amount`.
    function fund(bytes32 id) external breakerGuard nonReentrant {
        Escrow storage e = escrows[id];
        if (e.status != Status.Open) revert BadState();
 
        address buyerVault = _vaultOf(e.buyer);
        uint256 needed = e.amount - fundedAmount[id];
        if (needed == 0) revert Amount();
 
        uint256 allowance = vfide.allowance(buyerVault, address(this));
        if (allowance < needed) revert NoAllowance();
 
        bool ok = vfide.transferFrom(buyerVault, address(this), needed);
        require(ok, "pull fail");
 
        fundedAmount[id] += needed;
        e.status = Status.Funded;
 
        emit Funded(id, buyerVault, needed, fundedAmount[id]);
        _log(61 /*Funded*/, keccak256(abi.encode(id, buyerVault, needed, fundedAmount[id])));
    }
 
    /// @notice Buyer opens a dispute before release/refund.
    function openDispute(bytes32 id, string calldata reason) external breakerGuard nonReentrant {
        Escrow storage e = escrows[id];
        if (e.status != Status.Funded) revert BadState();
        if (msg.sender != e.buyer && msg.sender != e.seller) revert NotParty();
        Dispute storage d = disputes[id];
        if (d.open) revert AlreadyDisputed();
 
        d.open = true;
        d.openedBy = msg.sender;
        d.openedAt = uint64(block.timestamp);
        d.reason = reason;
 
        e.status = Status.Disputed;
 
        // Risk actions (optional)
        _penalize(msg.sender, -1); // tiny nudge for opening; neutral in aggregate
        _flag(msg.sender, 1 /*dispute*/, true);
 
        emit DisputeOpened(id, msg.sender, reason);
        _log(62 /*DisputeOpened*/, keccak256(abi.encode(id, msg.sender, keccak256(bytes(reason)))));
    }
 
    /// @notice Close dispute without resolution (settled off-chain). Moves back to Funded.
    function closeDispute(bytes32 id) external breakerGuard nonReentrant {
        Escrow storage e = escrows[id];
        if (e.status != Status.Disputed) revert BadState();
        if (msg.sender != e.buyer && msg.sender != e.seller) revert NotParty();
 
        disputes[id].open = false;
        e.status = Status.Funded;
 
        _flag(msg.sender, 1 /*dispute*/, false);
 
        emit DisputeClosed(id);
        _log(63 /*DisputeClosed*/, keccak256(abi.encode(id, msg.sender)));
    }
 
    /// @notice Release to seller's vault (buyer or seller can call, but only when no dispute).
    function release(bytes32 id) external breakerGuard nonReentrant {
        _releaseFunds(id, msg.sender);
    }
 
    function _releaseFunds(bytes32 id, address caller) internal {
        Escrow storage e = escrows[id];
        if (e.status != Status.Funded) revert BadState();
        if (caller != e.buyer && caller != e.seller) revert NotParty();
 
        // Re-check seller trust (no recent block) before final release
        if (seer.isBlocked(e.seller)) revert TrustGate();
 
        address sellerVault = _vaultOf(e.seller);
        uint256 amt = fundedAmount[id];
 
        fundedAmount[id] = 0;
        e.status = Status.Released;
 
        bool ok = vfide.transfer(sellerVault, amt);
        require(ok, "xfer fail");
 
        // Rewards (optional)
        _reward(e.buyer, rewardOnSuccess);
        _reward(e.seller, rewardOnSuccess);
 
        emit Released(id, sellerVault, amt);
        _log(64 /*Released*/, keccak256(abi.encode(id, sellerVault, amt)));
    }
 
    /// @notice Refund back to buyer's vault (buyer or seller can call, but only when no dispute).
    function refund(bytes32 id) external breakerGuard nonReentrant {
        Escrow storage e = escrows[id];
        if (e.status != Status.Funded) revert BadState();
        if (msg.sender != e.buyer && msg.sender != e.seller) revert NotParty();
 
        address buyerVault = _vaultOf(e.buyer);
        uint256 amt = fundedAmount[id];
 
        fundedAmount[id] = 0;
        e.status = Status.Refunded;
 
        bool ok = vfide.transfer(buyerVault, amt);
        require(ok, "xfer fail");
 
        // Penalize seller slightly for failed transaction
        _penalize(e.seller, penaltyOnRefund);
 
        emit Refunded(id, buyerVault, amt);
        _log(65 /*Refunded*/, keccak256(abi.encode(id, buyerVault, amt)));
    }
 
    /// @notice Auto-release anyone after deadline (if enabled, no dispute).
    function autoRelease(bytes32 id) external breakerGuard nonReentrant {
        Escrow storage e = escrows[id];
        if (!e.autoRelease) revert BadState();
        if (e.status != Status.Funded) revert BadState();
        require(e.autoReleaseAt > 0 && block.timestamp >= e.autoReleaseAt + disputeWindowBuffer, "not due");
        require(!disputes[id].open, "disputed");
        _releaseFunds(id, msg.sender);
    }
 
    /// @notice DAO/arbitrator resolves dispute with split amounts.
    function resolveDispute(bytes32 id, uint256 toSeller, uint256 toBuyer, bool sellerAtFault) external breakerGuard nonReentrant onlyOwner {
        Escrow storage e = escrows[id];
        if (e.status != Status.Disputed) revert BadState();
        Dispute storage d = disputes[id];
        if (!d.open) revert NoDispute();
 
        uint256 total = fundedAmount[id];
        require(toSeller + toBuyer == total, "sum != total");
 
        address sellerVault = _vaultOf(e.seller);
        address buyerVault  = _vaultOf(e.buyer);
 
        fundedAmount[id] = 0;
        d.open = false;
        e.status = (toSeller > 0) ? Status.Released : Status.Refunded;
 
        if (toSeller > 0) require(vfide.transfer(sellerVault, toSeller), "xfer seller");
        if (toBuyer  > 0) require(vfide.transfer(buyerVault,  toBuyer),  "xfer buyer");
 
        // Penalties
        if (sellerAtFault) {
            _penalize(e.seller, penaltyOnBadActor);
            _quarantine(e.seller);
        } else {
            _penalize(e.buyer, penaltyOnBadActor);
            _quarantine(e.buyer);
        }
        // Small reward to the non-fault party
        _reward(sellerAtFault ? e.buyer : e.seller, rewardOnSuccess);
 
        emit Resolved(id, toSeller, toBuyer, msg.sender);
        _log(66 /*Resolved*/, keccak256(abi.encode(id, toSeller, toBuyer, sellerAtFault, msg.sender)));
    }
 
    /// @notice Cancel an open (unfunded) escrow
    function cancel(bytes32 id) external breakerGuard nonReentrant {
        Escrow storage e = escrows[id];
        if (e.status != Status.Open) revert BadState();
        if (msg.sender != e.buyer && msg.sender != e.seller) revert NotParty();
        e.status = Status.Canceled;
        emit Canceled(id);
        _log(67 /*Canceled*/, keccak256(abi.encode(id, msg.sender)));
    }

    function _reward(address user, int16 delta) internal {
        if (delta == 0) return;
        if (address(seerTune) != address(0)) {
            try seerTune.adjustScore(user, delta) {} catch {}
        }
    }
 
    function _penalize(address user, int16 delta) internal {
        if (delta == 0) return;
        if (address(seerTune) != address(0)) {
            try seerTune.adjustScore(user, delta) {} catch {}
        }
    }
 
    function _flag(address user, uint8 code, bool on) internal {
