import "@openzeppelin/contracts/security/ReentrancyGuard.sol";
 
/* Minimal interfaces */
interface ISeerView { function score(address user) external view returns (uint16); }
interface IProofLedger { function logEvent(address user, uint8 etype, bytes32 ctx) external; }
interface IEscrow {
    enum State { None, Initiated, Shipped, Delivered, Disputed, Released, Refunded }
    function orders(uint256 id) external view returns (
        address buyer, address merchant, uint256 amount,
        IEscrow.State state, uint64 tInitiated, uint64 tShipped, uint64 tDelivered, uint64 tDisputed, string memory meta
    );
}
 
/**
 * @title ReviewRegistry
 * @notice Purchase-verified, score-weighted merchant reviews.
 */
contract ReviewRegistry is Ownable2Step, ReentrancyGuard {
    ISeerView public seer;
    IProofLedger public proof;
    IEscrow public escrow;
    address public dao;
 
    uint64 public editGraceSecs = 2 days;
    bool   public allowEdits = true;
 
    struct Review {
        address buyer;
        address merchant;
        uint8   stars;        // 1..5
        int16   weightBp;     // weight in basis points (e.g., 1000 = 1.0x)
        string  text;         // short text (off-chain pointers recommended)
        uint64  createdAt;
        uint64  editedAt;
        bool    hidden;       // DAO moderation
    }
    // orderId -> review
    mapping(uint256 => Review) public reviewOfOrder;
 
    // aggregate per merchant
    struct Aggregate {
        uint256 sumWeighted; // sum(stars * weightBp)
        uint256 sumWeight;   // sum(weightBp)
    }
    mapping(address => Aggregate) public aggOf;
 
    event DAOUpdated(address dao);
    event SeerSet(address seer);
    event ProofSinkSet(address proof);
    event EscrowSet(address escrow);
    event SettingsUpdated(uint64 editGraceSecs, bool allowEdits);
 
    event Reviewed(uint256 indexed orderId, address indexed buyer, address indexed merchant, uint8 stars, int16 weightBp);
    event ReviewEdited(uint256 indexed orderId, uint8 newStars);
    event ReviewHidden(uint256 indexed orderId, bool hidden);
 
    error ZeroAddress();
    error NotGov();
    error NotBuyer();
    error BadStars();
    error NotCompleted();
    error AlreadyReviewed();
    error EditWindowClosed();
 
    modifier onlyGov() { if (msg.sender != owner() && msg.sender != dao) revert NotGov(); _; }
 
    constructor(address initialOwner) { Iif (initialOwner==address(0)) revert ZeroAddress(); _transferOwnership(initialOwner); }
 
    /* wiring */
    function setDAO(address d) external onlyGov { if (d==address(0)) revert ZeroAddress(); dao=d; emit DAOUpdated(d); }
    function setSeer(address s) external onlyGov { if (s==address(0)) revert ZeroAddress(); seer=ISeerView(s); emit SeerSet(s); }
    function setProofSink(address p) external onlyGov { proof = IProofLedger(p); emit ProofSinkSet(p); }
    function setEscrow(address e) external onlyGov { if (e==address(0)) revert ZeroAddress(); escrow=IEscrow(e); emit EscrowSet(e); }
    function setSettings(uint64 grace, bool edits) external onlyGov { editGraceSecs=grace; allowEdits=edits; emit SettingsUpdated(grace, edits); }
 
    /* core */
    function leaveReview(uint256 orderId, uint8 stars, string calldata text) external nonReentrant {
        if (stars<1 || stars>5) revert BadStars();
        (address buyer, address merchant,, IEscrow.State st,,,,,)= escrow.orders(orderId);
        if (buyer != msg.sender) revert NotBuyer();
        // only completed orders (Released or Refunded) can be reviewed
        if (!(st == IEscrow.State.Released || st == IEscrow.State.Refunded)) revert NotCompleted();
        if (reviewOfOrder[orderId].buyer != address(0)) revert AlreadyReviewed();
 
        int16 w = _weightFor(buyer); // e.g., 500..1500 bp
        reviewOfOrder[orderId] = Review({
            buyer: buyer, merchant: merchant, stars: stars, weightBp: w,
            text: text, createdAt: uint64(block.timestamp), editedAt: 0, hidden: false
        });
        _addAggregate(merchant, stars, w);
        emit Reviewed(orderId, buyer, merchant, stars, w);
        _log(buyer, 40 /*ReviewCreated*/, keccak256(abi.encode(orderId, stars)));
    }
 
    function editReview(uint256 orderId, uint8 newStars, string calldata newText) external nonReentrant {
        if (!allowEdits) revert EditWindowClosed();
        if (newStars<1 || newStars>5) revert BadStars();
        Review storage r = reviewOfOrder[orderId];
        if (r.buyer != msg.sender) revert NotBuyer();
        require(block.timestamp <= r.createdAt + editGraceSecs, "grace over");
        // adjust aggregate
        _subAggregate(r.merchant, r.stars, r.weightBp);
        r.stars = newStars;
        r.text  = newText;
        r.editedAt = uint64(block.timestamp);
        _addAggregate(r.merchant, r.stars, r.weightBp);
        emit ReviewEdited(orderId, newStars);
    }
 
    function hideReview(uint256 orderId, bool hidden) external onlyGov {
        Review storage r = reviewOfOrder[orderId];
        r.hidden = hidden;
        emit ReviewHidden(orderId, hidden);
    }
 
    /* views */
    function weightedRating(address merchant) external view returns (uint256 numerator, uint256 denominator, uint256 scaledStars1e2) {
        Aggregate memory a = aggOf[merchant];
        numerator = a.sumWeighted; denominator = a.sumWeight;
        scaledStars1e2 = (denominator==0) ? 0 : (numerator * 100) / denominator; // “stars * 100”
    }
 
    /* internals */
    function _weightFor(address buyer) internal view returns (int16) {
        if (address(seer) == address(0)) return 1000; // 1.0x if unset
        uint16 s = seer.score(buyer); // 0..1000
        // map 0..1000 → 500..1500 (0.5x..1.5x)
    int256 w = 500 + (int256(uint256(s)) * 1000) / 1000;
        return int16(uint16(uint256(w))); // safe range
    }
 
    function _addAggregate(address m, uint8 stars, int16 wBp) internal {
        Aggregate storage a = aggOf[m];
        a.sumWeighted += uint256(stars) * uint256(uint16(wBp));
        a.sumWeight   += uint256(uint16(wBp));
    }
    function _subAggregate(address m, uint8 stars, int16 wBp) internal {
        Aggregate storage a = aggOf[m];
        a.sumWeighted -= uint256(stars) * uint256(uint16(wBp));
        a.sumWeight   -= uint256(uint16(wBp));
    }
 
    function _log(address user, uint8 et, bytes32 ctx) internal { if (address(proof)!=address(0)) { try proof.logEvent(user, et, ctx) {} catch {} } }
}
