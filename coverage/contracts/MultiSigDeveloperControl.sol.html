pragma solidity 0.8.30;
 
import "@openzeppelin/contracts/security/ReentrancyGuard.sol";
 
/**
 * @title MultiSigDeveloperControl
 * @notice N-of-M multisig for pre-handover administration that can transition to DAO-only control.
 *         - Pre-handover: classic multisig (owners[] + threshold).
 *         - Timelocked execution (minDelay) for all txs.
 *         - Strict allowlist (targets + selectors) ON by default.
 *         - DAO Mode: once activated, only daoExecutor can submit/execute; signer confirmations disabled.
 *
 * Usage after deployment:
 *  1) owners set via constructor; threshold set.
 *  2) Configure allowlist (targets & selectors) and minDelay via multisig proposals.
 *  3) When ready, call `activateDAOMode(dao)` via multisig. From then on, DAO controls execution.
 */
contract MultiSigDeveloperControl is ReentrancyGuard {
    /* ───────────────────────────── State ───────────────────────────── */
 
    // Multisig owners (pre-handover)
    mapping(address => bool) public isOwner;
    address[] public owners;
    uint256 public threshold; // required confirmations pre-handover
 
    // DAO control (post-handover)
    address public daoExecutor;
    bool    public daoMode; // when true: only daoExecutor can submit/execute; multisig confirmations disabled
 
    // Timelock
    uint64  public minDelay = 24 hours;
 
    // Allowlist (strict by default)
    bool public whitelistStrict = true;
    mapping(address => bool) public allowedTarget;              // contract address allowlist
    mapping(bytes4  => bool) public allowedSelector;            // function selector allowlist
 
    // Transactions
    struct Tx {
        address target;
        uint256 value;
        bytes   data;
        bool    executed;
        uint64  eta; // earliest execution time (now + minDelay at submit)
    }
 
    Tx[] public txs;
    mapping(uint256 => mapping(address => bool)) public confirmedBy; // txId => owner => confirmed
    mapping(uint256 => uint256) public confirmations;                // txId => count
 
    /* ───────────────────────────── Events ─────────────────────────── */
 
    event OwnerAdded(address indexed owner);
    event OwnerRemoved(address indexed owner);
    event ThresholdChanged(uint256 threshold);
    event MinDelayChanged(uint64 minDelay);
    event WhitelistStrictSet(bool strict);
    event TargetAllowed(address indexed target, bool allowed);
    event SelectorAllowed(bytes4 indexed selector, bool allowed);
 
    event DAOModeActivated(address indexed daoExecutor);
    event DAOModeDeactivated(); // optional escape hatch if you ever need to revert (via DAO)
 
    event Submitted(uint256 indexed txId, address indexed submitter, address indexed target, uint256 value, bytes data, uint64 eta);
    event Confirmed(uint256 indexed txId, address indexed owner);
    event Revoked(uint256 indexed txId, address indexed owner);
    event Executed(uint256 indexed txId, address indexed executor, bytes result);
 
    /* ───────────────────────────── Errors ─────────────────────────── */
 
    error NotOwner();
    error AlreadyOwner();
    error InvalidThreshold();
    error ZeroAddress();
    error OnlyDAO();
    error NotDAO();
    error NotInDAOMode();
    error AlreadyInDAOMode();
    error TxNotFound();
    error AlreadyExecuted();
    error NotConfirmed();
    error TimelockNotExpired();
    error NotWhitelistedTarget();
    error NotWhitelistedSelector();
 
    /* ─────────────────────────── Modifiers ───────────────────────── */
 
    modifier onlyOwner() {
        if (!isOwner[msg.sender]) revert NotOwner();
        _;
    }
    modifier onlyDAO() {
        if (msg.sender != daoExecutor) revert OnlyDAO();
        _;
    }
 
    /* ───────────────────────── Constructor ───────────────────────── */
 
    /**
     * @param owners_       initial owner set (e.g., 3 or 5 dev keys)
     * @param threshold_    required confirmations (e.g., 2 or 3)
     * @param initialDelay  min timelock delay in seconds (e.g., 86400)
     */

    }
 
    /* ─────────────────────── View helpers ─────────────────────── */
 
    function ownersList() external view returns (address[] memory) { return owners; }
    function txCount() external view returns (uint256) { return txs.length; }
 
    /* ─────────────────────── Internal helpers ─────────────────────── */
 
    function _requireWhitelist(address target, bytes memory data) internal view {
        if (!whitelistStrict) return; // no checks
        if (!allowedTarget[target]) revert NotWhitelistedTarget();
        if (data.length >= 4) {
            bytes4 sel;
            assembly { sel := mload(add(data, 32)) }
            if (!allowedSelector[sel]) revert NotWhitelistedSelector();
        } else {
            // No selector (empty data) – treat as allowed only if explicit selector(0x00000000) is allowed.
            if (!allowedSelector[bytes4(0)]) revert NotWhitelistedSelector();
        }
    }
 
    function _isConfirmed(uint256 txId) internal view returns (bool) {
        return confirmations[txId] >= threshold;
    }
 
    /* ─────────────────────── Config (self-managed) ───────────────────────
       All these functions are intended to be called VIA multisig itself:
       i.e., create a tx that calls this contract with the encoded data below.
       In DAO mode, the DAO can call them directly (no confirmations).
    */
 
    // Add a new owner (multisig/DAO callable)
    function addOwner(address owner) external {
        if (daoMode) {
            if (msg.sender != daoExecutor) revert OnlyDAO();
        } else {
            if (!isOwner[msg.sender]) revert NotOwner(); // must be executed via this contract (see executeTx)
            // NOTE: In practice, this should be called as an internal call via submit/execute.
        }
        if (owner == address(0)) revert ZeroAddress();
        if (isOwner[owner]) revert AlreadyOwner();
        isOwner[owner] = true;
        owners.push(owner);
        emit OwnerAdded(owner);
        // threshold remains unchanged; if you want to increase, call changeThreshold.
    }
 
    // Remove an owner
    function removeOwner(address owner) external {
        if (daoMode) {
            if (msg.sender != daoExecutor) revert OnlyDAO();
        } else {
            if (!isOwner[msg.sender]) revert NotOwner();
        }
        if (!isOwner[owner]) revert NotOwner();
        isOwner[owner] = false;
 
        // compact owners array
        uint256 len = owners.length;
        for (uint256 i = 0; i < len; i++) {
            if (owners[i] == owner) {
                owners[i] = owners[len - 1];
                owners.pop();
                break;
            }
        }
 
        // ensure threshold still valid
        if (threshold > owners.length) {
            threshold = owners.length;
            emit ThresholdChanged(threshold);
        }
        emit OwnerRemoved(owner);
    }
 
    // Change confirmation threshold
    function changeThreshold(uint256 newThreshold) external {
        if (daoMode) {
            if (msg.sender != daoExecutor) revert OnlyDAO();
        } else {
            if (!isOwner[msg.sender]) revert NotOwner();
        }
        if (newThreshold == 0 || newThreshold > owners.length) revert InvalidThreshold();
        threshold = newThreshold;
        emit ThresholdChanged(newThreshold);
    }
 
    // Set timelock delay
    function setMinDelay(uint64 newDelay) external {
        if (daoMode) {
            if (msg.sender != daoExecutor) revert OnlyDAO();
        } else {
            if (!isOwner[msg.sender]) revert NotOwner();
        }
        minDelay = newDelay;
        emit MinDelayChanged(newDelay);
    }
 
    // Toggle whitelist strictness
    function setWhitelistStrict(bool strict) external {
        if (daoMode) {
            if (msg.sender != daoExecutor) revert OnlyDAO();
        } else {
            if (!isOwner[msg.sender]) revert NotOwner();
        }
        whitelistStrict = strict;
        emit WhitelistStrictSet(strict);
    }
 
    // Allow / disallow a target contract
    function setAllowedTarget(address target, bool allowed) external {
        if (daoMode) {
            if (msg.sender != daoExecutor) revert OnlyDAO();
        } else {
            if (!isOwner[msg.sender]) revert NotOwner();
        }
        if (target == address(0)) revert ZeroAddress();
        allowedTarget[target] = allowed;
        emit TargetAllowed(target, allowed);
    }
 
    // Allow / disallow a selector (function)
    function setAllowedSelector(bytes4 selector, bool allowed) external {
        if (daoMode) {
            if (msg.sender != daoExecutor) revert OnlyDAO();
        } else {
            if (!isOwner[msg.sender]) revert NotOwner();
        }
        allowedSelector[selector] = allowed;
        emit SelectorAllowed(selector, allowed);
    }
 
    /* ─────────────────────── DAO handover ─────────────────────── */
 
    /// @notice Activate DAO mode (irreversible in normal use). Signers no longer confirm; only DAO can operate.
    function activateDAOMode(address dao_) external onlyOwner {
        if (daoMode) revert AlreadyInDAOMode();
        if (dao_ == address(0)) revert ZeroAddress();
        daoExecutor = dao_;
        daoMode = true;
        emit DAOModeActivated(dao_);
    }
 
    /// @notice Optional: allow DAO to temporarily disable daoMode (escape hatch).
    function deactivateDAOMode() external onlyDAO {
        daoMode = false;
        emit DAOModeDeactivated();
    }
 
    /* ─────────────────────── Multisig / DAO ops ─────────────────────── */
 
    /**
     * @notice Submit a new transaction (multisig pre-handover) or by DAO in daoMode.
     * @param target   destination contract (must be allowlisted if whitelistStrict)
     * @param value    ETH value to send
     * @param data     calldata (selector + args) (selector must be allowlisted if strict)
     */
    function submitTx(address target, uint256 value, bytes calldata data) external returns (uint256 txId) {
        if (daoMode) {
            if (msg.sender != daoExecutor) revert OnlyDAO();
        } else {
            if (!isOwner[msg.sender]) revert NotOwner();
        }
        _requireWhitelist(target, data);
 
        uint64 eta = uint64(block.timestamp) + minDelay;
        txId = txs.length;
        txs.push(Tx({
            target: target,
            value: value,
            data: data,
            executed: false,
            eta: eta
        }));
 
        emit Submitted(txId, msg.sender, target, value, data, eta);
 
        // In pre-handover mode, the submitter may auto-confirm.
        if (!daoMode) {
            _confirm(txId, msg.sender);
        }
    }
 
    /// @notice Confirm a tx (pre-handover only).
    function confirmTx(uint256 txId) external onlyOwner {
        if (daoMode) revert NotInDAOMode();
        _confirm(txId, msg.sender);
    }
 
    function _confirm(uint256 txId, address owner) internal {
        if (txId >= txs.length) revert TxNotFound();
        if (txs[txId].executed) revert AlreadyExecuted();
        if (!confirmedBy[txId][owner]) {
            confirmedBy[txId][owner] = true;
            confirmations[txId] += 1;
            emit Confirmed(txId, owner);
        }
    }
 
    /// @notice Revoke confirmation (pre-handover only).
    function revokeConfirmation(uint256 txId) external onlyOwner {
        if (daoMode) revert NotInDAOMode();
        if (txId >= txs.length) revert TxNotFound();
        if (txs[txId].executed) revert AlreadyExecuted();
        if (confirmedBy[txId][msg.sender]) {
            confirmedBy[txId][msg.sender] = false;
            confirmations[txId] -= 1;
            emit Revoked(txId, msg.sender);
        }
    }
 
    /**
     * @notice Execute a timelocked transaction.
     *         - Pre-handover: requires threshold confirmations.
     *         - DAO mode: only daoExecutor may execute; no confirmations required.
     *         - Always enforces ETA and whitelist (if strict).
     */
    function executeTx(uint256 txId) external nonReentrant returns (bytes memory result) {
        if (txId >= txs.length) revert TxNotFound();
        Tx storage t = txs[txId];
        if (t.executed) revert AlreadyExecuted();
        if (block.timestamp < t.eta) revert TimelockNotExpired();
 
        _requireWhitelist(t.target, t.data);
 
        if (daoMode) {
            if (msg.sender != daoExecutor) revert OnlyDAO();
        } else {
            if (!_isConfirmed(txId)) revert NotConfirmed();
        }
 
        t.executed = true;
 
        // Effects done; perform external call
        (bool ok, bytes memory res) = t.target.call{value: t.value}(t.data);
        require(ok, _getRevertMsg(res));
 
        emit Executed(txId, msg.sender, res);
        return res;
    }
 
    /* ─────────────────────── Utils ─────────────────────── */
 
    receive() external payable {}
 
    function _getRevertMsg(bytes memory res) private pure returns (string memory) {
        if (res.length < 68) return "call failed";
        assembly {
            res := add(res, 0x04)
        }
        return abi.decode(res, (string));
    }
