pragma solidity 0.8.30;
 
import "@openzeppelin/contracts/access/Ownable2Step.sol";
import "@openzeppelin/contracts/security/ReentrancyGuard.sol";
import "@openzeppelin/contracts/token/ERC20/IERC20.sol";
 
/**
 * @title UserVault
 * @notice Per-user custody. Wallet = identity; Vault = funds. Only vault↔vault moves.
 * - Owner can approve system contracts (staking/router/market) to pull from the vault.
 * - Sends only to other registered vaults via factory registry.
 * - Logs to ProofLedger (best-effort).
 */
interface IProofLedgerUV {
    enum EventType {
        Unknown,
        VaultDeposit,
        VaultWithdraw,
        NextOfKinTransfer,
        ChainOfReturn,
        ProofPenalty,
        ProofBonus
    }
    function logEvent(address user, EventType etype, bytes32 ctx) external;
}
 
interface IVaultFactoryUV {
    function vaultOf(address user) external view returns (address);
    function isVault(address v) external view returns (bool);
}
 
contract UserVault is Ownable2Step, ReentrancyGuard {
    /* ---------------- Storage ---------------- */
    IERC20 public vfide;
    IProofLedgerUV public proof;
    address public guardianLock;    // optional
    address public factory;         // immutable after init
 
    bool private _inited;
    bool private _tokenLocked;
 
    /* ---------------- Events ----------------- */
    event Initialized(address owner, address proof, address guardianLock);
    event TokenSet(address token);
    event Pulled(address indexed fromOwner, uint256 amount);
    event SentToVault(address indexed toVault, uint256 amount);
    event Approved(address indexed spender, uint256 amount);
 
    /* ---------------- Errors ----------------- */
    error AlreadyInitialized();
    error ZeroAddress();
    error NotRegisteredVault(address to);
 
    /* ---------------- Modifiers -------------- */
    modifier tokenSet() {
        require(address(vfide) != address(0), "VFIDE token not set");
        _;
    }
 
    /* --------------- Initialization ---------- */
    function initialize(address owner_, address proofLedger_, address guardianLock_) external {
        if (_inited) revert AlreadyInitialized();
        if (owner_ == address(0) || proofLedger_ == address(0)) revert ZeroAddress();
        _transferOwnership(owner_);
        proof = IProofLedgerUV(proofLedger_);
        guardianLock = guardianLock_;
        factory = msg.sender; // factory is msg.sender at clone time
        _inited = true;
        emit Initialized(owner_, proofLedger_, guardianLock_);
    }
 
    /* --------------- Configuration ----------- */
    /// @notice Set the VFIDE token address once; permanently locks to prevent accidental changes.
    function setToken(address vfide_) external onlyOwner {
        if (_tokenLocked) revert("token locked");
        if (vfide_ == address(0)) revert ZeroAddress();
        vfide = IERC20(vfide_);
        _tokenLocked = true;
        emit TokenSet(vfide_);
    }
 
    /* --------------- User flows -------------- */
 
    /// @notice Pull VFIDE from owner EOA into the vault (requires EOA approval to this vault).
    function pullFromOwner(uint256 amount) external nonReentrant onlyOwner tokenSet {
        require(vfide.transferFrom(owner(), address(this), amount), "pull fail");
        _log(IProofLedgerUV.EventType.VaultDeposit, keccak256(abi.encode(amount)));
        emit Pulled(owner(), amount);
    }
 
    /// @notice Send VFIDE to another registered vault (never to an EOA).
    function sendToVault(address toUser, uint256 amount) external nonReentrant onlyOwner tokenSet {
        address toVault = IVaultFactoryUV(factory).vaultOf(toUser);
        if (toVault == address(0) || !IVaultFactoryUV(factory).isVault(toVault)) {
            revert NotRegisteredVault(toVault);
        }
        require(vfide.transfer(toVault, amount), "xfer fail");
        _log(IProofLedgerUV.EventType.VaultWithdraw, keccak256(abi.encode(amount, toVault)));
        emit SentToVault(toVault, amount);
    }
 
    /// @notice Approve a system contract (staking/router/market) to pull VFIDE from this vault.
    /// Uses safe-approve pattern to avoid allowance race issues.
    function approveSpender(address spender, uint256 amount) external onlyOwner tokenSet {
        uint256 curr = vfide.allowance(address(this), spender);
        if (curr != 0 && amount != 0) {
            require(vfide.approve(spender, 0), "approve reset fail");
        }
        require(vfide.approve(spender, amount), "approve fail");
        emit Approved(spender, amount);
    }
 
    /* --------------- Rescue (owner-only ops) -------------- */
 
    /// @notice Rescue arbitrary ERC20 tokens mistakenly sent here (including non-VFIDE).
    /// For VFIDE, the token's own vault-only policy still applies (EOA rescues will fail).
    function rescueERC20(address token_, address to, uint256 amount) external onlyOwner {
        if (to == address(0)) revert ZeroAddress();
        require(IERC20(token_).transfer(to, amount), "rescue fail");
    }
 
    /// @notice Accept ETH (if ever sent).
    receive() external payable {}
 
    /// @notice Rescue ETH sent here by mistake.
    function rescueETH(address payable to, uint256 amount) external onlyOwner {
        if (to == address(0)) revert ZeroAddress();
        (bool ok, ) = to.call{value: amount}("");
        require(ok, "eth rescue fail");
    }
 
    /* --------------- Internals --------------- */
    function _log(IProofLedgerUV.EventType et, bytes32 ctx) internal {
        // best-effort logging; swallow failures
        try proof.logEvent(owner(), et, ctx) {} catch {}
    }
}
