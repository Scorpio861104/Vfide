/* ─────────────────────────────────────────── MerchantPortal ───────────────────────────────────────────
 
Roles & Guarantees:
- No funds are custodied here. Payments remain vault→escrow→vault per CommerceEscrow.
- Safety checks:
  * Registry.canAcceptPayment(...) is called before initiating an order.
  * Vault-only custody and PanicGuard checks happen inside the Vault/Token/Guard stack you already wired.
- ProofLedger logging is best-effort (optional).
 
-------------------------------------------------------------------------------------------------------*/
contract MerchantPortal is Ownable2Step, ReentrancyGuard {
    /* ───────── Wiring ───────── */
    IVaultFactoryView public factory;
    IMerchantRegistry public registry;
    ICommerceEscrow   public escrow;
    IReviewRegistry   public reviews;
    ISeerView         public seer;        // for dashboards
    IProofLedger      public proof;       // optional audit sink
    address           public dao;         // post-handover governance
 
    /* ───────── Events ───────── */
    event DAOUpdated(address dao);
    event FactorySet(address factory);
    event RegistrySet(address registry);
    event EscrowSet(address escrow);
    event ReviewsSet(address reviews);
    event SeerSet(address seer);
    event ProofSinkSet(address proof);
 
    event PurchaseInitiated(uint256 indexed orderId, address indexed buyer, address indexed merchant, uint256 amount, string meta);
    event Shipped(uint256 indexed orderId);
    event Delivered(uint256 indexed orderId);
    event Disputed(uint256 indexed orderId, string reason);
 
    /* ───────── Errors ───────── */
    error ZeroAddress();
    error NotGov();
    error RegistryRejects(string why);
    error MerchantNotActive();
 
    modifier onlyGov() {
        if (msg.sender != owner() && msg.sender != dao) revert NotGov();
        _;
    }

        Iif (initialOwner == address(0)) revert ZeroAddress();

    /* ───────── Governance wiring ───────── */
 
    function setDAO(address dao_) external onlyGov {
        if (dao_ == address(0)) revert ZeroAddress();
        dao = dao_;
        emit DAOUpdated(dao_);
    }
 
    function setFactory(address factory_) external onlyGov {
        if (factory_ == address(0)) revert ZeroAddress();
        factory = IVaultFactoryView(factory_);
        emit FactorySet(factory_);
    }
 
    function setRegistry(address registry_) external onlyGov {
        if (registry_ == address(0)) revert ZeroAddress();
        registry = IMerchantRegistry(registry_);
        emit RegistrySet(registry_);
    }
 
    function setEscrow(address escrow_) external onlyGov {
        if (escrow_ == address(0)) revert ZeroAddress();
        escrow = ICommerceEscrow(escrow_);
        emit EscrowSet(escrow_);
    }
 
    function setReviews(address reviews_) external onlyGov {
        if (reviews_ == address(0)) revert ZeroAddress();
        reviews = IReviewRegistry(reviews_);
        emit ReviewsSet(reviews_);
    }
 
    function setSeer(address seer_) external onlyGov {
        if (seer_ == address(0)) revert ZeroAddress();
        seer = ISeerView(seer_);
        emit SeerSet(seer_);
    }
 
    function setProofSink(address proof_) external onlyGov {

        emit ProofSinkSet(proof_);
    }
 
    /* ───────── Core purchase flow (safe orchestration) ───────── */
 
    /**
     * @notice Initiate a purchase. Buyer’s vault must have approved the Escrow contract to pull `amount`.
     * @dev    Safety: calls MerchantRegistry.canAcceptPayment(...) before Escrow.initiate.
     */
    function buy(address merchant, uint256 amount, string calldata meta)
        external
        nonReentrant
        returns (uint256 orderId)
    {
        // quick merchant sanity (optional but helpful)
        if (!registry.isMerchant(merchant)) revert MerchantNotActive();
 
        // pre-payment safety check (Registry enforces Seer gates + allow/deny + status)
        (bool ok, string memory reason) = registry.canAcceptPayment(merchant, msg.sender);
        if (!ok) revert RegistryRejects(reason);
 
        // Route into escrow (vault→escrow pull happens inside Escrow)
        orderId = escrow.initiate(merchant, amount, meta);
 
        emit PurchaseInitiated(orderId, msg.sender, merchant, amount, meta);
        _log(msg.sender, 50 /*PortalPurchase*/, keccak256(abi.encode(orderId, merchant, amount)));
    }
 
    /// @notice Merchant marks an order shipped.
    function markShipped(uint256 orderId) external nonReentrant {
        escrow.markShipped(orderId);
        emit Shipped(orderId);
    }
 
    /// @notice Buyer confirms delivery.
    function markDelivered(uint256 orderId) external nonReentrant {
        escrow.markDelivered(orderId);
        emit Delivered(orderId);
    }
 
    /// @notice Buyer disputes an order within the escrow window.
    function dispute(uint256 orderId, string calldata reason) external nonReentrant {
        escrow.dispute(orderId, reason);
        emit Disputed(orderId, reason);
    }
 
    /* ───────── Reviews (purchase-verified) ───────── */
 
    function leaveReview(uint256 orderId, uint8 stars, string calldata text) external nonReentrant {
        reviews.leaveReview(orderId, stars, text);
        _log(msg.sender, 51 /*PortalReview*/, keccak256(abi.encode(orderId, stars)));
    }
 
    function editReview(uint256 orderId, uint8 newStars, string calldata newText) external nonReentrant {
        reviews.editReview(orderId, newStars, newText);
    }
 
    /* ───────── Dashboards / Views ───────── */
 
    struct MerchantSummary {
        // Registry
        IMerchantRegistry.Status status;
        address payoutVault;
        string  name;
        bytes32 metaHash;
        uint64  approvedAt;
 
        // Seer
        uint16  proofScore; // 0..1000
 
        // Reviews
        uint256 ratingNum;        // sum(stars * weightBp)
        uint256 ratingDen;        // sum(weightBp)
        uint256 ratingStars1e2;   // “stars * 100” (e.g., 432 = 4.32 stars)
 
        // Escrow KPIs (rolling, from CommerceEscrow)
        uint32  kpiTotal;
        uint32  kpiDisputes;
        uint32  kpiRefunds;
        uint32  kpiLateShips;
 
        // Orders list for UI (recent ids)
        uint256[] orders;
    }
 
    /// @notice One-shot summary for a merchant dashboard.
    function merchantSummary(address merchant) external view returns (MerchantSummary memory s) {
        // Registry slice
        (
            ,               // controller
            address payoutVault_,
            IMerchantRegistry.Status status_,
            ,               // appliedAt
            uint64 approvedAt_,
            ,               // lastCheckpoint
            string memory name_,
            bytes32 metaHash_
        ) = registry.merchantOf(merchant);
 
        s.status      = status_;
        s.payoutVault = payoutVault_;
        s.name        = name_;
        s.metaHash    = metaHash_;
        s.approvedAt  = approvedAt_;
 
        // Seer score (optional if unset)
        s.proofScore = address(seer) == address(0) ? 0 : seer.score(merchant);
 
        // Reviews aggregate
        (s.ratingNum, s.ratingDen, s.ratingStars1e2) = reviews.weightedRating(merchant);
 
        // KPIs
        ICommerceEscrow.KPIs memory k = escrow.kpiOf(merchant);
        s.kpiTotal    = k.total;
        s.kpiDisputes = k.disputes;
        s.kpiRefunds  = k.refunds;
        s.kpiLateShips= k.lateShips;
 
        // Orders
        s.orders = escrow.ordersOfMerchant(merchant);
    }
 
    /// @notice Lightweight read of a single order from Escrow (convenience for UIs).
    function orderInfo(uint256 orderId)
        external
        view
        returns (

            ICommerceEscrow.State state, uint64 tInitiated, uint64 tShipped, uint64 tDelivered, uint64 tDisputed, string memory meta
        )
    {
        return escrow.orders(orderId);
    }
 
    /* ───────── DAO helpers (pass-through to Escrow; registry keeps revoke power) ───────── */
 
    /// @notice DAO resolution for a dispute (refund or release). Registry retains revoke power for merchants.
    function daoResolve(uint256 orderId, bool refundBuyer, string calldata memo) external onlyGov nonReentrant {
        escrow.daoResolve(orderId, refundBuyer, memo);
        _log(msg.sender, 52 /*PortalDAOResolve*/, keccak256(abi.encode(orderId, refundBuyer)));
    }
 
    /// @notice Anyone can finalize after auto-release timeout (exposes escrow helper for convenience).
    function finalizeAfterTimeout(uint256 orderId) external nonReentrant {
        escrow.finalizeAfterTimeout(orderId);
    }
 
    /* ───────── Internals ───────── */
 
    function _log(address user, uint8 etype, bytes32 ctx) internal {
        if (address(proof) != address(0)) {
            try proof.logEvent(user, etype, ctx) {} catch {}
        }
    }
