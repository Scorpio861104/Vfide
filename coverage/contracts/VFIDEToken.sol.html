pragma solidity 0.8.30;
 
import "@openzeppelin/contracts/token/ERC20/extensions/ERC20Burnable.sol";
import "@openzeppelin/contracts/token/ERC20/extensions/ERC20Permit.sol";
import "@openzeppelin/contracts/access/Ownable2Step.sol";
 
/**
 * @title VFIDE Token (Vault-Only Custody)
 * @notice Fixed-supply ERC20 enforcing vault-only transfers via the factory registry.
 *         Includes burn routing to ecoTreasury/sanctum with a timelocked, delta-capped split update.
 *
 * Key agreements implemented:
 * - Total supply: 200,000,000 VFIDE (minted at deploy to a genesis receiver).
 * - Vault-only custody (EOAs cannot directly hold/receive VFIDE when enabled).
 * - System exemptions for necessary contracts (presale, staking, routers, etc).
 * - Burn routing: user-triggered split across burn/eco/sanctum.
 * - Split governance: propose -> 48h timelock -> activate, with a 10% per-change max delta.
 */
interface IVaultFactoryToken {
    function isVault(address v) external view returns (bool);
}
 
contract VFIDEToken is ERC20Burnable, ERC20Permit, Ownable2Step {
    /*//////////////////////////////////////////////////////////////
                                SUPPLY
    //////////////////////////////////////////////////////////////*/
    uint256 public constant MAX_SUPPLY = 200_000_000 * 1e18;
 
    /*//////////////////////////////////////////////////////////////
                        VAULT-ONLY CUSTODY (ENFORCEMENT)
    //////////////////////////////////////////////////////////////*/
    // Enforce that tokens live in vaults (not EOAs)
    bool public vaultOnly = true;
 
    // Authoritative vault registry
    IVaultFactoryToken public vaultFactory;
 
    // System contracts allowed to bypass vault-only rule if needed
    mapping(address => bool) public systemExempt; // e.g., presale, staking, ecoTreasury, routers
 
    event VaultPolicySet(bool enabled);
    event VaultFactorySet(address factory);
    event SystemExemptSet(address indexed who, bool enabled);
 
    /*//////////////////////////////////////////////////////////////
                           BURN ROUTING / SINKS
    //////////////////////////////////////////////////////////////*/
    // Sinks
    address public ecoTreasury;
    address public sanctum;
 
    event SinksSet(address ecoTreasury, address sanctum);
 
    // Basis points denominator
    uint16 public constant BPS_DENOM = 10_000;
 
    // Current split (must sum to 10_000)
    uint16 public burnBps;     // portion burned
    uint16 public ecoBps;      // portion to ecoTreasury
    uint16 public sanctumBps;  // portion to sanctum
 
    event SplitActivated(uint16 burnBps, uint16 ecoBps, uint16 sanctumBps);
 
    // Governance guardrails for split changes
    uint16 public constant MAX_SPLIT_CHANGE_BPS = 1_000; // 10% max change per field
    uint64 public constant SPLIT_CHANGE_DELAY   = 48 hours;
 
    // Pending split proposal
    uint16 public pendingBurnBps;
    uint16 public pendingEcoBps;
    uint16 public pendingSanctumBps;
    uint64 public splitChangeEta; // timestamp after which activateSplitChange() is allowed
 
    event SplitProposed(
        uint16 burnBps,
        uint16 ecoBps,
        uint16 sanctumBps,
        uint64 eta
    );
 
    /*//////////////////////////////////////////////////////////////
                               CONSTRUCTOR
    //////////////////////////////////////////////////////////////*/
    /**
     * @param genesisReceiver Address receiving the full initial supply (treasury or a vault).
     * @param vaultFactory_   Vault factory exposing isVault() for custody enforcement.
     * @param ecoTreasury_    Initial ecoTreasury sink (can be a vault or system-exempt contract).
     * @param sanctum_        Initial sanctum sink (can be a vault or system-exempt contract).
     * @param initBurnBps     Initial burn bps.
     * @param initEcoBps      Initial eco bps.
     * @param initSanctumBps  Initial sanctum bps. (All three must sum to 10_000.)
     */
    constructor(
        address genesisReceiver,
        address vaultFactory_,
        address ecoTreasury_,
        address sanctum_,
        uint16 initBurnBps,
        uint16 initEcoBps,
        uint16 initSanctumBps
    )
        ERC20("VFIDE", "VFIDE")
        ERC20Permit("VFIDE")
    {
        Erequire(genesisReceiver != address(0), "VFIDE: genesis=0");
        Erequire(vaultFactory_   != address(0), "VFIDE: factory=0");
        _validateSplit(initBurnBps, initEcoBps, initSanctumBps);
 
        // Mint full supply
        _mint(genesisReceiver, MAX_SUPPLY);
 
        // Wire factory & sinks
        vaultFactory = IVaultFactoryToken(vaultFactory_);
        emit VaultFactorySet(vaultFactory_);
 
        ecoTreasury = ecoTreasury_;
        sanctum     = sanctum_;
        emit SinksSet(ecoTreasury_, sanctum_);
 
        // Set initial active split
        burnBps    = initBurnBps;
        ecoBps     = initEcoBps;
        sanctumBps = initSanctumBps;
        emit SplitActivated(burnBps, ecoBps, sanctumBps);
    }
 
    /*//////////////////////////////////////////////////////////////
                                ADMIN
    //////////////////////////////////////////////////////////////*/
    /// @notice Toggle the vault-only transfer policy.
    function setVaultOnly(bool enabled) external onlyOwner {
        vaultOnly = enabled;
        emit VaultPolicySet(enabled);
    }
 
    /// @notice Point to the authoritative vault factory (must expose isVault()).
    function setVaultFactory(address f) external onlyOwner {
        Erequire(f != address(0), "VFIDE: factory=0");
        vaultFactory = IVaultFactoryToken(f);
        emit VaultFactorySet(f);
    }
 
    /// @notice Exempt (or unexempt) a system address from the vault-only rule (use sparingly).
    function setSystemExempt(address who, bool enabled) external onlyOwner {
        systemExempt[who] = enabled;
        emit SystemExemptSet(who, enabled);
    }
 
    /// @notice Set sinks. If these are not vaults, you may whitelist them via setSystemExempt.
    function setSinks(address ecoTreasury_, address sanctum_) external onlyOwner {
        Erequire(ecoTreasury_ != address(0), "VFIDE: eco=0");
        Erequire(sanctum_     != address(0), "VFIDE: sanctum=0");
        ecoTreasury = ecoTreasury_;
        sanctum     = sanctum_;
        emit SinksSet(ecoTreasury_, sanctum_);
    }
 
    /*//////////////////////////////////////////////////////////////
                     SPLIT GOVERNANCE (TIMELOCKED)
    //////////////////////////////////////////////////////////////*/
    /**
     * @notice Propose a new split; enforce delta caps per field; activate after 48h.
     *         Caller must be owner (governed). New split must sum to 10_000.
     */
    function proposeSplitChange(
        uint16 newBurnBps,
        uint16 newEcoBps,
        uint16 newSanctumBps
    ) external onlyOwner {
        _validateSplit(newBurnBps, newEcoBps, newSanctumBps);
        // Enforce max delta per field (10%)
        _enforceMaxDelta(burnBps,    newBurnBps);
        _enforceMaxDelta(ecoBps,     newEcoBps);
        _enforceMaxDelta(sanctumBps, newSanctumBps);
 
        pendingBurnBps    = newBurnBps;
        pendingEcoBps     = newEcoBps;
        pendingSanctumBps = newSanctumBps;
        splitChangeEta    = uint64(block.timestamp + SPLIT_CHANGE_DELAY);
 
        emit SplitProposed(newBurnBps, newEcoBps, newSanctumBps, splitChangeEta);
    }
 
    /// @notice After the delay has passed, activate the proposed split.
    function activateSplitChange() external onlyOwner {
        Erequire(splitChangeEta != 0 && block.timestamp >= splitChangeEta, "VFIDE: split not ready");
 
        burnBps    = pendingBurnBps;
        ecoBps     = pendingEcoBps;
        sanctumBps = pendingSanctumBps;
 
        // clear pending
        pendingBurnBps    = 0;
        pendingEcoBps     = 0;
        pendingSanctumBps = 0;
        splitChangeEta    = 0;
 
        emit SplitActivated(burnBps, ecoBps, sanctumBps);
    }
 
    /*//////////////////////////////////////////////////////////////
                         USER ACTION: BURN ROUTING
    //////////////////////////////////////////////////////////////*/
    /**
     * @notice Route a user-initiated burn across sinks according to the active split.
     * @dev The caller must already hold (custody via vault policy still applies to transfers).
     *      Any rounding dust is added to the burn portion.
     */
    function burnWithRouting(uint256 amount) external {
        require(amount > 0, "VFIDE: amount=0");
 
        // Calculate portions
        uint256 toBurn    = (amount * uint256(burnBps))    / BPS_DENOM;
        uint256 toEco     = (amount * uint256(ecoBps))     / BPS_DENOM;
        uint256 toSanctum = (amount * uint256(sanctumBps)) / BPS_DENOM;
 
        // Rounding remainder → burn
        uint256 routed = toBurn + toEco + toSanctum;
        if (routed < amount) {
            toBurn += (amount - routed);
        }
 
        // Execute in-effects order: transfers (interactions) last
        // Burn from sender
        if (toBurn > 0) {
            _burn(msg.sender, toBurn);
        }
 
        // Transfer to sinks (only if non-zero). If sinks are not vaults, they must be systemExempt.
        if (toEco > 0) {
            require(ecoTreasury != address(0), "VFIDE: eco unset");
            _transfer(msg.sender, ecoTreasury, toEco);
        }
        if (toSanctum > 0) {
            require(sanctum != address(0), "VFIDE: sanctum unset");
            _transfer(msg.sender, sanctum, toSanctum);
        }
    }
 
    /*//////////////////////////////////////////////////////////////
                        TRANSFER GUARD (VAULT-ONLY)
    //////////////////////////////////////////////////////////////*/
    /**
     * @dev Enforce vault-only movements when enabled; allow mints/burns.
     * OZ v5-style transfer hook. If your stack uses OZ v4, switch to _beforeTokenTransfer.
     */

                bool toOk   = vaultFactory.isVault(to)   || systemExempt[to];
                require(fromOk && toOk, "VFIDE: vault-only");
            }

    }
 
    /*//////////////////////////////////////////////////////////////
                               HELPERS
    //////////////////////////////////////////////////////////////*/
    function _validateSplit(uint16 b, uint16 e, uint16 s) internal pure {
        require(uint256(b) + uint256(e) + uint256(s) == BPS_DENOM, "VFIDE: split != 100%");
    }
 
    function _enforceMaxDelta(uint16 oldBps, uint16 newBps) internal pure {
        // abs(new - old) <= MAX_SPLIT_CHANGE_BPS
        Eif (newBps >= oldBps) {
            Erequire(newBps - oldBps <= MAX_SPLIT_CHANGE_BPS, "VFIDE: split delta too high");
        } else {
            require(oldBps - newBps <= MAX_SPLIT_CHANGE_BPS, "VFIDE: split delta too high");
        }
    }
