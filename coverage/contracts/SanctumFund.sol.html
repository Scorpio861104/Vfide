/**
 * @title SanctumFund
 * @notice DAO-directed charitable disbursements of VFIDE to **approved** recipients only.
 *         - Sends only to recipient vaults (never EOAs).
 *         - Requires recipient to be **approved** by governance.
 *         - Seer risk-gate ON by default (blocks low/blocked recipients unless exempt).
 *         - Best-effort ProofLedger logging.
 *
 * IMPORTANT: VFIDEToken enforces vault-only transfers. Sanctum must be
 *            temporarily `systemExempt` on the token unless you make Sanctum
 *            itself a vault address.
 */
contract SanctumFund is Ownable2Step, ReentrancyGuard {
    /* -------------------- Immutable / Core -------------------- */
    address public immutable token;            // VFIDE ERC20
    IVaultFactoryView public factory;          // Vault registry (and ensure)
    ISeerView public seer;                     // Risk source (0..1000 scale)
    IProofLedger public proof;                 // Optional sink for logging
    address public dao;                        // DAO executor (post-handover)
 
    /* -------------------- Approvals (NEW) --------------------- */
    // Governance-managed allowlist of approved charity recipients (by controller EOA)
    mapping(address => bool) public approvedRecipient;
 
    /* -------------------- Risk gate (defaults ON) -------------- */
    bool   public riskGateEnabled = true;
    bool   public blockIfBlocked  = true;
    uint16 public minReceiverScore = 300;      // must be >= riskHighMin by policy
 
    // Exempt specific recipient vaults (e.g., known custodians / routers)
    mapping(address => bool) public riskExemptVault;
 
    /* -------------------- Events ------------------------------- */
    event DAOUpdated(address indexed dao);
    event ProofSinkUpdated(address indexed proof);
    event FactorySet(address indexed factory);
    event SeerSet(address indexed seer);
 
    event RiskGateToggled(bool enabled);
    event RiskParamsUpdated(bool blockIfBlocked, uint16 minReceiverScore);
    event RiskExemptVaultSet(address indexed vault, bool exempt);
 
    event RecipientApproved(address indexed recipient, bool approved);
    event RecipientsApproved(address[] recipients, bool approved);
 
    event CharityDisbursement(address indexed toVault, address indexed toUser, uint256 amount, string memo);
 
    /* -------------------- Errors ------------------------------- */
    error ZeroAddress();
    error NotApprovedRecipient();
    error RiskRejected(string reason);
 
    /* -------------------- Modifiers ---------------------------- */
    modifier onlyGov() {
        // Pre-handover: owner; Post-handover: dao
        require(msg.sender == owner() || msg.sender == dao, "SANCTUM:not-gov");
        _;
    }
 
    /* -------------------- Constructor -------------------------- */
    constructor(address token_, address initialOwner) {
        Iif (token_ == address(0) || initialOwner == address(0)) revert ZeroAddress();
        token = token_;
        _transferOwnership(initialOwner);
    }
 
    /* -------------------- Governance wiring -------------------- */
    function setDAO(address dao_) external onlyGov {
        if (dao_ == address(0)) revert ZeroAddress();
        dao = dao_;
        emit DAOUpdated(dao_);
    }
 
    function setProofSink(address proof_) external onlyGov {
        proof = IProofLedger(proof_); // 0 allowed (disable)
        emit ProofSinkUpdated(proof_);
    }
 
    function setFactory(address factory_) external onlyGov {
        if (factory_ == address(0)) revert ZeroAddress();
        factory = IVaultFactoryView(factory_);
        emit FactorySet(factory_);
    }
 
    function setSeer(address seer_) external onlyGov {
        if (seer_ == address(0)) revert ZeroAddress();
        seer = ISeerView(seer_);
        emit SeerSet(seer_);
    }
 
    /* -------------------- Approvals (NEW) ---------------------- */
    /// @notice Approve or revoke a single charity recipient (controller EOA).
    function setApprovedRecipient(address recipient, bool approved) external onlyGov {
        if (recipient == address(0)) revert ZeroAddress();
        approvedRecipient[recipient] = approved;
        emit RecipientApproved(recipient, approved);
    }
 
    /// @notice Batch approve or revoke many recipients at once.
    function setApprovedRecipients(address[] calldata recipients, bool approved) external onlyGov {
        uint256 n = recipients.length;
        for (uint256 i = 0; i < n; i++) {
            address r = recipients[i];
            if (r == address(0)) continue;
            approvedRecipient[r] = approved;
        }
        emit RecipientsApproved(recipients, approved);
    }
 
    /* -------------------- Risk-gate controls ------------------- */
    function toggleRiskGate(bool enabled) external onlyGov {
        riskGateEnabled = enabled;
        emit RiskGateToggled(enabled);
    }
 
    function setRiskParams(bool blockIfBlocked_, uint16 minReceiverScore_) external onlyGov {
        blockIfBlocked = blockIfBlocked_;
        minReceiverScore = minReceiverScore_;
        emit RiskParamsUpdated(blockIfBlocked, minReceiverScore);
    }
 
    function setRiskExemptVault(address vault, bool exempt) external onlyGov {
        riskExemptVault[vault] = exempt;
        emit RiskExemptVaultSet(vault, exempt);
    }
 
    /* -------------------- Core action: disburse ----------------- */
    /**
     * @notice Disburse `amount` VFIDE to `toUser`'s vault with an on-chain memo.
     * @dev    Enforces: recipient is **approved**, vault-only routing, and risk-gate (if enabled).
     *         Requires Sanctum to be system-exempt on VFIDEToken or Sanctum to be a vault.
     */
    function disburse(address toUser, uint256 amount, string calldata memo) external nonReentrant onlyGov {
        if (toUser == address(0)) revert ZeroAddress();
        require(amount > 0, "zero amount");
        require(address(factory) != address(0), "factory unset");
 
        // Enforce that the recipient is **approved** by governance
        if (!approvedRecipient[toUser]) revert NotApprovedRecipient();
 
        // Resolve/create the recipient's vault
        address toVault = factory.ensureVault(toUser);
        require(factory.isVault(toVault), "dest not vault");
 
        // Risk gate (recipient-focused)
        if (riskGateEnabled && !riskExemptVault[toVault]) {
            require(address(seer) != address(0), "seer unset");
            if (blockIfBlocked && seer.isBlocked(toUser)) revert RiskRejected("receiver blocked");
            uint16 s = seer.score(toUser);
            if (s < minReceiverScore) revert RiskRejected("receiver score");
        }
 
        // Transfer (VFIDEToken must allow it: Sanctum is exempt or is a vault)
        require(IERC20(token).transfer(toVault, amount), "transfer failed");
 
        emit CharityDisbursement(toVault, toUser, amount, memo);
 
        // Best-effort ProofLedger log
        if (address(proof) != address(0)) {
            try proof.logEvent(toUser, IProofLedger.EventType.SanctumDonation, keccak256(abi.encode(toUser, amount, memo))) {} catch {}
        }
    }
 
    /* -------------------- Views -------------------------------- */
    function balance() external view returns (uint256) {
        return IERC20(token).balanceOf(address(this));
    }
