// SPDX-License-Identifier: MIT
pragma solidity 0.8.30;
 
import "@openzeppelin/contracts/access/Ownable2Step.sol";
 
/**
 * @title VaultFactory
 * @notice Authoritative registry + creator for per-user vaults (EIP-1167 clones).
 * - One vault per user.
 * - Anyone can call ensureVault(user) to auto-create if missing.
 * - Tracks on-chain registry `isVault[vault] = true` for token-level enforcement.
 */
interface IUserVault {
    function initialize(address owner_, address proofLedger_, address guardianLock_) external;
}
 
contract VaultFactory is Ownable2Step {
    /* ---------------- Settings ---------------- */
    address public vaultImplementation;  // UserVault logic address
    address public proofLedger;          // ProofLedger reference (events/scores)
    address public guardianLock;         // optional (0 to disable)
 
    /* ---------------- Registry ---------------- */
    mapping(address => address) public vaultOf; // user => vault
    mapping(address => bool)    public isVault; // vault => true
    address[] public allVaults;
 
    /* ---------------- Events ------------------ */
    event ImplementationSet(address impl);
    event ProofLedgerSet(address proof);
    event GuardianLockSet(address guardian);
    event VaultCreated(address indexed user, address indexed vault);
 
    /* ---------------- Errors ------------------ */
    error ZeroAddress();
    error ImplementationUnset();
 
    constructor(address vaultImpl_, address proofLedger_, address guardianLock_) {
        if (vaultImpl_ == address(0) || proofLedger_ == address(0)) revert ZeroAddress();
        // SAFETY: ensure the implementation is actually a contract
        uint256 size;
        assembly { size := extcodesize(vaultImpl_) }
        require(size > 0, "impl !contract");
 
        vaultImplementation = vaultImpl_;
        proofLedger = proofLedger_;
        guardianLock = guardianLock_;
        emit ImplementationSet(vaultImpl_);
        emit ProofLedgerSet(proofLedger_);
        emit GuardianLockSet(guardianLock_);
    }
 
    /* --------------- Admin -------------------- */
    function setImplementation(address impl) external onlyOwner {
        if (impl == address(0)) revert ZeroAddress();
        // SAFETY: ensure the implementation is actually a contract
        uint256 size;
        assembly { size := extcodesize(impl) }
        require(size > 0, "impl !contract");
 
        vaultImplementation = impl;
        emit ImplementationSet(impl);
    }
 
    function setProofLedger(address proof) external onlyOwner {
        if (proof == address(0)) revert ZeroAddress();
        proofLedger = proof;
        emit ProofLedgerSet(proof);
    }
 
    function setGuardianLock(address guardian) external onlyOwner {
        guardianLock = guardian; // 0 allowed
        emit GuardianLockSet(guardian);
    }
 
    /* --------------- Create / Ensure ---------- */
    function createMyVault() external returns (address) {
        return _ensureVault(msg.sender);
    }
 
    /// @notice Ensure `user` has a vault; creates if missing. Callable by anyone.
    function ensureVault(address user) external returns (address) {
        return _ensureVault(user);
    }
 
    function _ensureVault(address user) internal returns (address v) {
        if (vaultImplementation == address(0)) revert ImplementationUnset();
        v = vaultOf[user];
        if (v != address(0)) return v;
 
        v = _clone(vaultImplementation);
        IUserVault(v).initialize(user, proofLedger, guardianLock);
        vaultOf[user] = v;
        isVault[v] = true;
        allVaults.push(v);
        emit VaultCreated(user, v);
    }
 
    /* --------------- Views -------------------- */
    function totalVaults() external view returns (uint256) { return allVaults.length; }
 
    /* --------------- EIP-1167 ----------------- */
    function _clone(address impl) internal returns (address instance) {
        assembly {
            let ptr := mload(0x40)
            mstore(ptr, 0x3d602d80600a3d3981f3)
            mstore(add(ptr, 0x14), shl(0x60, impl))
            mstore(add(ptr, 0x28), 0x5af43d82803e903d91602b57fd5bf3)
            instance := create(0, ptr, 0x37)
        }
        require(instance != address(0), "clone failed");
    }
}
