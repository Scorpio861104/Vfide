/* ─────────────────────────── DAO (VFIDE Dual-Control Hybrid) ───────────────────────────
 
Roles & Flow:
- Community makes proposals; votes are ProofScore-weighted (Seer).
- If quorum + pass ratio met → DAO queues the action into DAOTimelock → executes after delay.
- Emergency (pre-handover): developer multisig may queue/execute via timelock within an
  emergency window (expires at emergencyPowersExpireAt). Always breaker-aware.
 
Voting Model:
- Each address’s weight = Seer.score(address).
- Pass conditions:
   1) (for + against) ≥ minTotalParticipationScore
   2) for ≥ passRatioBps% of (for + against)
- Abstain counts toward participation but not for/against ratio (config toggle).
 
Safety:
- Seer gates: voters can vote even if low score, but blocked voters are ignored (toggleable).
- Proposer gate: proposer must meet minScoreProposer (if enabled).
- Breaker: queue/execute/cancel blocked while system paused.
 
All sensitive events are logged in ProofLedger (best-effort).
 
---------------------------------------------------------------------------------------- */

    function state(bytes32 id) public view returns (ProposalState) {
        Proposal storage p = proposals[id];
        if (p.createdAt == 0) return ProposalState.Expired; // not found -> treat as expired
        if (p.canceled) return ProposalState.Canceled;
        if (p.executed) return ProposalState.Executed;
 
        uint64 nowTs = uint64(block.timestamp);
        if (nowTs < p.voteStart) return ProposalState.Pending;
        if (nowTs <= p.voteEnd) return ProposalState.Active;
 
        // voting ended
        bool passed = _passed(p.forVotes, p.againstVotes, p.abstainVotes);
        if (!passed) return ProposalState.Defeated;
 
        if (p.queued) {
            // queued but not executed
            if (nowTs > p.deadline + 30 days) return ProposalState.Expired; // safety
            return ProposalState.Queued;
        }
 
        if (nowTs > p.deadline) return ProposalState.Expired;
        return ProposalState.Succeeded;
    }

    /// @param support 0=Against, 1=For, 2=Abstain
    function castVote(bytes32 id, uint8 support) external breakerGuard nonReentrant {
        Proposal storage p = proposals[id];
        ProposalState st = state(id);
        if (st != ProposalState.Active) revert TooLate();
        if (hasVoted[id][msg.sender]) revert AlreadyVoted();
        require(support <= 2, "bad support");
 
        uint256 weight = _weightOf(msg.sender);
        if (weight == 0) {
            // still record “zero weight” to block repeated attempts
            hasVoted[id][msg.sender] = true;
            emit VoteCast(id, msg.sender, support, 0);
            return;
        }
 
        hasVoted[id][msg.sender] = true;
 
        if (support == 0) {
            p.againstVotes += weight;
        } else if (support == 1) {
            p.forVotes += weight;
        } else {
            p.abstainVotes += weight;
        }
 
        emit VoteCast(id, msg.sender, support, weight);
    }
 
    function _weightOf(address voter) internal view returns (uint256) {
        if (address(seer) == address(0)) return 0;
        if (ignoreBlockedVoters && seer.isBlocked(voter)) return 0;
        return uint256(seer.score(voter)); // 0..1000
    }
 
    function _passed(uint256 forVotes, uint256 againstVotes, uint256 abstainVotes) internal view returns (bool) {
        uint256 total = forVotes + againstVotes + (countAbstainInParticipation ? abstainVotes : 0);
        if (total < minTotalParticipationScore) return false;
        uint256 denom = forVotes + againstVotes;
        if (denom == 0) return false; // all abstained
        return (forVotes * 10_000 / denom) >= passRatioBps;
    }

    function queue(bytes32 id) external breakerGuard nonReentrant {

    }

    function cancel(bytes32 id) external breakerGuard nonReentrant {
        Proposal storage p = proposals[id];
        ProposalState st = state(id);
        if (st == ProposalState.Executed || p.canceled) revert NotCancellable();
 
        // Allow: owner (pre-handover operator), proposer, or dev multisig within emergency window
        bool devActive = (developerMultisig != address(0) && block.timestamp <= emergencyPowersExpireAt && msg.sender == developerMultisig);
        require(msg.sender == owner() || msg.sender == p.proposer || devActive, "no auth");

            timelock.cancel(p.target, p.value, p.data, p.predecessor, p.salt);
        }
 
        p.canceled = true;
 
        emit ProposalCanceled(id);
        _log(236 /*ProposalCanceled*/, keccak256(abi.encode(id, msg.sender)));
    }
