/**
 * @title GuardianLock
 * @notice Authoritative lock layer for VFIDE vault transfers.
 *         - Vaults must call `canTransfer(owner, to, amount)` before moving funds.
 *         - DAO/Owner and trusted modules can freeze/quarantine/release accounts.
 *         - Optional Seer gates provide additional safety.
 *         - Emits clear events and logs to ProofLedger (best-effort).
 *
 * Design notes:
 * - This contract holds *no funds* and has *no token logic*.
 * - PanicGuard still provides hybrid monitoring; GuardianLock is the final "gate".
 * - GuardianRegistry/ChainOfReturn can open/extend quarantines during recovery.
 */
contract GuardianLock is Ownable2Step, ReentrancyGuard {
    /* ───────── Wiring ───────── */
    address public dao;                 // DAO (post-handover)
    IVaultFactoryView public factory;   // authenticate vault callers
    ISeerView public seer;              // optional block/score checks
    IProofLedger public proof;          // optional audit sink
 
    /* ───────── Global controls ───────── */
    bool public globalPause = false;    // emergency stop
 
    /* ───────── Per-user lock state ───────── */
    mapping(address => bool)  public frozen;            // hard freeze (manual, indefinite)
    mapping(address => uint64) public quarantineUntil;  // temporary lock until timestamp (auto or manual)
    mapping(address => bool)  public systemExemptUser;  // skip checks (routers, treasury ops, etc.)
 
    /* ───────── Seer gates (optional) ───────── */
    bool   public seerGatesEnabled = true;
    bool   public blockIfBlocked   = true;
    bool   public enforceMinScore  = false; // off by default; PanicGuard enforces min scores
    uint16 public minScore         = 300;
 
    /* ───────── Destination policy ───────── */
    mapping(address => bool) public allowDestination; // bypass seer checks for this dest (still blocks freeze/pause)
    mapping(address => bool) public denyDestination;  // always blocked destination
 
    /* ───────── Trusted callers ─────────
       Modules allowed to set/extend/release locks besides owner/DAO.
       Examples: GuardianRegistry, PanicGuard, ChainOfReturn.
    ---------------------------------------------------------------- */
    mapping(address => bool) public isTrustedSetter;
 
    /* ───────── Events ───────── */
    event DAOUpdated(address dao);
    event FactorySet(address factory);
    event SeerSet(address seer);
    event ProofSinkSet(address proof);
 
    event GlobalPauseSet(bool paused);
 
    event FrozenSet(address indexed user, bool isFrozen);
    event Quarantined(address indexed user, uint64 untilTs);
    event Released(address indexed user);
 
    event SystemExemptSet(address indexed user, bool exempt);
 
    event SeerGatesToggled(bool enabled, bool blockIfBlocked, bool enforceMinScore, uint16 minScore);
    event DestinationAllowSet(address indexed dest, bool allowed);
    event DestinationDenySet(address indexed dest, bool denied);
 
    event TrustedSetterSet(address indexed module, bool trusted);
 
    /* ───────── Errors ───────── */
    error ZeroAddress();
    error NotGov();
    error NotTrusted();
    error FactoryUnset();
 
    /* ───────── Modifiers ───────── */
    modifier onlyGov() {
        if (msg.sender != owner() && msg.sender != dao) revert NotGov();
        _;
    }
 
    modifier onlyGovOrTrusted() {
        if (msg.sender != owner() && msg.sender != dao && !isTrustedSetter[msg.sender]) revert NotTrusted();
        _;
    }

    /* ───────── Wiring ───────── */
 
    function setDAO(address dao_) external onlyGov {
        if (dao_ == address(0)) revert ZeroAddress();
        dao = dao_;
        emit DAOUpdated(dao_);
    }
 
    function setFactory(address factory_) external onlyGov {
        if (factory_ == address(0)) revert ZeroAddress();
        factory = IVaultFactoryView(factory_);
        emit FactorySet(factory_);
    }
 
    function setSeer(address seer_) external onlyGov {
        if (seer_ == address(0)) revert ZeroAddress();
        seer = ISeerView(seer_);
        emit SeerSet(seer_);
    }
 
    function setProofSink(address proof_) external onlyGov {
        proof = IProofLedger(proof_); // 0 allowed (disable)
        emit ProofSinkSet(proof_);
    }
 
    function setTrustedSetter(address module, bool trusted) external onlyGov {
        if (module == address(0)) revert ZeroAddress();
        isTrustedSetter[module] = trusted;
        emit TrustedSetterSet(module, trusted);
    }
 
    /* ───────── Global / policy ───────── */
 
    function setGlobalPause(bool paused) external onlyGov {
        globalPause = paused;
        emit GlobalPauseSet(paused);
    }
 
    function setSystemExempt(address who, bool exempt) external onlyGov {
        systemExemptUser[who] = exempt;
        emit SystemExemptSet(who, exempt);
    }
 
    function toggleSeerGates(bool enabled, bool blockBlocked, bool enforceMin, uint16 minScore_) external onlyGov {
        require(minScore_ <= 1000, "score>max");
        seerGatesEnabled = enabled;
        blockIfBlocked   = blockBlocked;
        enforceMinScore  = enforceMin;
        minScore         = minScore_;
        emit SeerGatesToggled(enabled, blockBlocked, enforceMin, minScore_);
    }
 
    function setAllowDestination(address dest, bool allowed) external onlyGov {
        allowDestination[dest] = allowed;
        emit DestinationAllowSet(dest, allowed);
    }
 
    function setDenyDestination(address dest, bool denied) external onlyGov {
        denyDestination[dest] = denied;
        emit DestinationDenySet(dest, denied);
    }
 
    /* ───────── Core enforcement (called by vaults BEFORE transfer) ───────── */
    /**
     * @notice View-only gate to decide if a transfer is allowed.
     * @dev    Vaults should `require(guardianLock.canTransfer(owner,to,amount))`.
     */
    function canTransfer(address owner, address to, uint256 /*amount*/) external view returns (bool) {
        // System exemptions: skip all checks for designated system actors
        if (systemExemptUser[owner] || systemExemptUser[to]) return true;
 
        // Global stop
        if (globalPause) return false;
 
        // Destination policy
        if (denyDestination[to]) return false;
 
        // Lock state
        if (frozen[owner]) return false;
        uint64 q = quarantineUntil[owner];
        if (q != 0 && block.timestamp < q) return false;
 
        // Optional Seer gates (defense in depth; PanicGuard also enforces)
        if (seerGatesEnabled && !allowDestination[to]) {
            if (address(seer) == address(0)) return true; // no source; don't block
            if (blockIfBlocked && (seer.isBlocked(owner) || seer.isBlocked(to))) return false;
            if (enforceMinScore) {
                if (seer.score(owner) < minScore) return false;
                if (seer.score(to)    < minScore) return false;
            }
        }
 
        return true;
    }
 
    /* ───────── State mutations (DAO/Owner or trusted modules) ───────── */
 
    /// @notice Indefinite hard freeze (manual).
    function setFrozen(address user, bool isFrozen) external onlyGovOrTrusted nonReentrant {
        frozen[user] = isFrozen;
        if (isFrozen) {
            emit FrozenSet(user, true);
            _log(user, 10 /*FrozenSet*/, keccak256(abi.encode(user, true)));
        } else {
            emit FrozenSet(user, false);
            _log(user, 11 /*FrozenCleared*/, keccak256(abi.encode(user, false)));
        }
    }
 
    /// @notice Set/extend quarantine until a specific timestamp.
    function setQuarantine(address user, uint64 untilTs) external onlyGovOrTrusted nonReentrant {
        if (untilTs <= uint64(block.timestamp)) {
            // treat as release if timestamp is in the past
            quarantineUntil[user] = 0;
            emit Released(user);
            _log(user, 13 /*QuarantineCleared*/, keccak256(abi.encode(user)));
            return;
        }
        quarantineUntil[user] = untilTs;
        emit Quarantined(user, untilTs);
        _log(user, 12 /*Quarantined*/, keccak256(abi.encode(user, untilTs)));
    }
 
    /// @notice Clear both freeze & quarantine states.
    function release(address user) external onlyGovOrTrusted nonReentrant {
        if (frozen[user]) {
            frozen[user] = false;
            emit FrozenSet(user, false);
        }
        if (quarantineUntil[user] != 0) {
            quarantineUntil[user] = 0;
            emit Released(user);
        }
        _log(user, 14 /*Released*/, keccak256(abi.encode(user)));
    }
 
    /* ───────── Vault-auth helper (optional) ───────── */
    /**
     * @notice Convenience check for vaults that don't know the owner upfront.
     * @dev    Authenticates `msg.sender` is a valid vault, resolves its owner, and applies `canTransfer`.
     */
    function canVaultSend(address to, uint256 amount) external view returns (bool) {
        if (address(factory) == address(0)) return false;
        if (!factory.isVault(msg.sender)) return false;
        address owner = factory.ownerOfVault(msg.sender);
        return this.canTransfer(owner, to, amount);
    }
 
    /* ───────── Views ───────── */
    function isQuarantined(address user) external view returns (bool active, uint64 untilTs) {
        uint64 q = quarantineUntil[user];
        return (q != 0 && block.timestamp < q, q);
    }
 
    /* ───────── Internal logging ───────── */
    function _log(address user, uint8 etype, bytes32 ctx) internal {
        if (address(proof) != address(0)) {
            try proof.logEvent(user, etype, ctx) {} catch {}
        }
    }
