pragma solidity 0.8.30;
 
import "@openzeppelin/contracts/access/Ownable2Step.sol";
import "@openzeppelin/contracts/security/ReentrancyGuard.sol";
 
/**
 * @title ProofLedger (Core of Trust Intelligence)
 * @notice Records immutable behavioral events and maintains ProofScores for VFIDE users.
 * @dev    Called by Seer, Sanctum, BurnRouter, Presale, Vaults, GuardianLock, etc.
 *         Exposes a single ABI-stable logger: logEvent(address user, uint8 etype, bytes32 ctx).
 *         Enums in other contracts compile to uint8, so they interoperate cleanly here.
 */
contract ProofLedger is Ownable2Step, ReentrancyGuard {
    /* -------------------------------------------------------------------------- */
    /*                                    TYPES                                   */
    /* -------------------------------------------------------------------------- */
 
    /// @notice Canonical risk view for dashboards/UIs (aligned with Seer’s 0..1000 scale)
    enum RiskLevel { Critical, Weak, Neutral, Strong, Fortress }
 
    /// @notice Append-only event entry for auditability
    struct Entry {
        uint64  ts;        // block.timestamp (seconds)
        uint8   etype;     // arbitrary event type code (from caller’s enum)
        int16   delta;     // score delta applied for this event
        uint16  score;     // resulting score after applying delta (clamped 0..1000)
        bytes32 ctx;       // compact context hash (e.g., off-chain memo key)
    }
 
    /* -------------------------------------------------------------------------- */
    /*                                   STORAGE                                  */
    /* -------------------------------------------------------------------------- */
 
    // Score book (0..1000)
    mapping(address => uint16) private _score;
 
    // Per-user append-only event log
    mapping(address => Entry[]) private _events;
 
    // Governance: DAO executor (post-handover). Before that, owner runs controls.
    address public dao;
 
    // Reporters allowed to call logEvent (Seer, Sanctum, BurnRouter, Presale, Vaults, GuardianLock, etc.)
    mapping(address => bool) public authorizedReporter;
 
    // Per-event-type score weights (etype => delta). Defaults to 0 unless set.
    mapping(uint8 => int16) public weightOf;
 
    // Global clamps
    uint16 public constant MAX_SCORE = 1000;
    uint16 public minClamp = 0;      // lower bound (usually 0)
    uint16 public maxClamp = 1000;   // upper bound (usually 1000)
 
    /* -------------------------------------------------------------------------- */
    /*                                    EVENTS                                  */
    /* -------------------------------------------------------------------------- */
 
    event DAOUpdated(address indexed dao);
    event ReporterAuthorized(address indexed reporter, bool authorized);
    event WeightSet(uint8 indexed etype, int16 delta);
    event ClampsSet(uint16 minClamp, uint16 maxClamp);
    event ScoreSet(address indexed user, uint16 newScore); // emergency/governance
    event Logged(address indexed user, uint8 indexed etype, int16 delta, uint16 newScore, bytes32 ctx);
 
    /* -------------------------------------------------------------------------- */
    /*                                   ERRORS                                   */
    /* -------------------------------------------------------------------------- */
 
    error ZeroAddress();
    error NotGov();
    error NotReporter();
    error BadClamp();
 
    /* -------------------------------------------------------------------------- */
    /*                                 MODIFIERS                                  */
    /* -------------------------------------------------------------------------- */
 
    modifier onlyGov() {
        if (msg.sender != owner() && msg.sender != dao) revert NotGov();
        _;
    }
 
    modifier onlyReporter() {
        if (!authorizedReporter[msg.sender]) revert NotReporter();
        _;
    }
 
    /* -------------------------------------------------------------------------- */
    /*                                CONSTRUCTOR                                 */
    /* -------------------------------------------------------------------------- */
 
    constructor(address initialOwner) {
        Iif (initialOwner == address(0)) revert ZeroAddress();
        _transferOwnership(initialOwner);
 
        // bootstrap: deployer is an authorized reporter until DAO + modules are set
        authorizedReporter[msg.sender] = true;
        emit ReporterAuthorized(msg.sender, true);
    }
 
    /* -------------------------------------------------------------------------- */
    /*                               GOVERNANCE OPS                               */
    /* -------------------------------------------------------------------------- */
 
    /// @notice Set the DAO executor (post-handover governance)
    function setDAO(address dao_) external onlyGov {
        if (dao_ == address(0)) revert ZeroAddress();
        dao = dao_;
        emit DAOUpdated(dao_);
    }
 
    /// @notice Authorize or revoke a module to write events (Seer, Sanctum, Routers, etc.)
    function setReporter(address reporter, bool authorized) external onlyGov {
        if (reporter == address(0)) revert ZeroAddress();
        authorizedReporter[reporter] = authorized;
        emit ReporterAuthorized(reporter, authorized);
    }
 
    /// @notice Batch set reporters for convenience.
    function setReporters(address[] calldata reporters, bool authorized) external onlyGov {
        for (uint256 i = 0; i < reporters.length; i++) {
            address r = reporters[i];
            if (r == address(0)) continue;
            authorizedReporter[r] = authorized;
            emit ReporterAuthorized(r, authorized);
        }
    }
 
    /// @notice Set score weight for a given event type code (etype).
    /// @dev     Positive = reward, Negative = penalty.
    function setWeight(uint8 etype, int16 delta) external onlyGov {
        weightOf[etype] = delta;
        emit WeightSet(etype, delta);
    }
 
    /// @notice Batch set weights to make seeding cheaper.
    function setWeights(uint8[] calldata etypes, int16[] calldata deltas) external onlyGov {
        require(etypes.length == deltas.length, "len mismatch");
        for (uint256 i = 0; i < etypes.length; i++) {
            weightOf[etypes[i]] = deltas[i];
            emit WeightSet(etypes[i], deltas[i]);
        }
    }
 
    /// @notice Set global clamp bounds (inclusive). Must respect 0..MAX_SCORE and min<=max.
    function setClamps(uint16 minClamp_, uint16 maxClamp_) external onlyGov {
        if (minClamp_ > maxClamp_) revert BadClamp();
        require(maxClamp_ <= MAX_SCORE, "max>MAX_SCORE");
        minClamp = minClamp_;
        maxClamp = maxClamp_;
        emit ClampsSet(minClamp, maxClamp);
    }
 
    /// @notice Emergency: directly set a user’s score (e.g., migration/repair). Emits an event.
    function setScore(address user, uint16 newScore) external onlyGov {
        if (user == address(0)) revert ZeroAddress();
        require(newScore <= MAX_SCORE, "score>MAX");
        _score[user] = _clamp(newScore);
        emit ScoreSet(user, _score[user]);
    }
 
    /* -------------------------------------------------------------------------- */
    /*                                CORE LOGGING                                */
    /* -------------------------------------------------------------------------- */
 
    /**
     * @notice Log a behavior event for `user` with compact context `ctx`.
     * @param user  The subject wallet (EOA controller address, not the vault)
     * @param etype Arbitrary event type code (uint8). Callers may use their own enums.
     * @param ctx   Context hash (e.g., keccak of structured data; off-chain systems can index it)
     *
     * Effects:
     *  - Applies the configured score delta for `etype` (weightOf[etype]) to user’s score (clamped).
     *  - Appends an immutable Entry to the user’s event log.
     *
     * Access:
     *  - Only authorized reporters (Seer, Sanctum, BurnRouter, Presale, Vaults, GuardianLock, etc.)
     */
    function logEvent(address user, uint8 etype, bytes32 ctx) external nonReentrant onlyReporter {
        if (user == address(0)) revert ZeroAddress();
 
        // Compute new score
        uint16 prev = _score[user];
        int256 next = int256(uint256(prev)) + int256(weightOf[etype]);
        if (next < int256(uint256(minClamp))) next = int256(uint256(minClamp));
        if (next > int256(uint256(maxClamp))) next = int256(uint256(maxClamp));
        uint16 newScore = uint16(uint256(next));
 
        // Persist score + entry
        _score[user] = newScore;
        _events[user].push(
            Entry({
                ts: uint64(block.timestamp),
                etype: etype,
                delta: weightOf[etype],
                score: newScore,
                ctx: ctx
            })
        );
 
        emit Logged(user, etype, weightOf[etype], newScore, ctx);
    }
 
    /* -------------------------------------------------------------------------- */
    /*                                   VIEWS                                    */
    /* -------------------------------------------------------------------------- */
 
    /// @notice Canonical live score (0..1000). Seer reads this via IProofLedgerView.
    function currentScore(address user) external view returns (uint16) {
        return _score[user];
    }
 
    /// @notice Alias for compatibility with older interfaces
    function scoreOf(address user) external view returns (uint16) {
        return _score[user];
    }
 
    /// @notice Read a single event entry by index
    function eventOf(address user, uint256 index) external view returns (Entry memory) {
        return _events[user][index];
    }
 
    /// @notice Return all events for a user (beware of gas for very large histories)
    function eventsOf(address user) external view returns (Entry[] memory) {
        return _events[user];
    }
 
    /// @notice A compact risk band for dashboards (aligned to Seer defaults)
    function riskLevelOf(address user) external view returns (RiskLevel) {
        uint16 s = _score[user];
        if (s >= 900) return RiskLevel.Fortress;
        if (s >= 700) return RiskLevel.Strong;
        if (s >= 500) return RiskLevel.Neutral;
        if (s >= 300) return RiskLevel.Weak;
        return RiskLevel.Critical;
    }
 
    /// @notice Total number of logged entries for `user`
    function totalEvents(address user) external view returns (uint256) {
        return _events[user].length;
    }
 
    /* -------------------------------------------------------------------------- */
    /*                                  INTERNALS                                 */
    /* -------------------------------------------------------------------------- */
 
    function _clamp(uint16 v) internal view returns (uint16) {
        if (v < minClamp) return minClamp;
        if (v > maxClamp) return maxClamp;
        return v;
    }
}
